1   /**
  2    * @name $.simplicityGeoFn
  3    * @namespace Contains various geographic related utility functions.
  4    */
  5 (function ($) {
  6   var radiusMiles = 3956;
  7   var radiusKm = 6367;
  8   var geoFn = {
  9     debug: false
 10   };
 11   $.simplicityGeoFn = geoFn;
 12   $.simplicityGeoFn.degreesToRadians = function (v) {
 13     return v * Math.PI / 180.0;
 14   };
 15   $.simplicityGeoFn.radiansToDegrees = function (v) {
 16     return v * 180 / Math.PI;
 17   };
 18   $.simplicityGeoFn.distanceRadians = function (lat1, lng1, lat2, lng2) {
 19     lat1 = geoFn.degreesToRadians(lat1);
 20     lng1 = geoFn.degreesToRadians(lng1);
 21     lat2 = geoFn.degreesToRadians(lat2);
 22     lng2 = geoFn.degreesToRadians(lng2);
 23     var dLat = lat2 - lat1;
 24     var dLng = lng2 - lng1;
 25     var a = Math.sin(dLat / 2) * Math.sin(dLat / 2) +
 26       Math.cos(lat1) * Math.cos(lat2) * Math.sin(dLng / 2) * Math.sin(dLng / 2);
 27     var c = 2 * Math.atan2(Math.sqrt(a), Math.sqrt(1 - a));
 28     return c;
 29   };
 30   /**
 31    * Calculates the distance between two coordinates in miles.
 32    *
 33    * @param lat1
 34    * @param lng1
 35    * @param lat2
 36    * @param lng2
 37    */
 38   $.simplicityGeoFn.distanceMiles = function (lat1, lng1, lat2, lng2) {
 39     return radiusMiles * geoFn.distanceRadians(lat1, lng1, lat2, lng2);
 40   };
 41   /**
 42    * Calculates the distance between two coordinates in kilometers.
 43    *
 44    * @param lat1
 45    * @param lng1
 46    * @param lat2
 47    * @param lng2
 48    */
 49   $.simplicityGeoFn.distanceKm = function (lat1, lng1, lat2, lng2) {
 50     return radiusKm * geoFn.distanceRadians(lat1, lng1, lat2, lng2);
 51   };
 52   /**
 53    * Returns the (initial) heading from this point to the supplied point, in degrees.
 54    *
 55    * @see <a href="http://williams.best.vwh.net/avform.htm#Crs">http://williams.best.vwh.net/avform.htm#Crs</a>
 56    * @param   [{Number} lat, {Number} lng] p1: First point
 57    * @param   [{Number} lat, {Number} lng] p2: Second point
 58    * @returns {Number} Initial heading in degrees from North
 59    */
 60   $.simplicityGeoFn.computeHeading = function (p1, p2) {
 61     var dLonr = geoFn.degreesToRadians(p2[1] - p1[1]);
 62     var lat1r = geoFn.degreesToRadians(p1[0]);
 63     var lat2r = geoFn.degreesToRadians(p2[0]);
 64     var br = Math.atan2(
 65         Math.sin(dLonr) * Math.cos(lat2r),
 66         Math.cos(lat1r) * Math.sin(lat2r) - Math.sin(lat1r) * Math.cos(lat2r) * Math.cos(dLonr));
 67     var result = geoFn.radiansToDegrees(br);
 68     if (result < 0) {
 69       result += 360;
 70     }
 71     return result;
 72   };
 73   /**
 74    * Returns the destination point from a point having traveled the given distance (in km) on the
 75    * given initial bearing (bearing may vary before destination is reached).
 76    *
 77    * @see <a href="http://williams.best.vwh.net/avform.htm#LL">http://williams.best.vwh.net/avform.htm#LL</a>
 78    * @param   [{Number} lat, {Number} lng] p: Point from which to start traveling
 79    * @param   {Number} dist: Distance in meters
 80    * @param   {Number} heading: Initial heading in degrees
 81    * @returns [{Number} lat, {Number} lng] Destination point
 82    */
 83   $.simplicityGeoFn.travel = function (p, dist, head) {
 84     var d = dist / 1000 / radiusKm; // convert dist to km then to angular distance in radians
 85     var h = geoFn.degreesToRadians(head);
 86     var lat1 = geoFn.degreesToRadians(p[0]);
 87     var lon1 = geoFn.degreesToRadians(p[1]);
 88     var lat2 = Math.asin(Math.sin(lat1) * Math.cos(d) +
 89                  Math.cos(lat1) * Math.sin(d) * Math.cos(h));
 90     var lon2 = lon1 + Math.atan2(Math.sin(h) * Math.sin(d) * Math.cos(lat1),
 91                  Math.cos(d) - Math.sin(lat1) * Math.sin(lat2));
 92     lon2 = (lon2 + 3 * Math.PI) % (2 * Math.PI) - Math.PI; // normalize to -180..+180º
 93     return [geoFn.radiansToDegrees(lat2), geoFn.radiansToDegrees(lon2)];
 94   };
 95   /**
 96    * Returns the point of intersection of two paths defined by point and heading.
 97    *
 98    * @see <a href="http://www.movable-type.co.uk/scripts/latlong.html">http://www.movable-type.co.uk/scripts/latlong.html</a>
 99    *      <a href="http://williams.best.vwh.net/avform.htm#Intersection">http://williams.best.vwh.net/avform.htm#Intersection</a>
100    * @param   [{Number} lat, {Number} lng] p1: First point
101    * @param   {Number} heading1: Initial heading from first point
102    * @param   [{Number} lat, {Number} lng] p2: Second point
103    * @param   {Number} heading2: Initial heading from second point
104    */
105   $.simplicityGeoFn.intersection = function (p1, heading1, p2, heading2) {
106     var lat1 = geoFn.degreesToRadians(p1[0]);
107     var lon1 = geoFn.degreesToRadians(p1[1]);
108     var lat2 = geoFn.degreesToRadians(p2[0]);
109     var lon2 = geoFn.degreesToRadians(p2[1]);
110     var heading13 = geoFn.degreesToRadians(heading1);
111     var heading23 = geoFn.degreesToRadians(heading2);
112     var dLat = lat2 - lat1;
113     var dLon = lon2 - lon1;
114 
115     var dist12 = 2 * Math.asin(Math.sqrt(Math.sin(dLat / 2) * Math.sin(dLat / 2) +
116       Math.cos(lat1) * Math.cos(lat2) * Math.sin(dLon / 2) * Math.sin(dLon / 2)));
117     if (dist12 === 0) {
118       if (geoFn.debug) {
119         console.log('dist12', 0);
120       }
121       return null;
122     }
123 
124     // initial/final bearings between points
125     var headingA = Math.acos((Math.sin(lat2) - Math.sin(lat1) * Math.cos(dist12)) /
126       (Math.sin(dist12) * Math.cos(lat1)));
127     if (isNaN(headingA)) {
128       headingA = 0; // protect against rounding
129     }
130     var headingB = Math.acos((Math.sin(lat1) - Math.sin(lat2) * Math.cos(dist12)) /
131       (Math.sin(dist12) * Math.cos(lat2)));
132 
133     var heading12, heading21;
134     if (Math.sin(lon2 - lon1) > 0) {
135       heading12 = headingA;
136       heading21 = 2 * Math.PI - headingB;
137     } else {
138       heading12 = 2 * Math.PI - headingA;
139       heading21 = headingB;
140     }
141 
142     var alpha1 = (heading13 - heading12 + Math.PI) % (2 * Math.PI) - Math.PI; // angle 2-1-3
143     var alpha2 = (heading21 - heading23 + Math.PI) % (2 * Math.PI) - Math.PI; // angle 1-2-3
144 
145     if (Math.sin(alpha1) === 0 && Math.sin(alpha2) === 0) {
146       if (geoFn.debug) {
147         console.log('Infinite intersections');
148       }
149       return null;// infinite intersections
150     }
151     if (Math.sin(alpha1) * Math.sin(alpha2) < 0) {
152       if (geoFn.debug) {
153         console.log('Ambiguous intersection');
154       }
155       return null;  // ambiguous intersection
156     }
157 
158     //alpha1 = Math.abs(alpha1);
159     //alpha2 = Math.abs(alpha2);
160     // ... Ed Williams takes abs of alpha1/alpha2, but seems to break calculation?
161 
162     var alpha3 = Math.acos(-Math.cos(alpha1) * Math.cos(alpha2) +
163                          Math.sin(alpha1) * Math.sin(alpha2) * Math.cos(dist12));
164     var dist13 = Math.atan2(Math.sin(dist12) * Math.sin(alpha1) * Math.sin(alpha2),
165                          Math.cos(alpha2) + Math.cos(alpha1) * Math.cos(alpha3));
166     var lat3 = Math.asin(Math.sin(lat1) * Math.cos(dist13) +
167                       Math.cos(lat1) * Math.sin(dist13) * Math.cos(heading13));
168     var dLon13 = Math.atan2(Math.sin(heading13) * Math.sin(dist13) * Math.cos(lat1),
169                          Math.cos(dist13) - Math.sin(lat1) * Math.sin(lat3));
170     var lon3 = lon1 + dLon13;
171     lon3 = (lon3 + 3 * Math.PI) % (2 * Math.PI) - Math.PI; // normalize to -180..+180º
172 
173     return [geoFn.radiansToDegrees(lat3), geoFn.radiansToDegrees(lon3)];
174   };
175 
176   // For backward compatibility
177   $.simplicityHaversineRadiusMiles = radiusMiles;
178   $.simplicityHaversineRadiusKm = radiusKm;
179   $.simplicityHaversineDistanceRadians = geoFn.distanceRadians;
180   $.simplicityHaversineDistanceMiles = geoFn.distanceMiles;
181   $.simplicityHaversineDistanceKm = geoFn.distanceKm;
182 }(jQuery));
