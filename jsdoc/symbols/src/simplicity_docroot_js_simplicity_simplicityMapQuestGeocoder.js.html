1 /**
  2  * @name $.ui.simplicityMapQuestGeocoder
  3  * @namespace A MapQuest geocoder.
  4  * <p>
  5  * MapQuest geocoder widget. Also provides a jQueri UI autocomplete source.
  6  * <p>
  7  * See the MapQuest JavaScript SDK v6 Geocoding <a href="http://platform.beta.mapquest.com/sdk/js/v6.0.0/geocoding.html">documentation</a>.
  8  */
  9 (function ($) {
 10   $.widget("ui.simplicityMapQuestGeocoder", $.ui.simplicityWidget, {
 11     /**
 12      * Widget options.
 13      *
 14      * <dl>
 15      *   <dt>geocodeOptions</dt>
 16      *   <dd>
 17      *     Options passed to the MapQuest geocode function.
 18      *     Defaults to <code>{thumbMaps: false}</code>.
 19      *   </dd>
 20      *   <dt>disambiguateResults</dt>
 21      *   <dd>
 22      *     When <code>true</code> post-processes the normalized items to look for
 23      *     any duplicates and alter their <code>value</code>s to allow for
 24      *     disambiguation. Defaults to <code>true</code>.
 25      *   </dd>
 26      *   <dt>debug</dt>
 27      *   <dd>
 28      *     Enable logging of key events to console.log. Defaults to <code>false</code>.
 29      *   </dd>
 30      * </dl>
 31      * @name $.ui.simplicityMapQuestGeocoder.options
 32      */
 33     options : {
 34       geocodeOptions: {
 35         thumbMaps: false
 36       },
 37       disambiguateResults: true,
 38       debug: false
 39     },
 40     _create: function () {
 41       this._addClass('ui-simplicity-mapquest-geocoder');
 42     },
 43     /**
 44      * Perform geocoding of the <code>geocodeRequest</code> calling the
 45      * <code>callback</code> on completion.
 46      * <p>
 47      * Triggers two events.
 48      * <dl>
 49      *   <dt><code>request</code></dt>
 50      *   <dd>
 51      *     Final step before sending the geocode request, it is passed
 52      *     as the custom object to this event to allow for manipulation.
 53      *   </dd>
 54      *   <dt><code>response</code></dt>
 55      *   <dd>
 56      *     Final step before calling the <code>callback</code> with the
 57      *     geocode response to allow for manipulation.
 58      *     Triggered after any normalization phase.
 59      *     <p>
 60      *     The custom object passed to this event looks like so:
 61      *     <pre>
 62      *     {
 63      *       vendor: {
 64      *         // Original vendor response
 65      *       },
 66      *       items: [
 67      *         // Array of normalized items, see <code>normalizeResults</code>.
 68      *       ]
 69      *     }
 70      *     </pre>
 71      *   </dd>
 72      * </dl>
 73      *
 74      * @param geocodeRequest
 75      *   The request as expected by the upstream geocode vendor.
 76      * @param callback
 77      *   The callback triggered at geocode completion. Is give one
 78      *   argument which is the same as custom <code>Object</code>
 79      *   passed to the <code>response</code> event.
 80      *
 81      * @name $.ui.simplicityMapQuestGeocoder.geocode
 82      * @function
 83      */
 84     geocode: function (locations, callback) {
 85       var geocodeRequest = {
 86         locations: locations,
 87         options: $.extend({}, this.options.geocodeOptions)
 88       };
 89       this._trigger('request', {}, geocodeRequest);
 90       if (this.options.debug) {
 91         console.log('simplicityMapQuestGeocoder: request', geocodeRequest);
 92       }
 93       MQA.Geocoder.geocode(geocodeRequest.locations, geocodeRequest.options, null, $.proxy(function (geocodeResponse) {
 94         var items = this.normalizeResults(geocodeResponse);
 95         var response = {
 96           vendor: geocodeResponse,
 97           items: items
 98         };
 99         this._trigger('response', {}, response);
100         if (this.options.debug) {
101           console.log('simplicityMapQuestGeocoder: response', response);
102         }
103         callback(response);
104       }, this));
105     },
106     /**
107      * Create an autocomplete source function that returns normalized
108      * geocoded addresses.
109      *
110      * @name $.ui.simplicityMapQuestGeocoder.autocompleteSource
111      * @function
112      */
113     autocompleteSource: function () {
114       return $.proxy(function (request, responseCallback) {
115         this.geocode(request.term, function (response) {
116           responseCallback(response.items);
117         });
118       }, this);
119     },
120     /**
121      * Normalize the geocoder response. Returns an array of items.
122      * <pre>
123      *   Example output
124      *   [
125      *     {
126      *       value: 'Statue of Liberty, New York, NY 11231, USA',
127      *       latitude: 40.6892437,
128      *       longitude: -74.0445142,
129      *       vendor: {
130      *         // Upstream vendor response for this single location
131      *       }
132      *     }
133      *   ]
134      * </pre>
135      *
136      * @name $.ui.simplicityMapQuestGeocoder.normalizeResults
137      * @function
138      * @private
139      */
140     normalizeResults: function (response) {
141       var items = [];
142       if (response.info.statuscode === 0 && $.isArray(response.results)) {
143         $.each(response.results, $.proxy(function (i, result) {
144           $.each(result.locations, $.proxy(function (i, location) {
145             var value = this.normalizeAddress(location);
146             if (Boolean(value)) {
147               var item = {
148                 value: value,
149                 vendor: location
150               };
151               if (typeof location.latLng !== 'undefined') {
152                 $.extend(item, {
153                   latitude: location.latLng.lat,
154                   longitude: location.latLng.lng
155                 });
156               }
157               items.push(item);
158             }
159           }, this));
160         }, this));
161         if (this.options.disambiguateResults) {
162           items = this.disambiguateItems(items);
163         }
164       }
165       return items;
166     },
167     /**
168      * Normalize a single geocoded address. This gets injected
169      * as the <code>value</code> of the <code>item</code> object.
170      *
171      * @name $.ui.simplicityMapQuestGeocoder.normalizeAddress
172      * @function
173      * @private
174      */
175     normalizeAddress: function (location) {
176       var value = '';
177       if (typeof location !== 'undefined') {
178         value = location.street;
179         if (location.adminArea5 !== '') {
180           if (value !== '') {
181             value += ', ';
182           }
183           value += location.adminArea5;
184         }
185         if (location.adminArea4 !== '' && location.geocodeQuality === 'COUNTY') {
186           if (value !== '') {
187             value += ', ';
188           }
189           value += location.adminArea4;
190         }
191         if (location.adminArea3 !== '') {
192           if (value !== '') {
193             value += ', ';
194           }
195           value += location.adminArea3;
196         }
197         if (location.postalCode !== '') {
198           if (value !== '') {
199             value += ' ';
200           }
201           value += location.postalCode;
202         }
203       }
204       return value;
205     },
206     disambiguateItems: function (items) {
207       labelToItem = {};
208       $.each(items, function (i, item) {
209         var suggestionLabel = item.value;
210         var duplicateItem = labelToItem[suggestionLabel];
211         if (typeof duplicateItem === 'undefined') {
212           if (item.vendor.geocodeQuality === 'STATE') {
213             if (item.vendor.adminArea1 === 'US') {
214               suggestionLabel = 'State of ' + suggestionLabel;
215             }
216           }
217           labelToItem[suggestionLabel] = item;
218         } else {
219           // Duplicate label -- disambiguate via County
220           suggestionLabel += ' (' + item.vendor.adminArea4 + ')';
221           var labelToDisambiguate = duplicateItem.label;
222           labelToDisambiguate += ' (' + duplicateItem.vendor.adminArea4 + ')';
223           duplicateItem.label = labelToDisambiguate;
224         }
225         item.label = suggestionLabel;
226       });
227       return items;
228     }
229   });
230 }(jQuery));
