1 /**
  2  * @name $.ui.simplicityGoogleMapResults
  3  * @namespace A Google map.
  4  * <p>
  5  * Widget that listens for <code>simplicityResultSet</code> events which it uses to add markers to the map for the search results.
  6  *
  7  * @example
  8  *   <div id="map" style="width: 300px; height: 300px;"></div>
  9  *   <script type="text/javascript">
 10  *     $('#map').simplicityGoogleMapResults();
 11  *   </script>
 12  *
 13  * @see Google Maps JavaScript API V3 <a href="https://developers.google.com/maps/documentation/javascript/reference">documentation</a>.
 14  */
 15 (function ($) {
 16   $.widget("ui.simplicityGoogleMapResults", $.ui.simplicityWidget, {
 17     /**
 18      * Widget options.
 19      *
 20      * <dl>
 21      *   <dt>searchElement</dt>
 22      *   <dd>
 23      *     The simplicityDiscoverySearch widget that this widget binds it's events to. Defaults to <code>'body'</code>.
 24      *   </dd>
 25      *   <dt>latitudeField</dt>
 26      *   <dd>
 27      *     Field to find the latitude of the result item in the <code>simplicityResultSet</code>
 28      *     item properties. Defaults to <code>'latitude'</code>.
 29      *   </dd>
 30      *   <dt>longitudeField</dt>
 31      *   <dd>
 32      *     Field to find the longitude of the result item in the <code>simplicityResultSet</code>
 33      *     item properties. Defaults to <code>'longitude'</code>.
 34      *   </dd>
 35      *   <dt>map</dt>
 36      *   <dd>
 37      *     Optional map instance, if not provided one will be looked up. Defaults to <code>''</code>.
 38      *   </dd>
 39      *   <dt>updateBounds</dt>
 40      *   <dd>
 41      *     Whether or not the map bounds should be updated to include the result locations. Defaults to true.
 42      *   </dd>
 43      *   <dt>markerType</dt>
 44      *   <dd>
 45      *     Optional marker type, either "overlay" or "google". Defaults to <code>overlay</code>.
 46      *   </dd>
 47      * </dl>
 48      * @name $.ui.simplicityGoogleMapResults.options
 49      */
 50     options: {
 51       searchElement: 'body',
 52       latitudeField: 'latitude',
 53       longitudeField: 'longitude',
 54       updateBounds: true,
 55       map: '',
 56       markerType: 'overlay'
 57     },
 58     _create: function () {
 59       this._addClass('ui-simplicity-google-map-results');
 60       this._markers = [];
 61       this._map = this.options.map !== '' ? this.options.map : this.element.simplicityGoogleMap('map');
 62       this
 63         ._bind(this.options.searchElement, 'simplicitySearchResponse', this._resultSetHandler)
 64         ._bind('simplicitygooglemapboundscoordinatorcalculatebounds', this._calcBoundsHandler);
 65     },
 66     /**
 67      * Override of <code>_setOption</code> that is used to ensure that the
 68      * map is refreshed when the <code>markerType</code> changes.
 69      *
 70      * @name $.ui.simplicityGoogleMapResults._setOption
 71      * @function
 72      * @private
 73      */
 74     _setOption: function (option, value) {
 75       $.ui.simplicityWidget.prototype._setOption.apply(this, arguments);
 76       if ('markerType' === option) {
 77         this.refreshMap();
 78       }
 79     },
 80     /**
 81      * Return the actual map object.
 82      *
 83      * @name $.ui.simplicityGoogleMapResults.map
 84      * @function
 85      */
 86     map: function () {
 87       return this._map;
 88     },
 89     /**
 90      * Makes the widget re-handle the last <code>simplicitySearchResponse</code> event to reapply
 91      * any map markers.
 92      *
 93      * @name $.ui.simplicityGoogleMapResults.refreshMap
 94      * @function
 95      */
 96     refreshMap: function () {
 97       this.removeMarkers();
 98       this.addMarkers();
 99     },
100     /**
101      * Event handler for the <code>simplicitySearchResponse</code> event. Extracts the coordinates
102      * of each result item by using the property fields defined by the
103      * <code>latitudeField</code> and <code>longitudeField</code> options of this widget and
104      * places a marker on the map for each valid coordinate. The map is then reset to best
105      * display the current set of markers.
106      *
107      * @name $.ui.simplicityGoogleMapResults._resultSetHandler
108      * @function
109      * @private
110      */
111     _resultSetHandler: function (evt, searchResponse) {
112       this.removeMarkers();
113       this.addMarkers(searchResponse);
114     },
115     _calcBoundsHandler: function (evt, ui) {
116       var bounds = ui.bounds;
117       if ('undefined' !== typeof bounds && this.options.updateBounds) {
118         $.each(this._markers, function (idx, marker) {
119           bounds.extend(marker.getPosition());
120         });
121       }
122     },
123     /**
124      * Removes any markers that were added to the map by <code>addMarkers</code>.
125      *
126      * @name $.ui.simplicityGoogleMapResults.removeMarkers
127      * @function
128      * @private
129      */
130     removeMarkers: function () {
131       $.each(this._markers, $.proxy(function (idx, marker) {
132         var eventData = {
133           map: this._map,
134           marker: marker
135         };
136         this._trigger('removemarker', {}, eventData);
137         marker = eventData.marker;
138         if ('undefined' !== typeof marker) {
139           marker.setMap(null);
140         }
141       }, this));
142       this._markers.length = 0;
143     },
144     /**
145      * Adds any markers that can be extracted from the given <code>searchResponse</code>.
146      *
147      * @name $.ui.simplicityGoogleMapResults.addMarkers
148      * @function
149      * @private
150      */
151     addMarkers: function (searchResponse) {
152       if ('undefined' === typeof searchResponse) {
153         searchResponse = $(this.options.searchElement).simplicityDiscoverySearch('searchResponse');
154       }
155       var markerFactory = $.simplicityGoogleMarker[this.options.markerType === 'google' ? 'createMarker' : 'createOverlayMarker'];
156       $.fn.simplicityDiscoverySearchItemEnumerator(searchResponse, $.proxy(function (idx, row) {
157         var properties = row.properties;
158         if ('undefined' !== typeof properties) {
159           var latitude = properties[this.options.latitudeField];
160           var longitude = properties[this.options.longitudeField];
161           if ('undefined' !== typeof latitude && 'undefined' !== typeof longitude) {
162             var marker = markerFactory(
163             {
164               position: new google.maps.LatLng(latitude, longitude),
165               row: row,
166               html: row.index1,
167               zIndex: -row.index1
168             });
169             var eventData = {
170               row: row,
171               map: this._map,
172               marker: marker
173             };
174             this._trigger('marker', {}, eventData);
175             marker = eventData.marker;
176             if ('undefined' !== typeof marker) {
177               marker.setMap(this._map);
178               this._markers.push(marker);
179             }
180           }
181         }
182       }, this));
183     }
184   });
185 }(jQuery));
