1 /**
  2  * @name $.ui.simplicitySlider
  3  * @namespace Slider bound to an input per handle.
  4  * <p>
  5  * Creates a progressive enhanced jQuery UI slider that is backed by one <code>input</code> element
  6  * per handle.
  7  * <p>
  8  * The widget options are passed through to the created slider. For more information see the
  9  * jQuery UI slider <a href="http://jqueryui.com/demos/slider/">documentation</a>.
 10  *
 11  * @example
 12  *   Single-handled slider
 13  *   <input id="weight" name="w" />
 14  *   <div id="weightSlider"></div>
 15  *   <script type="text/javascript">
 16  *     $('#weightSlider').simplicitySlider({
 17  *       input: '#weight',
 18  *       min: 0,
 19  *       max: 10,
 20  *       any: 0
 21  *     });
 22  *   </script>
 23  *
 24  * @example
 25  *   Double-handled slider
 26  *   <input id="minPrice" name="minP" />
 27  *   <input id="maxPrice" name="maxP" />
 28  *   <div id="priceSlider"></div>
 29  *   <script type="text/javascript">
 30  *     $('#priceSlider').simplicitySlider({
 31  *       input: ['#minPrice', '#maxPrice'],
 32  *       min: 0,
 33  *       max: 100,
 34  *       values: [0, 100],
 35  *       any: [0, 100]
 36  *     });
 37  *   </script>
 38  */
 39 (function ($) {
 40   $.widget("ui.simplicitySlider", $.ui.simplicityWidget, {
 41     /**
 42      * Widget options.
 43      *
 44      * <dl>
 45      *   <dt>input</dt>
 46      *   <dd>
 47      *     A single <code>input</code> or an array of two <code>input</code>s to associate the
 48      *     slider handles with.
 49      *   </dd>
 50      *   <dt>changeOnSlide</dt>
 51      *   <dd>
 52      *      Set this to <code>true</code> to cause <code>slide</code> events to change the
 53      *      bound <code>input</code>. Defaults to <code>false</code>.
 54      *   </dd>
 55      *   <dt>any</dt>
 56      *   <dd>
 57      *     Single value (or array of two values for a double-handled slider) which represents
 58      *     slider values that are mapped to the empty string (not selected). Usually one end
 59      *     of the slider acts in this manner.
 60      *   </dd>
 61      *   <dt>handleToInput</dt>
 62      *   <dd>
 63      *     Optional callback used to map the handle value to the <code>input</code> value.
 64      *   </dd>
 65      *   <dt>inputToHandle</dt>
 66      *   <dd>
 67      *     Optional callback used to map the <code>input</code> value to the handle value.
 68      *   </dd>
 69      *   <dt>sliderImpl</dt>
 70      *   <dd>
 71      *     Widget to use for the slider, defaults to <code>'slider'</code>.
 72      *   </dd>
 73      * </dl>
 74      * @name $.ui.simplicitySlider.options
 75      */
 76     options : {
 77       input: [],
 78       changeOnSlide: false,
 79       any: [],
 80       handleToInput: '',
 81       inputToHandle: '',
 82       sliderImpl: 'slider'
 83     },
 84     _create : function () {
 85       this._addClass('ui-simplicity-slider');
 86       if (this.options.sliderImpl !== '') {
 87         this.element[this.options.sliderImpl](this.options);
 88       }
 89       this._bind('slidechange', this._sliderChangeHandler);
 90       if (this.options.changeOnSlide) {
 91         this._bind('slide', this._sliderChangeHandler);
 92       }
 93       this._inputMinChangeHandler = this._inputValuesChangeHandlerFactory(0);
 94       this._inputMaxChangeHandler = this._inputValuesChangeHandlerFactory(1);
 95       if ($.isArray(this.options.input)) {
 96         this
 97           ._bind(this.options.input[0], 'change', this._inputMinChangeHandler)
 98           ._bind(this.options.input[1], 'change', this._inputMaxChangeHandler);
 99         $(this.options.input[0]).each($.proxy(function (i, elem) {
100           this._inputMinChangeHandler.call(this, {target: elem});
101         }, this));
102         $(this.options.input[1]).each($.proxy(function (i, elem) {
103           this._inputMaxChangeHandler.call(this, {target: elem});
104         }, this));
105       } else {
106         this._bind(this.options.input, 'change', this._inputValueChangeHandler);
107         $(this.options.input).each($.proxy(function (i, elem) {
108           this._inputValueChangeHandler.call(this, {target: elem});
109         }, this));
110       }
111     },
112     /**
113      * Event handler factory for the <code>change</code> event on an <code>input</code>
114      * bound to a double-handled slider.
115      * Updates the slider with the new value.
116      *
117      * @param valIdx
118      *   The slider handle number. <code>0</code> for the left handle and <code>1</code> or the right one.
119      * @name $.ui.simplicitySlider._inputValuesChangeHandlerFactory
120      * @function
121      * @private
122      */
123     _inputValuesChangeHandlerFactory: function (valIdx) {
124       return $.proxy(function (evt, ui) {
125         if (!this._ignoreChangeEvent) {
126           var values = this.element.slider('option', 'values');
127           var newValue = $(evt.target).val();
128           if (String(newValue) === '') {
129             newValue = this.options.any[valIdx];
130           } else if ($.isFunction(this.options.inputToHandle)) {
131             newValue = this.options.inputToHandle(newValue);
132           }
133           if (values[valIdx] !== newValue) {
134             values[valIdx] = newValue;
135             this.element.slider('option', 'values', values);
136           }
137         }
138       }, this);
139     },
140     /**
141      * Event handler for the <code>change</code> event on an <code>input</code> bound to a
142      * single-handled slider.
143      * Updates the slider with the new value.
144      *
145      * @name $.ui.simplicitySlider._inputValueChangeHandler
146      * @function
147      * @private
148      */
149     _inputValueChangeHandler: function (evt, ui) {
150       if (!this._ignoreChangeEvent) {
151         var value = this.element.slider('option', 'value');
152         var newValue = $(evt.target).val();
153         if (String(newValue) === '') {
154           newValue = this.options.any || '';
155         }
156         if (value !== newValue) {
157           this.element.slider('option', 'value', newValue);
158         }
159       }
160     },
161     /**
162      * Event handler for the <code>slidechange</code> and <code>slide</code> events.
163      * Updates the associated <code>input</code> elements with the new slider values.
164      * <p>
165      * This handler calls <code>change()</code> on the <code>input</code> elements
166      * immediately after changing their values.
167      *
168      * @name $.ui.simplicitySlider._sliderChangeHandler
169      * @function
170      * @private
171      */
172     _sliderChangeHandler: function (evt, ui) {
173       try {
174         this._ignoreChangeEvent = true;
175         var inputs = this.options.input;
176         var anys = this.options.any;
177         var values;
178         if ($.isArray(inputs)) {
179           // Double handled slider
180           values = this.element.slider('option', 'values');
181         } else {
182           // Single handled slider
183           inputs = [inputs];
184           anys = [anys];
185           values = [this.element.slider('option', 'value')];
186         }
187         $.each(inputs, $.proxy(function (idx, elem) {
188           var newVal = values[idx];
189           if (String(anys[idx]) === String(newVal)) {
190             newVal = '';
191           } else if ($.isFunction(this.options.handleToInput)) {
192             newVal = this.options.handleToInput(newVal);
193           }
194           if (String($(elem).val()) !== String(newVal)) {
195             $(elem).val(newVal);
196             $(elem).change();
197           }
198         }, this));
199       } finally {
200         this._ignoreChangeEvent = false;
201       }
202     },
203     destroy: function () {
204       if (this.options.sliderImpl !== '') {
205         this.element[this.options.sliderImpl]('destroy');
206       }
207       $.ui.simplicityWidget.prototype.destroy.apply(this, arguments);
208     }
209   });
210 }(jQuery));
