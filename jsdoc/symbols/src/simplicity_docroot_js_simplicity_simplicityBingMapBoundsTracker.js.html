1 /**
  2  * @name $.ui.simplicityBingMapBoundsTracker
  3  * @namespace A Bing map.
  4  * <p>
  5  * An invisible jquery ui widget which tracks changes in a Bing Map's bounds. Whenever the map bound change, a
  6  * simplicitybingmapboundstrackerbounds event is triggered with ui.bounds being the normalized map bounds.
  7  *
  8  * @example
  9  *   <div id="map" style="position: absolute; width: 300px; height: 300px;"></div>
 10  *   <script type="text/javascript">
 11  *     $('#map').simplicityBingMapBoundsTracker({
 12  *       credentials: 'Your credentials go here'
 13  *     });
 14  *   </script>
 15  *
 16  * @see Bing Maps AJAX Control v7 <a href="http://msdn.microsoft.com/en-us/library/gg427610.aspx">documentation</a>.
 17  */
 18 (function ($) {
 19   $.widget("ui.simplicityBingMapBoundsTracker", $.ui.simplicityWidget, {
 20     /**
 21      * Widget options.
 22      *
 23      * <dl>
 24      *   <dt>map</dt>
 25      *   <dd>
 26      *     Optional map instance, if not provided one will be created. Defaults to <code>''</code>.
 27      *   </dd>
 28      *   <dt>mapMoveEvents</dt>
 29      *   <dd>
 30      *     Provides an override of which vendor specific map events are used to determine
 31      *     when the position of the map changes. Expects a comma separated list of event names.
 32      *     Defaults to <code>'viewchangeend'</code>.
 33      *   </dd>
 34      * </dl>
 35      * @name $.ui.simplicityBingMapBoundsTracker.options
 36      */
 37     options : {
 38       map: '',
 39       mapMoveEvents: 'viewchangeend'
 40     },
 41     _create : function () {
 42       this._addClass('ui-simplicity-bing-map-bounds-tracker');
 43       this._map = this.options.map !== '' ? this.options.map : this.element.simplicityBingMap('map');
 44       this._boundsShapes = [];
 45       this._boundsChangeListeners = {};
 46       // TODO: tie in to deferred loading of map API
 47       $.each(this.options.mapMoveEvents.split(','), $.proxy(function (idx, eventName) {
 48         eventName = $.trim(eventName);
 49         if (eventName !== '') {
 50           var listener;
 51           if (this.options.eventThrottleInterval === '') {
 52             listener = Microsoft.Maps.Events.addHandler(this._map, eventName, $.proxy(this._mapBoundsChangeHandler, this));
 53           } else {
 54             listener = Microsoft.Maps.Events.addThrottledHandler(this._map, eventName, $.proxy(this._mapBoundsChangeHandler, this), this.options.eventThrottleInterval);
 55           }
 56           this._boundsChangeListeners[eventName] = listener;
 57         }
 58       }, this));
 59     },
 60     /**
 61      * Return the actual map object.
 62      *
 63      * @name $.ui.simplicityBingMapBoundsTracker.map
 64      * @function
 65      */
 66     map: function () {
 67       return this._map;
 68     },
 69     _mapBoundsChangeHandler: function () {
 70       var bounds = this.bounds();
 71       if (this.options.debug) {
 72         console.log('simplicityBingMapBoundsTracker: Bounds changed', bounds);
 73       }
 74       this._trigger('bounds', {}, bounds);
 75     },
 76     /**
 77      * Returns the normalized bounds for this map.
 78      *
 79      * @name $.ui.simplicityBingMapBoundsTracker.bounds
 80      * @function
 81      */
 82     bounds: function () {
 83       return this.normalizeBounds(this._map.getBounds(), this._map.getCenter());
 84     },
 85     /**
 86      * Normalizes the bounds for this map.
 87      *
 88      * @param bounds in vendor supplied format
 89      * @param center point in vendor supplied format
 90      * @name $.ui.simplicityBingMapBoundsTracker.normalizeBounds
 91      * @function
 92      * @private
 93      */
 94     normalizeBounds: function (bounds, center) {
 95       var result = {
 96         map: this._map,
 97         bounds: {
 98           vendor: bounds,
 99           north: bounds.getNorth(),
100           east: bounds.getEast(),
101           south: bounds.getSouth(),
102           west: bounds.getWest()
103         },
104         center: {
105           vendor: center,
106           latitude: center.latitude,
107           longitude: center.longitude
108         }
109       };
110       var radiusMiles1 = $.simplicityGeoFn.distanceMiles(
111         center.latitude, center.longitude, center.latitude, bounds.getEast());
112       var radiusMiles2 = $.simplicityGeoFn.distanceMiles(
113         center.latitude, center.longitude, bounds.getNorth(), center.longitude);
114       var minMiles = Math.min(radiusMiles1, radiusMiles2);
115       var maxMiles = Math.max(radiusMiles1, radiusMiles2);
116       $.extend(result, {
117         minRadius: {
118           meters: minMiles * 1609.344,
119           feet: minMiles * 5280,
120           miles:  minMiles,
121           km: minMiles * 1.609344
122         },
123         maxRadius: {
124           meters: maxMiles * 1609.344,
125           feet: maxMiles * 5280,
126           miles:  maxMiles,
127           km: maxMiles * 1.609344
128         }
129       });
130       return result;
131     },
132     /**
133      * Removes the bounds from the map.
134      *
135      * @name $.ui.simplicityBingMapBoundsTracker.hideBounds
136      * @function
137      * @private
138      */
139     hideBounds: function () {
140       $.each(this._boundsShapes, $.proxy(function (idx, shape) {
141         this._map.entities.remove(shape);
142       }, this));
143       this._boundsShapes.length = 0;
144     },
145     /**
146      * Adds an overlay for the bounds on the map.
147      *
148      * @param bounds Optional bounds to display, if missing the current bounds are used.
149      * @name $.ui.simplicityBingMapBoundsTracker.showBounds
150      * @function
151      * @private
152      */
153     showBounds: function (bounds) {
154       if ('undefined' === typeof bounds) {
155         bounds = this.bounds();
156       }
157       var boundsRect = new Microsoft.Maps.Polygon([
158           new Microsoft.Maps.Location(bounds.bounds.north, bounds.bounds.west),
159           new Microsoft.Maps.Location(bounds.bounds.north, bounds.bounds.east),
160           new Microsoft.Maps.Location(bounds.bounds.south, bounds.bounds.east),
161           new Microsoft.Maps.Location(bounds.bounds.south, bounds.bounds.west),
162           new Microsoft.Maps.Location(bounds.bounds.north, bounds.bounds.west)
163         ], {
164           strokeColor: new Microsoft.Maps.Color(255, 0, 0, 0),
165           fillColor: new Microsoft.Maps.Color(0)
166         });
167       var centerLatitude = new Microsoft.Maps.Polyline([
168           new Microsoft.Maps.Location(bounds.center.latitude, bounds.bounds.east),
169           new Microsoft.Maps.Location(bounds.center.latitude, bounds.bounds.west)
170         ], {
171           strokeColor: new Microsoft.Maps.Color(255, 0, 0, 0),
172           strokeThickness: 1,
173           fillColor: new Microsoft.Maps.Color(0)
174         });
175       var centerLongitude = new Microsoft.Maps.Polyline([
176           new Microsoft.Maps.Location(bounds.bounds.north, bounds.center.longitude),
177           new Microsoft.Maps.Location(bounds.bounds.south, bounds.center.longitude)
178         ], {
179           strokeColor: new Microsoft.Maps.Color(255, 0, 0, 0),
180           strokeThickness: 1,
181           fillColor: new Microsoft.Maps.Color(0)
182         });
183       var shapes = {
184         boundsRect: boundsRect,
185         // minCircle and maxCircle fields are missing due to lack of support for circle entities.
186         centerLatutide: centerLatitude,
187         centerLongitude: centerLongitude
188       };
189       this._trigger('boundsshapes', {}, shapes);
190       $.each(shapes, $.proxy(function (name, shape) {
191         this._map.entities.push(shape);
192         this._boundsShapes.push(shape);
193       }, this));
194     },
195     destroy: function () {
196       $.each(this._boundsChangeListeners, $.proxy(function (eventName, listener) {
197         Microsoft.Maps.Events.removeHandler(listener);
198       }, this));
199       this._boundsChangeListeners = {};
200       delete this._boundsShapes;
201       $.ui.simplicityWidget.prototype.destroy.apply(this, arguments);
202     }
203   });
204 }(jQuery));
