1 /**
  2  * @name $.ui.simplicityYahooMap
  3  * @namespace A Yahoo! map.
  4  * <p>
  5  * Yahho! Map widget that creates the map and listens for <code>simplicitySearchResponse</code> events
  6  * which it uses to add markers to the map for the search results.
  7  *
  8  * @example
  9  *   <div id="map" style="width: 300px; height: 300px;"></div>
 10  *   <script type="text/javascript">
 11  *     $('#map').simplicityYahooMap();
 12  *   </script>
 13  *
 14  * @see Yahoo! Maps Web Services - AJAX API <a href="http://developer.yahoo.com/maps/ajax/">documentation</a>.
 15  */
 16 (function ($) {
 17   $.widget("ui.simplicityYahooMap", $.ui.simplicityWidget, {
 18     /**
 19      * Widget options.
 20      *
 21      * <dl>
 22      *   <dt>map</dt>
 23      *   <dd>
 24      *     Optional map instance, if not provided one will be created. Defaults to <code>''</code>.
 25      *   </dd>
 26      *   <dt>mapOptions</dt>
 27      *   <dd>
 28      *     Options used when creating the map. Defaults to <code>''</code> which is expanded at
 29      *     runtime to
 30      *     <pre>
 31      *     {
 32      *       center: new YGeoPoint(0, 0),
 33      *       zoom: 16,
 34      *       mapTypeId: YAHOO_MAP_REG
 35      *     }
 36      *     </pre>
 37      *     Can be either an <code>Object</code> or a <code>function</code>.
 38      *   </dd>
 39      * </dl>
 40      * @name $.ui.simplicityYahooMap.options
 41      */
 42     options : {
 43       map: '',
 44       mapOptions: '',
 45       // The following options are for internal use only
 46       apiKey: '',
 47       mapVersion: '3.8'
 48     },
 49     _create: function () {
 50       this._addClass('ui-simplicity-yahoo-map');
 51       if (this.options.map !== '') {
 52         this._map = this.options.map;
 53       } else {
 54         this._initWhenAvailable();
 55       }
 56     },
 57     /**
 58      * Lazy initialization method used to create the map only when the necessary JavaScript
 59      * is available. Intended to be called from any of this widgets public methods that need to
 60      * access the this._map.
 61      *
 62      * @name $.ui.simplicityYahooMap._initWhenAvailable
 63      * @function
 64      * @private
 65      */
 66     _initWhenAvailable: function () {
 67       var wasAvailable = 'undefined' !== typeof this._map;
 68       if (wasAvailable) {
 69         // Already available, do nothing
 70       } else if (this.options.map !== '') {
 71         this._map = this.options.map;
 72       } else if ('undefined' !== typeof YMap) {
 73         var defaultMapOptions = {
 74           center: new YGeoPoint(0, 0),
 75           zoom: 16,
 76           mapType: YAHOO_MAP_REG
 77         };
 78         var mapOptions;
 79         if (this.options.mapOptions === '') {
 80           mapOptions = defaultMapOptions;
 81         } else if ($.isFunction(this.options.mapOptions)) {
 82           mapOptions = $.extend(defaultMapOptions, this.options.mapOptions.call(this));
 83         } else {
 84           mapOptions = $.extend(defaultMapOptions, this.options.mapOptions);
 85         }
 86         this._map = new YMap(this.element[0]);
 87         this._map.setMapType(mapOptions.mapType);
 88         this._map.drawZoomAndCenter(mapOptions.center, mapOptions.zoom);
 89         this._trigger('create', {}, {
 90           map: this._map
 91         });
 92       }
 93       var isAvailable = 'undefined' !== typeof this._map;
 94       return isAvailable;
 95     },
 96     /**
 97      * Return the actual map object.
 98      *
 99      * @name $.ui.simplicityYahooMap.map
100      * @function
101      */
102     map: function () {
103       this._initWhenAvailable();
104       return this._map;
105     },
106     /**
107      * Try to dynamically load the necessary JavaScript from the upstream vendor that will
108      * allow the map to function.
109      *
110      * @name $.ui.simplicityYahooMap.loadMap
111      * @function
112      * @private
113      */
114     loadMap: function () {
115       if ('undefined' === typeof this._map) {
116         if ('undefined' === typeof YMap) {
117           var src = 'http://api.maps.yahoo.com/ajaxymap?v=' + this.options.mapVersion;
118           if ('undefined' === typeof $.simplicityLoadJs) {
119             src = src + '&appid=YOUR_APP_ID';
120             alert('Dynamic loading of Yahoo! maps is not supported. Enable this widget by adding the following script tag to your page:\n\n' +
121               '<script type="text/javascript" src="' + src + '"></script>');
122           } else {
123             if (this.options.apiKey !== '') {
124               src = src + '&appid=' + this.options.apiKey;
125             }
126             $.simplicityLoadJs(src, $.proxy(function () {
127               this.refreshMap();
128             }, this));
129           }
130         }
131       }
132     },
133     destroy: function () {
134       delete this._map;
135       $.ui.simplicityWidget.prototype.destroy.apply(this, arguments);
136     }
137   });
138 }(jQuery));
