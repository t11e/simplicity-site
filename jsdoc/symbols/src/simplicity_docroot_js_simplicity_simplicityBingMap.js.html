1 /**
  2  * @name $.ui.simplicityBingMap
  3  * @namespace A Bing map.
  4  * <p>
  5  * Bing Map widget that creates the map and listens for <code>simplicitySearchResponse</code> events
  6  * which it uses to add markers to the map for the search results.
  7  *
  8  * @example
  9  *   <div id="map" style="position: absolute; width: 300px; height: 300px;"></div>
 10  *   <script type="text/javascript">
 11  *     $('#map').simplicityBingMap({
 12  *       credentials: 'Your credentials go here'
 13  *     });
 14  *   </script>
 15  *
 16  * @see Bing Maps AJAX Control v7 <a href="http://msdn.microsoft.com/en-us/library/gg427610.aspx">documentation</a>.
 17  */
 18 (function ($) {
 19   $.widget("ui.simplicityBingMap", $.ui.simplicityWidget, {
 20     /**
 21      * Widget options.
 22      *
 23      * <dl>
 24      *   <dt>map</dt>
 25      *   <dd>
 26      *     Optional map instance, if not provided one will be created. Defaults to <code>''</code>.
 27      *   </dd>
 28      *   <dt>credentials</dt>
 29      *   <dd>
 30      *     Mandatory option that contains your Bing credentials.
 31      *   </dd>
 32      *   <dt>mapOptions</dt>
 33      *   <dd>
 34      *     Options used when creating the map. Defaults to <code>''</code> which is expanded at
 35      *     runtime to
 36      *     <pre>
 37      *     {
 38      *       center: new Microsoft.Maps.Location(0, 0),
 39      *       zoom: 1,
 40      *       mapTypeId: Microsoft.Maps.MapTypeId.road
 41      *     }
 42      *     </pre>
 43      *     Can be either an <code>Object</code> or a <code>function</code>.
 44      *   </dd>
 45      * </dl>
 46      * @name $.ui.simplicityBingMap.options
 47      */
 48     options : {
 49       credentials: '',
 50       map: '',
 51       mapOptions: '',
 52       // The following options are for internal use only
 53       mapVersion: '7.0'
 54     },
 55     _create : function () {
 56       this._addClass('ui-simplicity-bing-map');
 57       if (this.options.map !== '') {
 58         this._map = this.options.map;
 59       } else {
 60         this._initWhenAvailable();
 61       }
 62     },
 63     /**
 64      * Lazy initialization method used to create the map only when the necessary JavaScript
 65      * is available. Intended to be called from any of this widgets public methods that need to
 66      * access the this._map.
 67      *
 68      * @name $.ui.simplicityBingMap._initWhenAvailable
 69      * @function
 70      * @private
 71      */
 72     _initWhenAvailable: function () {
 73       var wasAvailable = 'undefined' !== typeof this._map;
 74       if (wasAvailable) {
 75         // Already available, do nothing
 76       } else if (this.options.map !== '') {
 77         this._map = this.options.map;
 78       } else if ('undefined' !== typeof Microsoft && 'undefined' !== typeof Microsoft.Maps && 'undefined' !== typeof Microsoft.Maps.Map) {
 79         var defaultMapOptions = {
 80           credentials: this.options.credentials,
 81           center: new Microsoft.Maps.Location(0, 0),
 82           zoom: 1,
 83           mapTypeId: Microsoft.Maps.MapTypeId.road
 84         };
 85         var mapOptions;
 86         if (this.options.mapOptions === '') {
 87           mapOptions = defaultMapOptions;
 88         } else if ($.isFunction(this.options.mapOptions)) {
 89           mapOptions = $.extend(defaultMapOptions, this.options.mapOptions.call(this));
 90         } else {
 91           mapOptions = $.extend(defaultMapOptions, this.options.mapOptions);
 92         }
 93         this._map = new Microsoft.Maps.Map(this.element[0], mapOptions);
 94         this._trigger('create', {}, {
 95           map: this._map
 96         });
 97       }
 98       var isAvailable = 'undefined' !== typeof this._map;
 99       return isAvailable;
100     },
101     /**
102      * Return the actual map object.
103      *
104      * @name $.ui.simplicityBingMap.map
105      * @function
106      */
107     map: function () {
108       this._initWhenAvailable();
109       return this._map;
110     },
111     /**
112      * Try to dynamically load the necessary JavaScript from the upstream vendor that will
113      * allow the map to function.
114      *
115      * @name $.ui.simplicityBingMap.loadMap
116      * @function
117      * @private
118      */
119     loadMap: function () {
120       if ('undefined' === typeof this._map) {
121         if ('undefined' === typeof Microsoft || 'undefined' === typeof Microsoft.Maps || 'undefined' === typeof Microsoft.Maps.Map) {
122           var src = 'http://ecn.dev.virtualearth.net/mapcontrol/mapcontrol.ashx?v=' + this.options.mapVersion;
123           if ('undefined' === typeof $.simplicityLoadJs) {
124             alert('Dynamic loading of Bing maps is not supported. Enable this widget by adding the following script tag to your page:\n\n' +
125               '<script charset="UTF-8" type="text/javascript" src="' + src + '"></script>');
126           } else {
127             $.simplicityLoadJs(src, $.proxy(function () {
128               this.refreshMap();
129             }, this));
130           }
131         }
132       }
133     },
134     destroy: function () {
135       delete this._map;
136       $.ui.simplicityWidget.prototype.destroy.apply(this, arguments);
137     }
138   });
139 }(jQuery));
