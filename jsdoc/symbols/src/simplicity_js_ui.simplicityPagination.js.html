1 /**
  2  * @name $.ui.simplicityPagination
  3  * @namespace Pagination widget for simplicityDiscoverySearch.
  4  * <p>
  5  * Pagination widget bound to the <code>simplicitySearchResponse</code> event.
  6  * <p>
  7  * The current page can be bound either via an <code>input</code> element or
  8  * directly to the <code>simplicityState</code>.
  9  *
 10  * @example
 11  *   // Bound directly to simplicityState
 12  *   <div id="pagination"></div>
 13  *   <script type="text/javascript">
 14  *     $('#pagination').simplicityPagination();
 15  *   </script>
 16  *
 17  * @example
 18  *   // Bound to an input
 19  *   <input id="page" name="page" />
 20  *   <div id="pagination"></div>
 21  *   <script type="text/javascript">
 22  *     $('#page').simplicityInputs();
 23  *     $('#pagination').simplicityPagination({
 24  *       input: '#page'
 25  *     });
 26  *   </script>
 27  */
 28 (function ($, window) {
 29   $.widget("ui.simplicityPagination", $.ui.simplicityWidget, {
 30     /**
 31      * Widget options.
 32      *
 33      * <dl>
 34      *   <dt>stateElement</dt>
 35      *   <dd>
 36      *     The location of the simplicityState widget. Defaults to <code>'body'</code>.
 37      *   </dd>
 38      *   <dt>searchElement</dt>
 39      *   <dd>
 40      *     The location of the simplicityDiscoverySearch widget. Defaults to <code>'body'</code>.
 41      *   </dd>
 42      *   <dt>pageParam</dt>
 43      *   <dd>
 44      *     Used when binding directly to the <code>simplicityState</code> and determines which
 45      *     field the state holds the current page number. Defaults to <code>'page'</code>.
 46      *   </dd>
 47      *   <dt>input</dt>
 48      *   <dd>
 49      *     When set binds the current page to an <code>input</code> element instead of directly
 50      *     to the state via <code>pageParam</code>. Defaults to <code>''</code>.
 51      *   </dd>
 52      *   <dt>applyClass</dt>
 53      *   <dd>
 54      *     Classes to apply to the Next, Prev and Page elements. Defaults to <code>ui-corner-all</code>.
 55      *   </dd>
 56      *   <dt>scrollTopSelector</dt>
 57      *   <dd>
 58      *     Determines the jQuery selector to apply <code>$(window).scrollTop(0)</code> when
 59      *     a page changes. Defaults to <code>window</code>.
 60      *   </dd>
 61      *   <dt>scrollTopPosition</dt>
 62      *   <dd>
 63      *     Determines the scroll top position to use for <code>$(window).scrollTop(0)</code>. Defaults to <code>0</code>.
 64      *   </dd>
 65      *   <dt>num_display_entries</dt>
 66      *   <dd>
 67      *     Maximum number of pagination links to display.
 68      *     Setting to <code>0</code> will trim the navigation down to simply, "Previous" and "Next" links.
 69      *     Use an odd number so the current page will be displayed in the center of the
 70      *     page links when there are more than <code>num_display_entries</code> pages. Defaults to <code>11</code>.
 71      *   </dd>
 72      *   <dt>num_edge_entries</dt>
 73      *   <dd>
 74      *     Number of page links to show at the min and max range regardless of other constraints.
 75      *     If this number is set to 1, then links to the first and the last page are shown regardless of the current page
 76      *     and the visibility constraints set by num_display_entries. Setting to a larger number shows more links.
 77      *     Defaults to  <code>0</code>.
 78      *   </dd>
 79      *   <dt>link_to</dt>
 80      *   <dd>
 81      *     <code>href</code> template for the pagination links. The special value <code>__id__</code> will be replaced by the
 82      *     page number. Defaults to <code>#</code>.
 83      *   </dd>
 84      *   <dt>prev_text</dt>
 85      *   <dd>
 86      *     Text for the "previous" link. Setting to an empty string will omit the link. Defaults to <code>Prev</code>.
 87      *   </dd>
 88      *   <dt>next_text</dt>
 89      *     Text for the "next" link. Setting to an empty string will omit the link. Defaults to <code>Next</code>.
 90      *   <dd>
 91      *   </dd>
 92      *   <dt>ellipse_text</dt>
 93      *   <dd>
 94      *      When there is a gap between the numbers created by num_edge_entries and the displayed number interval,
 95      *      this text is inserted into the gap (in a span tag). Setting to an empty string omits the additional tag.
 96      *      Defaults to <code>...</code>.
 97      *   </dd>
 98      *   <dt>prev_show_always</dt>
 99      *   <dd>
100      *     When set to false, the "previous"-link is only shown if there is a previous page. Defaults to <code>true</code>.
101      *   </dd>
102      *   <dt>next_show_always</dt>
103      *   <dd>
104      *     When set to false, the "next"-link is only shown if there is a next page. Defaults to <code>true</code>.
105      *   </dd>
106      *   <dt>debug</dt>
107      *   <dd>
108      *     Enable logging of key events to <code>console.log</code>. Defaults to <code>false</code>.
109      *   </dd>
110      * </dl>
111      * @name $.ui.simplicityPagination.options
112      */
113     options : {
114       stateElement: 'body',
115       searchElement: 'body',
116       pageParam: 'page',
117       input: '',
118       applyClass: 'ui-corner-all',
119       scrollTopSelector: window,
120       scrollTopPosition: 0,
121       // for compatibility with jquery.pagination
122       num_display_entries: 11,
123       num_edge_entries: 0,
124       link_to: '#',
125       prev_text: 'Prev',
126       next_text: 'Next',
127       ellipse_text: '...',
128       prev_show_always: true,
129       next_show_always: true,
130       debug: false
131     },
132     _create : function () {
133       this
134         ._addClass('ui-simplicity-pagination ui-helper-clearfix')
135         ._bind(this.options.searchElement, 'simplicitySearchResponse', this._searchResponseHandler)
136         ._bind(this.options.stateElement, 'simplicityStateReset', this._stateResetHandler)
137         ._bind(this.element, 'setPage', this.setPage)
138         ._bind(this.element, 'prevPage', this.prevPage)
139         ._bind(this.element, 'nextPage', this.nextPage);
140       this._currentPage = 0;
141     },
142     /**
143      * Get the current page - 1 based.
144      *
145      * @name $.ui.simplicityPagination.getPage
146      * @function
147      */
148     getPage: function () {
149       return this._currentPage + 1;
150     },
151     /**
152      * Move to a specific page in the results. Causes a new search.
153      *
154      * @name $.ui.simplicityPagination.setPage
155      * @function
156      */
157     setPage: function (page1) {
158       this._setPage(page1 - 1);
159     },
160     /**
161      * Move to the next page in the results. Causes a new search if the last search indicated that there will be a next page.
162      *
163      * @name $.ui.simplicityPagination.nextPage
164      * @function
165      */
166     nextPage: function () {
167       this._setPage(this.getPage());
168     },
169     /**
170      * Move to the previous page in the results. Causes a new search if the current page is not the first page.
171      *
172      * @name $.ui.simplicityPagination.prevPage
173      * @function
174      */
175     prevPage: function () {
176       this._setPage(this.getPage() - 2);
177     },
178     /**
179      * Event handler for the <code>simplicitySearchResponse</code> event. Recreates
180      * the pagination widget to reflect the current search response.
181      *
182      * @name $.ui.simplicityPagination._searchResponseHandler
183      * @function
184      * @private
185      */
186     _searchResponseHandler: function (evt, searchResponse) {
187       var target = $('<div/>');
188       if (searchResponse) {
189         var discoveryResponse = searchResponse._discovery || {};
190         var resultSet = discoveryResponse.response || {startIndex: 0, pageSize: 0, totalSize: 0};
191         this._numPages = Math.ceil(resultSet.totalSize / resultSet.pageSize);
192         this._currentPage = resultSet.startIndex / resultSet.pageSize;
193         try {
194           this._ignoreCallback = true;
195           var startEnd = this._getStartEnd();
196           var start = startEnd[0];
197           var end = startEnd[1];
198           if (this.options.prev_text && (this.options.prev_show_always || this._currentPage > 0)) {
199             target.append(this._makeLink(this._currentPage - 1, this.options.prev_text, 'ui-simplicity-pagination-prev'));
200           }
201           if (start > 0 && this.options.num_edge_entries > 0) {
202             var lowEnd = Math.min(this.options.num_edge_entries, start);
203             this._makeLinks(target, 0, lowEnd, 'ui-simplicity-pagination-sp');
204             if (this.options.ellipse_text && this.options.num_edge_entries < start) {
205               $('<span/>').text(this.options.ellipse_text).appendTo(target);
206             }
207           }
208           this._makeLinks(target, start, end);
209           if (end < this._numPages && this.options.num_edge_entries > 0) {
210             if (this.options.ellipse_text && this._numPages - this.options.num_edge_entries > end) {
211               $('<span/>').text(this.options.ellipse_text).appendTo(target);
212             }
213             var startHigh = Math.max(this._numPages - this.options.num_edge_entries, end);
214             this._makeLinks(target, startHigh, this._numPages, 'ui-simplicity-pagination-ep');
215           }
216           if (this.options.next_text && (this.options.next_show_always || this._currentPage < this._numPages - 1)) {
217             target.append(this._makeLink(this._currentPage + 1, this.options.next_text, 'ui-simplicity-pagination-next'));
218           }
219           this.element.html(target.contents());
220         } finally {
221           this._ignoreCallback = false;
222         }
223       }
224     },
225     /**
226      * Helper that makes a range of pagination links, appending them to the parent.
227      *
228      * @name $.ui.simplicityPagination._makeLinks
229      * @function
230      * @private
231      */
232     _makeLinks: function (parent, start, end, classes) {
233       var i = start;
234       for (; i < end; i += 1) {
235         this._makeLink(i, i + 1, classes).appendTo(parent);
236       }
237     },
238     /**
239      * Helper that makes a pagination link. All element event handlers and data are set in this method, so returned link is ready
240      * to be used.
241      *
242      * @name $.ui.simplicityPagination._makeLink
243      * @function
244      * @private
245      */
246     _makeLink: function (page, text, classes) {
247       if (page < 0) {
248         page = 0;
249       } else if (page >= this._numPages) {
250         page = this._numPages - 1;
251       }
252       var result = $('<a/>');
253       if (page === this._currentPage) {
254         result = $('<span/>').addClass('ui-simplicity-pagination-current ui-priority-primary').text(text);
255       } else {
256         result = $('<a/>')
257           .attr('href', this.options.link_to.replace(/__id__/, page))
258           .text(text)
259           .addClass('ui-state-default')
260           .click($.proxy(this._paginationCallback, this));
261       }
262       if (classes) {
263         result.addClass(classes);
264       }
265       result.addClass(this.options.applyClass);
266       result.data('page', page);
267       if (page === this._currentPage) {
268         cssClass = (result.hasClass('ui-simplicity-pagination-prev') || result.hasClass('ui-simplicity-pagination-next')) ? 'ui-state-disabled' : 'ui-state-active';
269         result.addClass(cssClass);
270       }
271       return result;
272     },
273     /**
274      * Calculates the range of pagination links that might be displayed based on <code>num_display_entries</code>,
275      * <code>currentPage</code>, and <code>numPages</code>.
276      *
277      * @name $.ui.simplicityPagination._getStartEnd
278      * @function
279      * @private
280      */
281     _getStartEnd: function () {
282       var halfNumEntries = Math.floor(this.options.num_display_entries / 2);
283       var max = this._numPages - this.options.num_display_entries;
284       var start = 0;
285       if (this._currentPage > halfNumEntries) {
286         start = Math.max(Math.min(this._currentPage - halfNumEntries, max), 0);
287       }
288       var end = 0;
289       if (this._currentPage > halfNumEntries) {
290         end = Math.min(this._currentPage + halfNumEntries + this.options.num_display_entries % 2, this._numPages);
291       } else {
292         end = Math.min(this.options.num_display_entries, this._numPages);
293       }
294       return [start, end];
295     },
296     /**
297      * Callback for when a pagination link is clicked.
298      *
299      * @name $.ui.simplicityPagination._paginationCallback
300      * @function
301      * @private
302      */
303     _paginationCallback: function (evt) {
304       var page = $(evt.target).data('page');
305       this._setPage(page);
306       return false;
307     },
308     /**
309      * Changes the underlying search to reflect the requested page if page is different from the current page. Page is zero based.
310      *
311      * @name $.ui.simplicityPagination._setPage
312      * @function
313      * @private
314      */
315     _setPage: function (page0) {
316       if (!this._ignoreCallback && page0 !== this._currentPage && page0 >= 0) {
317         if (this.options.scrollTopSelector !== '') {
318           $(this.options.scrollTopSelector).scrollTop(this.options.scrollTopPosition);
319         }
320         if (this.options.input !== '') {
321           // Store the page in an input
322           $(this.options.input).val(page0 + 1);
323           $(this.options.input).change();
324         } else {
325           // Store the page directly in the search state
326           var state = $(this.options.stateElement).simplicityState('state');
327           state[this.options.pageParam] = String(page0 + 1);
328           $(this.options.stateElement).simplicityState('state', state);
329         }
330         if (this.options.searchElement !== '') {
331           if (false === $(this.options.searchElement).simplicityDiscoverySearch('option', 'searchOnStateChange')) {
332             $(this.options.searchElement).simplicityDiscoverySearch('search');
333           }
334         }
335       }
336     },
337     /**
338      * Event handler for the <code>simplicityStateReset</code> event. Resets the current page
339      * if the <code>pageParam</code> option is set.
340      *
341      * @name $.ui.simplicityPagination._stateResetHandler
342      * @function
343      * @private
344      */
345     _stateResetHandler: function (evt, state) {
346       if (this.options.input === '') {
347         if (this.options.debug) {
348           console.log('simplicityPagination: Resetting state parameter', name, 'for', this.element);
349         }
350         delete state[this.options.pageParam];
351       }
352     }
353   });
354 }(jQuery, window));
