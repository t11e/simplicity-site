1 /**
  2  * @name $.simplicity.haversineDistanceRadians
  3  * @function
  4  * @description
  5  *
  6  * Calculates the haversine distance in radians between two coordinates.
  7  *
  8  * @param lat1
  9  *   The latitude of the first coordinate in degrees.
 10  * @param lng1
 11  *   The longitude of the first coordinate in degrees.
 12  * @param lat2
 13  *   The latitude of the second coordinate in degrees.
 14  * @param lng2
 15  *   The longitude of the second coordinate in degrees.
 16  * @return
 17  *   The calculated distance in radians.
 18  */
 19 (function ($) {
 20   $.simplicity = $.simplicity || {};
 21   $.simplicity.haversineDistanceRadians = function (lat1, lng1, lat2, lng2) {
 22     var degreesToRadians = $.simplicity.degreesToRadians;
 23     lat1 = degreesToRadians(lat1);
 24     lng1 = degreesToRadians(lng1);
 25     lat2 = degreesToRadians(lat2);
 26     lng2 = degreesToRadians(lng2);
 27     var dLat = lat2 - lat1;
 28     var dLng = lng2 - lng1;
 29     var a = Math.sin(dLat / 2) * Math.sin(dLat / 2) +
 30       Math.cos(lat1) * Math.cos(lat2) * Math.sin(dLng / 2) * Math.sin(dLng / 2);
 31     var c = 2 * Math.atan2(Math.sqrt(a), Math.sqrt(1 - a));
 32     return c;
 33   };
 34 }(jQuery));
