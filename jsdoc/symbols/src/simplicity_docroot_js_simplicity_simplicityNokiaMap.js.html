1 /**
  2  * @name $.ui.simplicityNokiaMap
  3  * @namespace A Nokia map.
  4  * <p>
  5  * Nokia Map widget. Wraps a Nokia map as a jquery ui widget. Will optionally create a map, or a map can be
  6  * passed in using the widget options.
  7  *
  8  * @example
  9  *   <div id="map" style="width: 300px; height: 300px;"></div>
 10  *   <script type="text/javascript">
 11  *     $('#map').simplicityNokiaMap();
 12  *   </script>
 13  *
 14  * @see Nokia Maps - JavaScript API <a href="http://api.maps.nokia.com/">documentation</a>.
 15  */
 16 (function ($) {
 17   $.widget("ui.simplicityNokiaMap", $.ui.simplicityWidget, {
 18     /**
 19      * Widget options.
 20      *
 21      * <dl>
 22      *   <dt>map</dt>
 23      *   <dd>
 24      *     Optional map instance, if not provided one will be created. Defaults to <code>''</code>.
 25      *   </dd>
 26      *   <dt>mapOptions</dt>
 27      *   <dd>
 28      *     Options used when creating the map. Defaults to <code>''</code> which is expanded at
 29      *     runtime to
 30      *     <pre>
 31      *     {
 32      *       center: [0, 0],
 33      *       zoomLevel: 1
 34      *     }
 35      *     </pre>
 36      *     Can be either an <code>Object</code> or a <code>function</code>.
 37      *   </dd>
 38      * </dl>
 39      * @name $.ui.simplicityNokiaMap.options
 40      */
 41     options : {
 42       map: '',
 43       mapOptions: '',
 44       // The following options are for internal use only
 45       mapVersion: '2.1.0'
 46     },
 47     _create: function () {
 48       this._addClass('ui-simplicity-nokia-map');
 49       if (this.options.map !== '') {
 50         this._map = this.options.map;
 51       } else {
 52         this._initWhenAvailable();
 53       }
 54     },
 55     /**
 56      * Lazy initialization method used to create the map only when the necessary JavaScript
 57      * is available. Intended to be called from any of this widgets public methods that need to
 58      * access the this._map.
 59      *
 60      * @name $.ui.simplicityNokiaMap._initWhenAvailable
 61      * @function
 62      * @private
 63      */
 64     _initWhenAvailable: function () {
 65       var wasAvailable = 'undefined' !== typeof this._map;
 66       if (wasAvailable) {
 67         // Already available, do nothing
 68       } else if (this.options.map !== '') {
 69         this._map = this.options.map;
 70       } else if ('undefined' !== typeof nokia && 'undefined' !== typeof nokia.maps && 'undefined' !== typeof nokia.maps.map && 'undefined' !== typeof nokia.maps.map.Display) {
 71         var defaultMapOptions = {
 72           center: [0, 0],
 73           zoomLevel: 1
 74         };
 75         var mapOptions;
 76         if (this.options.mapOptions === '') {
 77           mapOptions = defaultMapOptions;
 78         } else if ($.isFunction(this.options.mapOptions)) {
 79           mapOptions = $.extend(defaultMapOptions, this.options.mapOptions.call(this));
 80         } else {
 81           mapOptions = $.extend(defaultMapOptions, this.options.mapOptions);
 82         }
 83         this._map = new nokia.maps.map.Display(this.element[0], mapOptions);
 84         this._trigger('create', {}, {
 85           map: this._map
 86         });
 87       }
 88       var isAvailable = 'undefined' !== typeof this._map;
 89       return isAvailable;
 90     },
 91     /**
 92      * Return the actual map object.
 93      *
 94      * @name $.ui.simplicityNokiaMap.map
 95      * @function
 96      */
 97     map: function () {
 98       this._initWhenAvailable();
 99       return this._map;
100     },
101     /**
102      * Try to dynamically load the necessary JavaScript from the upstream vendor that will
103      * allow the map to function.
104      *
105      * @name $.ui.simplicityNokiaMap.loadMap
106      * @function
107      * @private
108      */
109     loadMap: function () {
110       if ('undefined' === typeof this._map) {
111         if ('undefined' === typeof YMap) {
112           var src = 'http://api.maps.nokia.com/' + this.options.mapVersion + '/jsl.js';
113           if ('undefined' === typeof $.simplicityLoadJs) {
114             src = src + '&appid=YOUR_APP_ID';
115             alert('Dynamic loading of Nokia maps is not supported. Enable this widget by adding the following script tag to your page:\n\n' +
116               '<script type="text/javascript" src="' + src + '"></script>');
117           } else {
118             $.simplicityLoadJs(src, $.proxy(function () {
119               this.refreshMap();
120             }, this));
121           }
122         }
123       }
124     },
125     destroy: function () {
126       delete this._map;
127       $.ui.simplicityWidget.prototype.destroy.apply(this, arguments);
128     }
129   });
130 }(jQuery));
