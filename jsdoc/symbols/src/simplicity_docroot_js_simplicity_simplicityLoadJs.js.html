1 /**
  2  * @name $.simplicityLoadJs
  3  * @function
  4  * @description
  5  *
  6  * Poor mans dynamic JavaScript injection. A lot of upstream vendors use document.write
  7  * in their script resources which prevents the scripts from being dynamically loaded. This
  8  * plugin tries to load the script and overrides the document.write and document.writeln functions
  9  * to pick up any injected scripts and detect when all asynchronous scripts have been loaded.
 10  * <p>
 11  * It is not intended to be used for production, instead it is a development only solution.
 12  * <p>
 13  * Possible bugs include:
 14  * <ul>
 15  *   <li>
 16  *     Failure to detect failed script loads. This would invalidate the nested counter
 17  *     and not replace the original document.write function correctly.
 18  *   </li>
 19  *   <li>
 20  *     Failure to deal with calls to document.write which do more than add a single simple
 21  *     script element.
 22  *   </li>
 23  * </ul>
 24  *
 25  * @param url
 26  *   The URL to dynamically load.
 27  * @param callback
 28  *   Callback that is called after the JavaScript and any depedent scripts have all
 29  *   been loaded. Called with no arguments.
 30  *
 31  */
 32 (function ($) {
 33   var nestingCounter = 0;
 34   var oldDocumentWrite = undefined;
 35   var oldDocumentWriteln = undefined;
 36 
 37   var uninstall = function () {
 38     if ('undefined' !== typeof oldDocumentWrite) {
 39       /*jslint evil: true */
 40       document.write = oldDocumentWrite;
 41       oldDocumentWrite = undefined;
 42     }
 43     if ('undefined' !== typeof oldDocumentWriteln) {
 44       document.writeln = oldDocumentWriteln;
 45       oldDocumentWriteln = undefined;
 46     }
 47   };
 48 
 49   var scriptCompleteCallback = function (url, callback) {
 50     return function () {
 51       if ($.simplicityLoadJs.debug) {
 52         console.log('simplicityLoadJs: Loaded script ' + url);
 53       }
 54       nestingCounter -= 1;
 55       if (nestingCounter === 0) {
 56         if ($.simplicityLoadJs.debug) {
 57           console.log('simplicityLoadJs: Restoring original document.write');
 58         }
 59         uninstall();
 60         if ($.isFunction(callback)) {
 61           setTimeout(callback, 0);
 62         }
 63       }
 64     };
 65   };
 66 
 67   var loadScript = function (url, callback) {
 68     nestingCounter += 1;
 69     if ($.simplicityLoadJs.debug) {
 70       console.log('simplicityLoadJs: Loading script ' + url);
 71     }
 72     $.ajax({
 73       url: url,
 74       dataType: 'script',
 75       complete: scriptCompleteCallback(url, callback)
 76     });
 77   };
 78 
 79   var customDocumentWrite = function (callback) {
 80     return function () {
 81       $.each(arguments, function (i, arg) {
 82         var contentHtml = arg;
 83         var content = $(contentHtml);
 84         $(content).each(function (i, elem) {
 85           if (elem.nodeName === 'SCRIPT') {
 86             loadScript(elem.src, callback);
 87           }
 88         });
 89       });
 90     };
 91   };
 92 
 93   $.simplicityLoadJs = function (url, callback) {
 94     if (nestingCounter === 0) {
 95       if ($.simplicityLoadJs.debug) {
 96         console.log('simplicityLoadJs: Saving original document.write and installing custom version');
 97       }
 98       oldDocumentWrite = document.write;
 99       oldDocumentWriteln = document.writeln;
100       document.write = customDocumentWrite(callback);
101       document.writeln = customDocumentWrite(callback);
102     }
103     loadScript(url, callback);
104   };
105   $.simplicityLoadJs.uninstall = uninstall;
106 }(jQuery));
