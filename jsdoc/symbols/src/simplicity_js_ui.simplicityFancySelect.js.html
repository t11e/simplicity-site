1 /**
  2  * @name $.ui.simplicityFancySelect
  3  * @namespace Progressively enhanced select.
  4  *
  5  * A progressive enhancement of a <code>select</code> input that maps the
  6  * <code>select</code> to  <code>ul</code> and each <code>option</code> to
  7  * a <code>li</code>. The template can be overriden to generate a different
  8  * mapping.
  9  *
 10  * @example
 11  *   <select id="mySelect" name="example">
 12  *     <option value="1">one</option>
 13  *     <option value="2">two</option>
 14  *     <option value="3">three</option>
 15  *   <select>
 16  *   <div id="myFancy"></div>
 17  *   <script type="text/javascript">
 18  *     $('#myFancy')
 19  *       .simplicityFancySelect({
 20  *         select: '#mySelect',
 21  *       })
 22  *      .hide();
 23  *   </script>
 24  */
 25 (function ($) {
 26   $.widget("ui.simplicityFancySelect", $.ui.simplicityWidget, {
 27      /**
 28      * Widget options.
 29      *
 30      * <dl>
 31      *   <dt>select</dt>
 32      *   <dd>
 33      *     Mandatory option that points to the <code>select</code> input to be enhanced. Can be a DOM node, jQuery node or selector.
 34      *   </dd>
 35      *   <dt>displaySelected</dt>
 36      *   <dd>
 37      *     Determines whether selected options should be created and displayed.
 38      *     Defaults to <code>true</code>.
 39      *   </dd>
 40      *   <dt>displayUnselected</dt>
 41      *   <dd>
 42      *     Determines whether unselected options should be created and displayed.
 43      *     Defaults to <code>true</code>.
 44      *   </dd>
 45      *   <dt>displayCount</dt>
 46      *   <dd>
 47      *     When negative, displays and creates options with no limit. Acts as a limit to the number of
 48      *     created and display options otherwise. Can be thought of as the current page size.
 49      *     Defaults to <code>-1</code>.
 50      *   </dd>
 51      *   <dt>skipSelected</dt>
 52      *   <dd>
 53      *     When <code>skipCount</code> is greater than zero determines whether a selected option
 54      *     counts when tallying up the number of skipped options.
 55      *     Defaults to <code>true</code>.
 56      *   </dd>
 57      *   <dt>skipUnselected</dt>
 58      *   <dd>
 59      *     When <code>skipCount</code> is greater than zero determines whether an unselected option
 60      *     counts when tallying up the number of skipped options.
 61      *     Defaults to <code>true</code>.
 62      *   </dd>
 63      *   <dt>skipCount</dt>
 64      *   <dd>
 65      *     The number of options that should be skipped before the <code>displaySelected</code>
 66      *     and <code>displayUnselected</code> section is reached. Can be thought of as the
 67      *     starting positing of the current page.
 68      *     Defaults to <code>0</code>.
 69      *   </dd>
 70      *   <dt>firstPathOnly</dt>
 71      *   <dd>
 72      *     When <code>true</code> a single path is followed through the options. This is backed by the
 73      *     <code>is-path</code> data attribute on the <code>option</code> elements.
 74      *     Defaults to <code>false</code>.
 75      *   </dd>
 76      *   <dt>ignorePath</dt>
 77      *   <dd>
 78      *     When <code>true</code> any path options (with the <code>is-path</code> data attribute set
 79      *     to <code>true</code>) will be ignored.
 80      *     Defaults to <code>false</code>.
 81      *   </dd>
 82      *   <dt>firstPathSelections</dt>
 83      *   <dd>
 84      *     When <code>true</code> causes the click handler when deselecting an <code>option</code> which is
 85      *     considered a path (from the <code>is-path</code> data attribute) to alter the current selection
 86      *     to be all the options from the root to it's parent.
 87      *     <p>
 88      *     In other words, if the current selection is <code>Real Estate -> For Sale -> Condo</code> and 
 89      *     <code>For Sale</code> is deselected then the new selection would be <code>Real Estate</code>.
 90      *     <p>
 91      *     Defaults to <code>false</code>.
 92      *   </dd>
 93      *   <dt>refreshOnChange</dt>
 94      *   <dd>
 95      *     When <code>true</code> this widget repopulates itself whenever the target <code>select</code>
 96      *     input triggers a <code>change</code> event.
 97      *     Defaults to <code>true</code>.
 98      *   </dd>
 99      *   <dt>refreshOnCreate</dt>
100      *   <dd>
101      *     When <code>true</code> this widget populates itself at creation. Exposed to allow for
102      *     widgets that compose instances of this one.
103      *     Defaults to <code>true</code>.
104      *   </dd>
105      *   <dt>hideWhenEmpty</dt>
106      *   <dd>
107      *     When <code>true</code> the target element is hidden if the number of created and displayed
108      *     options drops to zero and shown when it is larger.
109      *     Defaults to <code>false</code>.
110      *   </dd>
111      *   <dt>checkableInputSelector</dt>
112      *   <dd>
113      *     Determines the selector (within the option template) to use to identify a checkable DOM input that reflects the current
114      *     widget state. Defaults to <code>:checkbox.ui-simplicity-fancy-select-checkbox,:radio.ui-simplicity-fancy-select-radio</code>.
115      *   </dd>
116      *   <dt>radioStyle</dt>
117      *   <dd>
118      *     Determines that the single select must always have a value selected, as would be the case for a radio button group.
119      *     Defaults to <code>false</code>.
120      *   </dd>
121      *   <dt>template</dt>
122      *   <dd>
123      *     When <code>''</code> falls back on the original DOM contents of the attached element.
124      *     Replaces the contents otherwise.
125      *     See source code for default value.
126      *   </dd>
127      * </dl>
128      * @name $.ui.simplicityFancySelect.options
129      */
130     options : {
131       select: '',
132       displaySelected: true,
133       displayUnselected: true,
134       displayCount: -1,
135       skipSelected: true,
136       skipUnselected: true,
137       skipCount: 0,
138       firstPathOnly: false,
139       ignorePath: false,
140       firstPathSelections: false,
141       refreshOnChange: true,
142       refreshOnCreate: true,
143       hideWhenEmpty: false,
144       radioStyle: false,
145       checkableInputSelector: ':checkbox.ui-simplicity-fancy-select-checkbox,:radio.ui-simplicity-fancy-select-radio',
146       optionLabelSelector: '.ui-simplicity-fancy-select-label',
147       optionCountSelector: '.ui-simplicity-fancy-select-count',
148       template: '' +
149         '<ul class="ui-simplicity-fancy-select-options ui-helper-clearfix">' +
150           '<li class="ui-simplicity-fancy-select-option ui-helper-clearfix">' +
151             '<a href="#" class="ui-simplicity-fancy-select-label"/>' +
152             '<span class="ui-simplicity-fancy-select-count"/>' +
153           '</li>' +
154         '</ul>'
155     },
156     _create : function () {
157       var select = $(this.options.select);
158       if (select.length === 0 || !select.is('select')) {
159         return;
160       }
161       this
162         ._addClass('ui-simplicity-fancy-select')
163         ._bind('click', this._clickHandler)
164         ._bind(select, 'change', this._changeHandler);
165       if (this.options.template !== '') {
166         this.element.children().remove();
167         $(this.options.template).appendTo(this.element);
168       }
169       var template = this.element.find('.ui-simplicity-fancy-select-option:first');
170       if (template.length === 0) {
171         template = $('<div class="ui-simplicity-fancy-select-option"/>');
172       }
173       this._optionTemplate = template.detach();
174       this._checkableInputSelector = '';
175       if (this.options.checkableInputSelector !== '') {
176         if (this._optionTemplate.find(this.options.checkableInputSelector).length > 0) {
177           this._checkableInputSelector = this.options.checkableInputSelector;
178         }
179       }
180       if (this.options.refreshOnCreate) {
181         this.refresh();
182       }
183     },
184     _changeHandler: function () {
185       if (!this._ignoreChange && this.options.refreshOnChange) {
186         this.refresh();
187       }
188     },
189     _setOptions: function () {
190       this._dontRefresh = true;
191       try {
192         $.ui.simplicityWidget.prototype._setOptions.apply(this, arguments);
193       } finally {
194         this._dontRefresh = false;
195       }
196       this.refresh();
197     },
198     _setOption: function (option, value) {
199       $.ui.simplicityWidget.prototype._setOption.apply(this, arguments);
200       if (!this._dontRefresh) {
201         this.refresh();
202       }
203     },
204     /**
205      * Recreates this widget based on the current <code>option</code> elements of the target
206      * <code>select</code> input.
207      *
208      * @name $.ui.simplicityFancySelect.refresh
209      * @function
210      */
211     refresh: function () {
212       var target = this.element.find('.ui-simplicity-fancy-select-options:first');
213       if (target.length === 0) {
214         target = this.element;
215       }
216       target.children().remove();
217       this._displayedCount = 0;
218       var displayCount = Number(this.options.displayCount);
219       var remainingToSkip = Number(this.options.skipCount);
220       var firstPathOnly = this.options.firstPathOnly;
221       var ignorePath = this.options.ignorePath;
222       if (displayCount !== 0) {
223         this.select().find('option').each($.proxy(function (idx, option) {
224           var joption = $(option);
225           if (joption.val() !== '') {
226             var isPath = joption.data('is-path');
227             if (firstPathOnly) {
228               if (isPath) {
229                 target.append(this._createOption(this._optionTemplate, option));
230                 this._displayedCount += 1;
231               } else {
232                 return false; // Stop iteration early
233               }
234             } else {
235               var isSelected = option.selected;
236               if (ignorePath && isPath) {
237                 // Ignore
238               } else if (remainingToSkip > 0) {
239                 if ((isSelected && this.options.skipSelected) || (!isSelected && this.options.skipUnselected)) {
240                   remainingToSkip -= 1;
241                 }
242               } else {
243                 if ((isSelected && this.options.displaySelected) || (!isSelected && this.options.displayUnselected)) {
244                   target.append(this._createOption(this._optionTemplate, option));
245                   this._displayedCount += 1;
246                   if (this._displayedCount === displayCount) {
247                     return false; // Stop iteration early
248                   }
249                 }
250               }
251             }
252           }
253         }, this));
254       }
255       if (this.options.hideWhenEmpty) {
256         if (this._displayedCount === 0) {
257           this.element.hide();
258         } else {
259           this.element.show();
260         }
261       }
262     },
263     /**
264      * Returns the target <code>select</code> input.
265      *
266      * @name $.ui.simplicityFancySelect.select
267      * @function
268      */
269     select: function () {
270       return $(this.options.select);
271     },
272     /**
273      * Returns the number of options currently displayed.
274      *
275      * @name $.ui.simplicityFancySelect.size
276      * @function
277      */
278     size: function () {
279       return this._displayedCount;
280     },
281     /**
282      * Creates an option from the template. Adds <code>facet-id</code>, <code>is-path</code>
283      * and <code>count</code> data attributes to the created element. Sets the label (<code>.ui-simplicity-fancy-select-label</code>)
284      * and count (</code>.ui-simplicity-fancy-select-count</code>) to contained elements that match the appropriate class.
285      * Triggers a <code>createOption</code> widget event (externally seen as <code>simplicityfancyfacetscreateoption</code>)
286      * which can be used to customize the generated DOM element.
287      *
288      * @name $.ui.simplicityFancySelect._createOption
289      * @function
290      * @private
291      */
292     _createOption: function (template, option) {
293       option = $(option);
294       var isPath = option.data('is-path');
295       var path = option.data('path');
296       var count = option.data('count');
297       var elem = template.clone()
298         .data('facet-id', option.val())
299         .data('is-path', 'undefined' !== typeof isPath ? isPath : false)
300         .data('path', 'undefined' !== typeof path ? path : []);
301       elem.find(this.options.optionLabelSelector).text(option.text());
302       elem.find(this.options.optionCountSelector).text('undefined' !== typeof count ? count : 0);
303       var eventOptions = {
304         target: elem,
305         source: option,
306         widget: this.element
307       };
308       this._trigger('createOption', {}, eventOptions);
309       if ('undefined' !== eventOptions.target && null !== eventOptions.target) {
310         elem = eventOptions.target;
311       }
312       this._markOptionSelected(elem, option[0].selected);
313       return elem;
314     },
315     /**
316      * Marks the given DOM node as either selected or unselected. Currently
317      * toggles a <code>selected</code> class and looks for any contained
318      * inputs (identified by selector <code>checkableInputSelector</code>) and
319      * (un)checks them as appropriate.
320      *
321      * @name $.ui.simplicityFancySelect._markOptionSelected
322      * @function
323      * @private
324      */
325     _markOptionSelected: function (option, selected) {
326       if (selected) {
327         option.addClass('ui-simplicity-fancy-select-selected');
328       } else {
329         option.removeClass('ui-simplicity-fancy-select-selected');
330       }
331       if (this._checkableInputSelector !== '') {
332         option.find(this._checkableInputSelector).each(function () {
333           this.checked = selected;
334         });
335       }
336     },
337     _clickHandler: function (evt) {
338       var option = $(evt.target);
339       if (!option.is('.ui-simplicity-fancy-select-option')) {
340         option = option.parentsUntil(this.element, '.ui-simplicity-fancy-select-option').first();
341       }
342       if (option.length !== 0) {
343         var facetId = option.data('facet-id');
344         var isPath = option.data('is-path');
345         var select = this.select();
346         var selected = select.val();
347         var path = option.data('path');
348         if ($.isArray(selected)) {
349           var pos = $.inArray(facetId, selected);
350           if (pos === -1) {
351             // Multi select add
352             selected.push(facetId);
353             this._markOptionSelected(option, true);
354           } else {
355             // Multi select remove
356             if (isPath && this.options.firstPathSelections) {
357               selected.length = 0;
358               $.each(path, function (idx, id) {
359                 selected.push(id);
360               });
361               this.element.find('.ui-simplicity-fancy-select-option')
362                 .each($.proxy(function (idx, elem) {
363                   elem = $(elem);
364                   var isSelected = $.inArray(elem.data('facet-id'), selected) !== -1;
365                   this._markOptionSelected($(elem), isSelected);
366                 }, this));
367             } else {
368               selected.splice(pos, 1);
369               this._markOptionSelected(option, false);
370             }
371           }
372           this._markOptionSelected(option, pos === -1);
373         } else if (selected !== facetId) {
374           // Single select change
375           if (selected !== null) {
376             this.element.find('.ui-simplicity-fancy-select-option')
377               .each($.proxy(function (idx, elem) {
378                 elem = $(elem);
379                 if (elem.data('facet-id') === selected) {
380                   this._markOptionSelected($(elem), false);
381                 }
382               }, this));
383           }
384           selected = facetId;
385           this._markOptionSelected(option, true);
386         } else if (!this.options.radioStyle) {
387           // Single select clear
388           if (isPath && this.options.firstPathSelections) {
389             selected = path.length === 0 ? null : path[path.length - 1];
390           } else {
391             selected = null;
392           }
393           this._markOptionSelected(option, false);
394         }
395         select.val(selected);
396         this._ignoreChange = true;
397         try {
398           select.change();
399         } finally {
400           this._ignoreChange = false;
401         }
402       }
403       if (!$(evt.target).is(":input")) {
404         evt.preventDefault();
405       }
406     }
407   });
408 }(jQuery));
