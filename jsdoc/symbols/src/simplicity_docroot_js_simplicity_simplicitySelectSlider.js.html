1 /**
  2  * @name $.ui.simplicitySelectSlider
  3  * @namespace Single handled slider bound to a select input.
  4  * <p>
  5  * Creates a progressive enhanced jQuery UI slider that is backed by a <code>select</code> input.
  6  * <p>
  7  * The widget options are passed through to the created slider. For more information see the
  8  * jQuery UI slider <a href="http://jqueryui.com/demos/slider/">documentation</a>.
  9  *
 10  * @example
 11  *   <select id="weight" name="w">
 12  *     <option value="l">Light</option>
 13  *     <option value="m">Medium</option>
 14  *     <option value="h">Heavy</option>
 15  *   <select>
 16  *   <div id="weightSlider"></div>
 17  *   <script type="text/javascript">
 18  *     $('#weightSlider').simplicitySelectSlider({
 19  *       select: '#weight'
 20  *     });
 21  *   </script>
 22  */
 23 (function ($) {
 24   $.widget("ui.simplicitySelectSlider", $.ui.simplicityWidget, {
 25     /**
 26      * Widget options.
 27      *
 28      * <dl>
 29      *   <dt>select</dt>
 30      *   <dd>
 31      *     A single <code>select</code> input to associate the slider handle with.
 32      *   </dd>
 33      *   <dt>changeOnSlide</dt>
 34      *   <dd>
 35      *      Set this to <code>true</code> to cause <code>slide</code> events to change the
 36      *      bound <code>select</code> input. Defaults to <code>false</code>.
 37      *   </dd>
 38      *   <dt>showTicks</dt>
 39      *   <dd>
 40      *      Set this to <code>true</code> to cause the display of tick marks on the slider.
 41      *      Defaults to <code>true</code>.
 42      *   </dd>
 43      *   <dt>showLabels</dt>
 44      *   <dd>
 45      *     Set this to <code>true</code> to cause the display of labels under the tick marks
 46      *     of the slider. All labels are displayed, if you have too many labels then you
 47      *     will want to disable this option. Defaults to <code>true</code>.
 48      *   </dd>
 49      *   <dt>centerLabels</dt>
 50      *   <dd>
 51      *     When labels are displayed this option causes their css to be adjusted to cause them
 52      *     to be centered around their tick mark. You would disable this if you had fixed width
 53      *     labels and wanted to fully control their position from css.
 54      *     Defaults to <code>true</code>.
 55      *   </dd>
 56      *   <dt>justifyEndLabels</dt>
 57      *   <dd>
 58      *     When labels are displayed this option causes the first label to be left justified and
 59      *     the right label to be right justified with the ends of the slider. You would disable
 60      *     this if you had fixed width labels and wanted to fully control their position from css.
 61      *     Defaults to <code>true</code>.
 62      *   </dd>
 63      *   <dt>showTooltip</dt>
 64      *   <dd>
 65      *     Set this to <code>true</code> to cause a tooltip to appear above the slider handle.
 66      *     Defaults to <code>false</code>.
 67      *   </dd>
 68      *   <dt>centerTooltip</dt>
 69      *   <dd>
 70      *     When tooltip display is enabled this option causes the tooltip to be centered on
 71      *     the slider handle. You would disable this if you had a fixed width tooltip and wanted
 72      *     to fully control it's position from css.
 73      *     Defaults to <code>true</code>.
 74      *   </dd>
 75      * </dl>
 76      * @name $.ui.simplicitySelectSlider.options
 77      */
 78     options : {
 79       select: '',
 80       changeOnSlide: false,
 81       showTicks: true,
 82       showLabels: true,
 83       centerLabels: true,
 84       justifyEndLabels: true,
 85       showTooltip: true,
 86       centerTooltip: true,
 87       scaleTemplate: '' +
 88         '<ol class="scale ui-helper-reset">' +
 89           '<li class="position">' +
 90             '<span class="label"></span>' +
 91             '<span class="tick ui-widget-content"></span>' +
 92           '</li>' +
 93         '</ol>',
 94       tooltipTemplate: '' +
 95         '<span class="tooltip ui-widget-content ui-corner-all">' +
 96           '<span class="tooltip-content"></span>' +
 97           '<span class="tooltip-pointer-down ui-widget-content">' +
 98             '<span class="tooltip-pointer-down-inner"></span>' +
 99           '</span>' +
100         '</span>'
101     },
102     _create : function () {
103       var select = $(this.options.select);
104       if (select.length === 0 || !select.is('select')) {
105         return;
106       }
107       this._addClass('ui-simplicity-select-slider ui-simplicity-slider');
108       this.element.slider($.extend({
109           min: 0,
110           max: select.find('option').length - 1
111         }, this.options));
112 
113       var scale = $(this.options.scaleTemplate);
114       scale.find('.position').remove();
115       this.element.append(scale);
116 
117       var tooltip = $(this.options.tooltipTemplate);
118       if (!this.options.showTooltip) {
119         tooltip.hide();
120       }
121       this.element.find('.ui-slider-handle').append(tooltip);
122 
123       this._bind('slidechange', this._sliderChangeHandler);
124       this._bind('slide', this._sliderChangeHandler);
125       if (this.options.changeOnSlide) {
126       }
127       this._bind(select, 'change', this._selectChangeHandler);
128       this._refreshScale();
129       this._refreshTooltip();
130     },
131     /**
132     * Override of <code>_setOption</code> that is used to refresh the
133     * slider tick marks, labels and tooltip when the assocaited options
134     * are changed.
135     *
136     * @name $.ui.simplicitySelectSlider._setOption
137     * @function
138     * @private
139     */
140     _setOption: function (option, value) {
141       $.ui.simplicityWidget.prototype._setOption.apply(this, arguments);
142       switch (option) {
143       case 'showTicks':
144       case 'showLabels':
145       case 'centerLabels':
146       case 'justifyEndLabels':
147         this._refreshScale();
148         break;
149       case 'showTooltip':
150       case 'centerTooltip':
151         this._refreshTooltip();
152         break;
153       }
154     },
155     /**
156      * Event handler for the <code>change</code> event on the <code>select</code> input.
157      *
158      * Updates the slider with the new value.
159      *
160      * @name $.ui.simplicitySelectSlider._selectChangeHandler
161      * @function
162      * @private
163      */
164     _selectChangeHandler: function (evt, ui) {
165       if (!this._ignoreChangeEvent) {
166         var changed = false;
167         var max = $(this.options.select).find('option').length - 1;
168         if (max !== this.element.slider('option', 'max')) {
169           // The select input was dynamically changed, alter the accepted range of
170           // values for the slider.
171           this.element.slider('option', 'max', max);
172           this._refreshScale();
173           changed = true;
174         }
175         var value = evt.target.selectedIndex;
176         if (changed || value !== this.element.slider('option', 'value')) {
177           this.element.slider('option', 'value', value);
178         }
179       }
180     },
181     /**
182      * Recreates the scale on the slider. This contains the tick marks and labels.
183      * Tick marks and labels are created only if shown. Toggling the <code>showTicks</code>
184      * or <code>showLabels</code> options will cause this method to be called and the
185      * ticks or labels to be created.
186      *
187      * @name $.ui.simplicitySelectSlider._refreshScale
188      * @function
189      * @private
190      */
191     _refreshScale: function () {
192       var scale = this.element.find('.scale');
193       scale.find('.position').remove();
194       var positionTemplate = $(this.options.scaleTemplate).find('.position').remove();
195       if (positionTemplate.length) {
196         var min = this.element.slider('option', 'min');
197         var max = this.element.slider('option', 'max');
198         for (var i = min; i <= max; i += 1) {
199           var left = (i / (max - min) * 100).toFixed(2);
200           var position = positionTemplate.clone()
201             .addClass(i === min ? 'first' : i === max ? 'last' : '')
202             .attr('style', 'left:' + left + '%');
203           if (!this.options.showTicks) {
204             position.find('.tick').hide();
205           }
206           var label = null;
207           if (this.options.showLabels) {
208             label = position.find('.label');
209             var labelText = $(this.options.select).find('option:eq(' + i + ')').text();
210             label.text(labelText);
211           }
212           scale.append(position);
213           var center = this.options.centerLabels;
214           var justify = this.options.justifyEndLabels;
215           if (label !== null && (center || justify)) {
216             if (justify && i === min) {
217               // Do nothing.
218             } else if (justify && i === max) {
219               label.css('marginLeft', -label.width());
220             } else if (center) {
221               label.css('marginLeft', -label.width() / 2);
222             }
223           }
224         }
225       }
226     },
227     /**
228      * Updates the content of the tooltip and hides or shows it as appropriate.
229      *
230      * @name $.ui.simplicitySelectSlider._refreshTooltip
231      * @function
232      * @private
233      */
234     _refreshTooltip: function (value) {
235       var tooltip = this.element.find('.tooltip');
236       if (tooltip.length) {
237         if (!this.options.showTooltip) {
238           tooltip.hide();
239         }
240         if ('undefined' === typeof value) {
241           value = this.element.slider('value');
242         }
243         var label = $(this.options.select).find('option:eq(' + value + ')').text();
244         tooltip.find('.tooltip-content').text(label);
245         if (this.options.showTooltip && this.options.centerTooltip) {
246           var left = -(tooltip.width() / 2) - 2;
247           tooltip.css('marginLeft', left + 'px');
248         }
249         if (this.options.showTooltip) {
250           tooltip.show();
251         }
252       }
253     },
254     /**
255      * Event handler for the <code>slidechange</code> and <code>slide</code> events.
256      * Updates the associated <code>select</code> input with the new slider value.
257      * <p>
258      * This handler calls <code>change()</code> on the <code>select</code> input
259      * immediately after changing it's value.
260      *
261      * @name $.ui.simplicitySelectSlider._sliderChangeHandler
262      * @function
263      * @private
264      */
265     _sliderChangeHandler: function (evt, ui) {
266       try {
267         this._ignoreChangeEvent = true;
268         var select = $(this.options.select)[0];
269         var value = select.selectedIndex;
270         var newVal = ui.value;
271         if (value !== newVal) {
272           select.selectedIndex = newVal;
273         }
274         this._refreshTooltip(newVal);
275         if (evt.type !== 'slide' || this.options.changeOnSlide) {
276           $(select).change();
277         }
278       } finally {
279         this._ignoreChangeEvent = false;
280       }
281     },
282     destroy: function () {
283       this.element.slider('destroy');
284       $.ui.simplicityWidget.prototype.destroy.apply(this, arguments);
285     }
286   });
287 }(jQuery));
