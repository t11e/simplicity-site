1 /**
  2  * @name $.ui.simplicityFacetedSelect
  3  * @namespace Dynamically populates the <code>option</code> elements in a <code>select</code> based on the facet response from the controller.
  4  * <p>
  5  * Listens to <code>simplicitySearchResponse</code> events and uses the facet output to update the
  6  * available <code>option</code> elements in the wrapped <code>select</code>.
  7  * <p>
  8  * Has special support for handling <code>option</code> elements with no value.
  9  *
 10  * @example
 11  *   <select id="category" name="cat">
 12  *     <option value="">Select...</option>
 13  *   <select>
 14  *   <script type="text/javascript">
 15  *     $('#category').simplicityFacetedSelect();
 16  *   </script>
 17  */
 18 (function ($) {
 19   $.widget("ui.simplicityFacetedSelect", $.ui.simplicityWidget, {
 20     /**
 21      * Widget options.
 22      *
 23      * <dl>
 24      *   <dt>facetsKey</dt>
 25      *   <dd>
 26      *     The key used in the facets request to identify the facets data.
 27      *     Defaults to the name attribute of the <code>select</code> element.
 28      *   </dd>
 29      *   <dt>searchElement</dt>
 30      *   <dd>
 31      *     The location of the simplicityDiscoverySearch widget.
 32      *     Defaults to <code>'body'</code>.
 33      *   </dd>
 34      *   <dt>optionTemplate</dt>
 35      *   <dd>
 36      *     Template used when <code>optionFormatter</code> is not set.
 37      *     Bracketed text will be replaced by it's mapped value,
 38      *     unknown variables and other text will remain as is.
 39      *     Defaults to <code>'{option}'</code>.
 40      *     <p>
 41      *     Supported substitutions:
 42      *     <dl>
 43      *       <dt><code>{option}</code></dt>
 44      *       <dd>The label for the facet. If no label is available then this is the id.</dd>
 45      *       <dt><code>{count}</code></dt>
 46      *       <dd>The count for the facet, if available. If not available defaults to
 47      *           the value of the <code>missingCount</code> option.</dd>
 48      *     </dl>
 49      *   </dd>
 50      *   <dt>missingCount</dt>
 51      *   <dd>
 52      *     Used during template substitution for facets that have no count.
 53      *     Defaults to <code>'?'</code>.
 54      *   </dd>
 55      *   <dt>optionFormatter</dt>
 56      *   <dd>
 57      *     Optional callback function that can be used to customize the label for each facet.
 58      *     Gets called with a single argument that is a dictionary containing <code>id</code>
 59      *     and <code>facet</code> entries. The <code>facet</code> entry consists of the returned
 60      *     data for the facet from the <code>simplicitySearchResponse</code> event.
 61      *   </dd>
 62      *   <dt>indent</dt>
 63      *   <dd>
 64      *     If set to a non-empty string, then this will be used as a leading indentation for each level of descendant facets.
 65      *     Note that actual whitespace characters are not rendered by most browsers.
 66      *     Setting to an empty string results in no indentation being applied for descendant facets.
 67      *     Defaults to <code>''</code>.
 68      *   </dd>
 69      *   <dt>maxDepth</dt>
 70      *   <dd>
 71      *     Option to control the maximum tree depth displayed. Setting this to a positive number causes the maximum
 72      *     depth to be that value. Setting this to <code>-1</code> makes the maximum depth unlimited and setting it to
 73      *     <code>0</code> causes no facets to be added.
 74      *   </dd>
 75      *   <dt>nextResponseOnly</dt>
 76      *   <dd>
 77      *     Option that causes the dynamic <code>option</code> population to trigger for the
 78      *     next <code>simplicitySearchResponse</code> event that contains facet data for the
 79      *     configured facet.
 80      *     Defaults to <code>false</code>.
 81      *   </dd>
 82      *   <dt>firstPathOnly</dt>
 83      *   <dd>
 84      *     Option that causes the dynamic <code>option</code> population to follow only the first
 85      *     selected path through the hierarchy of available facets.
 86      *     Defaults to <code>false</code>.
 87      *   </dd>
 88      * </dl>
 89      * @name $.ui.simplicityFacetedSelect.options
 90      */
 91     options: {
 92       facetsKey: '',
 93       searchElement: 'body',
 94       optionTemplate: '{option}',
 95       missingCount: '?',
 96       optionFormatter: '',
 97       indent: '',
 98       maxDepth: -1,
 99       nextResponseOnly: false,
100       firstPathOnly: false
101     },
102     _create: function () {
103       if (!this.element.is('select')) {
104         return;
105       }
106       this._frozen = false;
107       this
108         ._addClass('ui-simplicity-faceted-select')
109         ._bind(this.options.searchElement, 'simplicitySearchResponse', this._searchResponseHandler);
110     },
111     /**
112      * Event handler for the <code>simplicitySearchResponse</code> event.
113      * Calls <code>refresh</code> while respecting the <code>nextResponseOnly</code>
114      * option.
115      *
116      * @name $.ui.simplicityFacetedSelect._searchResponseHandler
117      * @function
118      * @private
119      */
120     _searchResponseHandler: function (evt, ui) {
121       if (!this._frozen) {
122         var refreshed = this.refresh(ui);
123         if (this.options.nextResponseOnly && refreshed) {
124           this._frozen = true;
125         }
126       }
127     },
128     /**
129      * Override of <code>_setOption</code> that is used to ensure that the
130      * <code>nextResponseOnly</code> option works as expected when changed.
131      *
132      * @name $.ui.simplicityFacetedSelect._setOption
133      * @function
134      * @private
135      */
136     _setOption: function (option, value) {
137       $.ui.simplicityWidget.prototype._setOption.apply(this, arguments);
138       if ('nextResponseOnly' === option) {
139         this._frozen = false;
140       }
141     },
142     /**
143      * Recreates the <code>option</code> elements in the wrapped <code>select</code>
144      * taking care to preserve the current selection.
145      *
146      * @param ui
147      *   The <code>simplicitySearchResponse</code> payload.
148      *
149      * @name $.ui.simplicityFacetedSelect.refresh
150      * @function
151      * @private
152      */
153     refresh: function (ui) {
154       var refreshed = false;
155       if (ui && ui._discovery && ui._discovery.response && ui._discovery.response.facets) {
156         var facetsKey = this.options.facetsKey || this.element.attr('name');
157         var facets = ui._discovery.response.facets[facetsKey];
158         if ('undefined' !== typeof facets) {
159           var selected = this.element.val();
160           this.element.find('option[value!=""]').remove();
161           if ('undefined' !== typeof facets.data) {
162             this._addFacets(facets.data, facets, 1, this.options.maxDepth);
163           }
164           if (selected !== null) {
165             this.element.val(selected);
166           }
167           this.element.change();
168           refreshed = true;
169         }
170       }
171       return refreshed;
172     },
173     _addFacets: function (data, facet, depth, maxDepth) {
174       if ('undefined' !== typeof facet) {
175         var firstPathOnly = this.options.firstPathOnly;
176         var hasNavIds = $.isArray(facet.navIds);
177         var hasChildIds = $.isArray(facet.childIds);
178         if ((maxDepth === -1 || depth <= maxDepth) && (hasNavIds || hasChildIds)) {
179           var navAdded = {};
180           if (firstPathOnly) {
181             if (hasNavIds && facet.navIds.length === 1) {
182               pathFacet = data[facet.navIds[0]];
183               if (pathFacet && $.isArray(pathFacet.childIds)) {
184                 // Part of a path, ignore all other branches
185                 this._addFacet(data, facet.navIds[0], depth, maxDepth, true);
186                 hasChildIds = false;
187                 hasNavIds = false;
188               }
189             } else {
190               // Possible multiple subtrees, stop any further recursion
191               maxDepth = depth;
192             }
193           }
194           if (hasNavIds) {
195             $.each(facet.navIds, function (idx, id) {
196               navAdded[id] = false;
197             });
198           }
199           if (hasChildIds) {
200             $.each(facet.childIds, $.proxy(function (idx, id) {
201               this._addFacet(data, id, depth, firstPathOnly ? depth : maxDepth, false);
202               if (hasNavIds && navAdded[id] === false) {
203                 navAdded[id] = true;
204               }
205             }, this));
206           }
207           if (hasNavIds) {
208             $.each(facet.navIds, $.proxy(function (idx, id) {
209               if (navAdded[id] === false) {
210                 this._addFacet(data, id, depth, maxDepth, false);
211               }
212             }, this));
213           }
214         }
215       }
216     },
217     _addFacet: function (data, id, depth, maxDepth, isPath) {
218       var facet = data[id] || {};
219       var text;
220       if ($.isFunction(this.options.optionFormatter)) {
221         text = this.options.optionFormatter.call(this, {
222           id: id,
223           facet: facet,
224           depth: depth
225         });
226       } else {
227         text = this.options.optionTemplate
228           .replace(/\{option\}/g, facet.label || id)
229           .replace(/\{count\}/g, 'undefined' !== typeof facet.count ? facet.count : this.options.missingCount);
230       }
231       var leading = "";
232       if ('' !== this.options.indent) {
233         var idx;
234         for (idx = 1; idx < depth; idx += 1) {
235           leading += this.options.indent;
236         }
237       }
238       if ('undefined' !== typeof text) {
239         var path = [];
240         var parentId = facet.parentId;
241         while ('undefined' !== typeof parentId && '' !== parentId) {
242           path.push(parentId);
243           parentId = (data[parentId] || {}).id;
244         }
245         path.reverse();
246         $('<option/>')
247           .val(id)
248           .text(leading + text)
249           .data('count', 'undefined' !== typeof facet.count ? facet.count : 0)
250           .data('depth', depth)
251           .data('is-path', isPath)
252           .data('path', path)
253           .appendTo(this.element);
254         this._addFacets(data, facet, depth + 1, maxDepth);
255       }
256     }
257   });
258 }(jQuery));
