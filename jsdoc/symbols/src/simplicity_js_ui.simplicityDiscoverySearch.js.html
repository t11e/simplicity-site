  2  * @name $.ui.simplicityDiscoverySearch
  3  * @namespace Performs a search against the Discovery Search Engine
  4  *
  5  * @example
  6  *   $('body').simplicityState();
  7  *   // Create all simplicityInputs widgets
  8  *   $('body')
  9  *     .simplicityState('mergeQueryParams')
 10  *     .simplicityHistory()
 11  *     .simplicityState('triggerChangeEvent')
 12  *     .simplicityPageSnapBack()
 13  *     <b>.simplicityDiscoverySearch({
 14  *       url: '/my_search_controller.php'
 15  *     })</b>
 16  *     <b>.simplicityDiscoverySearch('search');</b>
 17  */
 18 (function ($) {
 19   $.widget("ui.simplicityDiscoverySearch", $.ui.simplicityWidget, {
 20     /**
 21      * Widget options.
 22      *
 23      * <dl>
 24      *   <dt>url</dt>
 25      *   <dd>
 26      *     The required url to submit the search query.
 27      *   </dd>
 28      *   <dt>stateElement</dt>
 29      *   <dd>
 30      *     The location of the simplicityState widget. Defaults to <code>'body'</code>.
 31      *   </dd>
 32      *   <dt>searchOnStateChange</dt>
 33      *   <dd>
 34      *     When <code>true</code> triggers a search everytime the state changes. Defaults to <code>true</code>.
 35      *   </dd>
 36      *   <dt>initialSearchResponse</dt>
 37      *   <dd>
 38      *     The initial search response. Used when a page has a server-side search triggered during load.
 39      *     Defaults to <code>{}</code>.
 40      *   </dd>
 41      *   <dt>backend</dt>
 42      *   <dd>
 43      *     Defaults to 'controller', change to 'engine' to allow for direct queries to an engine.
 44      *   </dd>
 45      *   <dt>controllerCallback</dt>
 46      *   <dd>
 47      *     The a provided callback that completes or modifies the search query when option <code>backend</code>
 48      *     is 'engine'. Defaults to <code>''</code>.
 49      *   </dd>
 50      *   <dt>dataType</dt>
 51      *   <dd>
 52      *     Defaults to 'json', change to 'jsonp' to allow for cross domain search controller support.
 53      *     This requires that your search controller output valid JSONP.
 54      *   <dt>debug</dt>
 55      *   <dd>
 56      *     Enable logging of key events to <code>console.log</code>. Defaults to <code>false</code>.
 57      *   </dd>
 58      *   <dt>profile</dt>
 59      *   <dd>
 60      *     When <code>true</code> requests query profile data from the engine. The profile data
 61      *     can be used to analyze engine query response times. Defaults to <code>false</code>.
 62      *   </dd>
 63      *   <dt>triggerFacetCountEvent</dt>
 64      *   <dd>
 65      *     When <code>true</code> enables triggering of facet count events. Defaults to <code>false</code>.
 66      *   </dd>
 67      *   <dt>triggerResultSetEvent</dt>
 68      *   <dd>
 69      *     When <code>true</code> enables triggering of the simplicitySearchResultSet event. Defaults to <code>false</code>.
 70      *   </dd>
 71      *   <dt>extractResultSet</dt>
 72      *   <dd>
 73      *     When <code>true</code> a standardized resultSet dictionary is available. Defaults to <code>true</code>.
 74      *   </dd>
 75      * </dl>
 76      * @name $.ui.simplicityDiscoverySearch.options
 77      */
 78     options: {
 79       url: '',
 80       backend: 'controller',
 81       stateElement: 'body',
 82       searchOnStateChange: true,
 83       initialSearchResponse: {},
 84       dataType: 'json',
 85       debug: false,

 87       triggerResultSetEvent: false,

 89       profile: false,
 90       getPayloadParam: ''

 92     _create: function () {
 93       this._addClass('ui-simplicity-discovery-search');
 94       this.searchResponse(this.options.initialSearchResponse, false);
 95       this._bind(this.options.stateElement, 'simplicityStateChange', this._stateChangeHandler);
 96       this._ajaxHelper = $.simplicity.ajaxHelper(this);
 97     },
 98     _stateChangeHandler: function (evt, state) {
 99       if (this.options.searchOnStateChange) {
100         this.search(state);
101       }
102     },
103     /**
104      * Runs a search. The search happens asynchronously and will trigger multiple events.
105      * <ul>
106      *   <li>simplicitySearchSearching (before the ajax request is dispatched). This event may be
107      *   triggered multiple times in advance of the simplicitySearchResponse event. For example,
108      *   if any in-flight requests were aborted before the response has been handled, only the last
109      *   non-aborted response will trigger simplicitySearchResponse.</li>
110      *   <li>simplicitySearchResponse (original response)</li>
111      *   <li>ajaxSetup</li>
112      *   <li>ajaxSuccess</li>
113      *   <li>ajaxError</li>
114      *   <li>simplicityFacetCounts</li>
115      *   <li>simplicityResultSet</li>
116      * </ul>
117      *
118      * @param searchState
119      *   The search state as a JSON compatible <code>Object</code>. This is an optional parameter,
120      *   if not specified the search state is obtained from the <code>simplicityState</code> widget.
121      *
122      * @name $.ui.simplicityDiscoverySearch.search
123      * @function
124      *
125      * @example
126      *   // Bind to a search button
127      *   $('#searchButton').click(function (evt) {
128      *     $('body').simplicityDiscoverySearch('search');
129      *   });
130      */
131     search: function (stateOrDiscoveryRequest) {
132       if (this.options.url === '') {
133         if (this.options.debug) {
134           console.log("Option url is required.");
135         }
136         return;
137       }
138       if (arguments.length === 0) {
139         if (this.options.backend === 'engine' && !$.isFunction(this.options.controllerCallback)) {
140           if (this.options.debug) {
141             console.log("Backend type 'engine' requires controllerCallback to be valid function.");
142           }
143           return;
144         }
145         stateOrDiscoveryRequest = $(this.options.stateElement).simplicityState('state');
146       }
147       var request;
148       if ($.isFunction(this.options.controllerCallback)) {
149         request = this.options.controllerCallback(stateOrDiscoveryRequest);
150       } else {
151         request = stateOrDiscoveryRequest;
152       }
153       var ajaxData = request;
154       if (this.options.dataType === 'jsonp') {
155         if (this.options.getPayloadParam !== '') {
156           ajaxData = {};
157           ajaxData[this.options.getPayloadParam] = JSON.stringify(request);
158         }
159       } else if (this.options.backend === 'engine') {
160         ajaxData = JSON.stringify(request);
161       }
162       if (this.options.debug) {
163         console.log('Searching by "', this.options.backend, '" for ', this.element, 'with request', ajaxData);
164       }
165       var ajaxOptions = {
166           url: this.options.url,
167           type: this.options.backend === 'controller' || this.options.dataType === 'jsonp' ? "GET" : "POST",
168           contentType: 'application/json',
169           data: ajaxData,
170           dataType: this.options.dataType,
171           cache: false,
172           beforeSend: $.proxy(this._beforeSend, this),
173           debug: this.options.debug,
174           error: this._errorHandler,
175           success: function (data, textStatus, xhr) {
176             if (this.options.debug) {
177               console.log('Search success for', this.element, 'with response', data);
178             }
179             this.element.triggerHandler('ajaxSuccess', data, textStatus, xhr);
180             this._addHeadersToResponse(data, xhr);
181             if (this.options.backend === "engine") {
182               // decorate engine response for compatibility with widgets
183               data = {
184                 _discovery: {
185                   request: request,
186                   response: data
187                 }
188               };
189             }
190             this.searchResponse(data);
191           }
192         };
193       this.element.triggerHandler('simplicitySearchSearching');
194       this.element.triggerHandler('ajaxSetup', ajaxOptions);
195       this._ajaxHelper.ajax(ajaxOptions);
196     },
197     _beforeSend: function (xhr) {
198       if (this.options.profile) {
199         xhr.setRequestHeader('X-Profile', 1);
200       }
201     },
202     _addHeadersToResponse: function (response, xhr) {
203       var profile = xhr.getResponseHeader('X-Profiling-Trace');
204       if (profile > '') {
205         response._discovery.profile = profile.substr(5).split(', >>>   ');
206       }
207     },
208     _errorHandler: function (xhr, textStatus, message) {
209       if (this.options.debug) {
210         console.log('Search error for', this.element, 'textStatus:', textStatus, 'arguments', arguments);
211       }
212       this.element.triggerHandler('ajaxError', xhr, textStatus, message);
213       this.searchResponse({
214         error: true,
215         xhr: xhr,
216         status: xhr.status,
217         statusText: textStatus,
218         message: message
219       });
220     },
221     /**
222      * Get or set (and process) the search response. Called with no arguments to retrieve the
223      * last response. Called with arguments to process the current response (called by
224      * <code>search</code> on success or failure).
225      *
226      * In processing mode, this method triggers <code>simplicitySearchSearching</code>,
227      * <code>simplicitySearchResponse</code> and <code>simplicitySearchResponseHandled</code> events, then calls
228      * <code>facetCounts</code> and <code>resultSet</code>.
229      *
230      *
231      * @param searchResponse
232      *   The search response to process.
233      * @param triggerEvent
234      *   Optional parameter, set it to <code>false</code> to prevent triggering of events.
235      *   Defaults to <code>true</code> if not specified.
236      *
237      * @example
238      *   // Get the last search response
239      *   var response = $('body').simplicityDiscoverySearch('searchResponse');
240      *
241      * @example
242      *   // Process a fake search response for testing purposes
243      *   var response = {
244      *     // Fake response
245      *   };
246      *   $('body').simplicityDiscoverySearch('searchResponse', response);
247      *
248      * @name $.ui.simplicityDiscoverySearch.searchResponse
249      * @function
250      */
251     searchResponse: function (searchResponse, triggerEvent) {
252       if (arguments.length === 0) {
253         return JSON.parse(this._searchResponse);
254       }
255       var discoveryResponse = searchResponse._discovery || {};
256       var response = discoveryResponse.response || {};
257       if (this.options.triggerFacetCountEvent) {
258         searchResponse.drillDown = this.extractFacetCounts(response);
259       }
260       if (this.options.extractResultSet) {
261         searchResponse.resultSet = this.extractResultSet(response);
262       }
263       this._searchResponse = JSON.stringify(searchResponse);
264       if (triggerEvent !== false) {
265         this.element.triggerHandler('simplicitySearchResponse',
266           [JSON.parse(this._searchResponse)]);
267       }
268       if (this.options.triggerFacetCountEvent) {
269         this.facetCounts(searchResponse.drillDown, triggerEvent);
270       }
271       if (this.options.triggerResultSetEvent && this.options.extractResultSet) {
272         this.resultSet(searchResponse.resultSet, triggerEvent);
273       }
274       if (triggerEvent !== false) {
275         this.element.triggerHandler('simplicitySearchResponseHandled');
276       }
277     },
278     /**
279      * Get the or set the facet counts. Called with no arguments for get behavior.
280      * Triggers a <code>simplicityFacetCounts</code> event on set.
281      *
282      * @param facetCounts

285      *   Optional parameter, set it to <code>false</code> to prevent triggering of events.

293         return JSON.parse(this._facetCounts);

298       }
299     },

322     /**
323      * Converts the search response to a facet counts object using
324      * the drillDown response.
325      *
326      * @example
327      *   Input
328      *   {
329      *     drillDown: [
330      *       {
331      *         dimension: shape,
332      *         ids: ['a', 'b', 'c'],
333      *         exactCounts: [1, 3, 4],
334      *         fuzzyCounts: [2, 6, 8]
335      *       }
336      *     ]
337      *   }
338      *
339      * @example
340      *   Output
341      *   {
342      *     shape: {
343      *       exact: {'a': 1, 'b': 3, 'c': 4},
344      *       fuzzy: {'a': 2, 'b': 6, 'c': 8}
345      *     }
346      *   }
347      *
348      * @param discoveryResponse
349      * @name $.ui.simplicityDiscoverySearch.extractFacetCounts
350      * @function
351      */
352     extractFacetCounts: function (discoveryResponse) {
353       var facetCounts = {};
354       if ($.isArray(discoveryResponse.drillDown)) {
355         $.each(discoveryResponse.drillDown, function (idx, elem) {
356           var dimFacetCounts = facetCounts[elem.dimension] || {};
357           facetCounts[elem.dimension] = dimFacetCounts;
358           var exactCounts = dimFacetCounts.exact || {};
359           dimFacetCounts.exact = exactCounts;
360           var fuzzyCounts = dimFacetCounts.fuzzy || {};
361           dimFacetCounts.fuzzy = fuzzyCounts;
362           $.each(elem.ids, function (idx, id) {
363             exactCounts[id] = elem.exactCounts[idx];
364             fuzzyCounts[id] = elem.fuzzyCounts[idx];
365           });
366         });
367       }
368       return facetCounts;
369     },
370     /**
371      * Converts the search response to a resultset object.
372      * @example
373      *   Input
374      *   {
375      *     startIndex: 0
376      *     pageSize: 10,
377      *     currentPageSize: 10,
378      *     exactSize: 2,
379      *     totalSize: 120,
380      *     datasetSize: 55000,
381      *     itemIds: ['a', 'b', 'c'],
382      *     relevanceValues: [1.0, 1.0, 0.9],
383      *     exactMatches: [true, true, false],
384      *     properties: [
385      *       {title: 'aaa'},
386      *       {title: 'bbb'},
387      *       {title: 'ccc'}
388      *     ],
389      *     highlighting: [
390      *       {},
391      *       {'title': '[b]bbb[/b]'},
392      *       {}
393      *     ]
394      *   }
395      *
396      * @example
397      *   Output
398      *   <pre>
399      *   {
400      *     startIndex: 0,
401      *     pageSize: 10,
402      *     exactSize: 2,
403      *     totalSize: 120,
404      *     datasetSize: 55000,
405      *     numRows: 10,
406      *     rows: [
407      *       {
408      *         id: 'a',
409      *         exact: true,
410      *         score: 1.0
411      *         properties: {
412      *           title: 'aaa'
413      *         },
414      *         highlighting: {}
415      *       },
416      *       {
417      *         id: 'b',
418      *         exact: true,
419      *         score: 1.0
420      *         properties: {
421      *           title: 'bbb'
422      *         },
423      *         highlighting: {
424      *           title: '[b]bbb[/b]'
425      *         }
426      *       },
427      *       {
428      *         id: 'c',
429      *         exact: false,
430      *         score: 0.9
431      *         properties: {
432      *           title: 'ccc'
433      *         },
434      *         highlighting: {}
435      *       }
436      *     ]
437      *   }
438      *   </pre>
439      *
440      * @param discoveryResponse
441      * @name $.ui.simplicityDiscoverySearch.extractResultSet
442      * @function
443      */
444     extractResultSet: function (discoveryResponse) {
445       var resultSet = {
446         startIndex: discoveryResponse.startIndex || 0,
447         pageSize: discoveryResponse.pageSize || 0,
448         exactSize: discoveryResponse.exactSize || 0,
449         totalSize: discoveryResponse.totalSize || 0,
450         datasetSize: discoveryResponse.datasetSize || 0
451       };
452       var rows = [];
453       if ('undefined' !== typeof discoveryResponse.itemIds) {
454         var itemIds = discoveryResponse.itemIds;
455         var exactMatches = discoveryResponse.exactMatches;
456         var relevanceValues = discoveryResponse.relevanceValues;
457         var properties = discoveryResponse.properties;
458         var highlighting = discoveryResponse.highlighting;
459         if (!$.isArray(properties)) {
460           properties = null;
461         }
462         if (!$.isArray(highlighting)) {
463           highlighting = null;
464         }
465         $.each(itemIds, function (idx, itemId) {
466           row = {
467             id: itemId,
468             exact: exactMatches[idx],
469             score: relevanceValues[idx]
470           };
471           if (properties !== null) {
472             row.properties = properties[idx];
473           }
474           if (highlighting !== null) {
475             row.highlighting = highlighting[idx];
476           }
477           rows.push(row);
478         });
479       }
480       resultSet.numRows = rows.length;
481       resultSet.rows = rows;
482       return resultSet;
483     },
484     searchStats: function () {
485       return this._ajaxHelper.getStats();
486     }
487   });
488   $.fn.simplicityDiscoverySearchItemEnumerator = function (searchResponse, callback) {
489     var discovery = searchResponse._discovery || {};
490     var discoveryResponse = discovery.response || {};
491     if ('undefined' !== typeof discoveryResponse.itemIds) {
492       var startIndex = discoveryResponse.startIndex;
493       var itemIds = discoveryResponse.itemIds;
494       var exactMatches = discoveryResponse.exactMatches;
495       var relevanceValues = discoveryResponse.relevanceValues;
496       var properties = discoveryResponse.properties;
497       var highlighting = discoveryResponse.highlighting;
498       if (!$.isArray(properties)) {
499         properties = null;
500       }
501       if (!$.isArray(highlighting)) {
502         highlighting = null;
503       }
504       $.each(itemIds, function (idx, itemId) {
505         row = {
506           id: itemId,
507           exact: exactMatches[idx],
508           score: relevanceValues[idx],
509           resultsIndex0: idx + startIndex,
510           index0: idx,
511           index1: idx + 1
512         };
513         if (properties !== null) {
514           row.properties = properties[idx];
515         }
516         if (highlighting !== null) {
517           row.highlighting = highlighting[idx];
518         }
519         callback(idx, row);
520       });
521     }
522   };
