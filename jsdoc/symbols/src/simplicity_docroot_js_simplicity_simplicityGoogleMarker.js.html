1   /**
  2    * @name $.simplicityGoogleMarker
  3    * @namespace Contains factory methods and classes to create a custom Google OverlayView marker.
  4    */
  5 (function ($) {
  6   var _initialized = false;
  7   $.simplicityGoogleMarker = {};
  8   /**
  9    * @private
 10    */
 11   $.simplicityGoogleMarker._markerTemplateCache = {};
 12   /**
 13    * A custom Google Maps Overlay View for a jQuery-based standard marker. This class must be
 14    * instantiated using the factory method $.simplicityGoogleMarker.create().
 15    *
 16    * @class $.simplicityGoogleMarker.OverlayMarker
 17    * @throws Error when not instantiated using $.simplicityGoogleMarker.create()
 18    * @extends google.maps.OverlayView
 19    * @param {jQuery} element jQuery element to use for the marker
 20    * @param {Dictionary} options
 21    * <dl>
 22    *   <dt>template</dt>
 23    *   <dd>
 24    *     The HTML for the marker. The template is converted to a jQuery object.
 25    *     Defaults to <pre>
 26    * <div>
 27    *   <div class="row"/>
 28    *   <div class="ptr"/>
 29    *   <div class="shadow"/>
 30    * </div></pre>
 31    *   </dd>
 32    *   <dt>html</dt>
 33    *   <dd>
 34    *     The html or text to add to the marker.
 35    *   </dd>
 36    *   <dt>title</dt>
 37    *   <dd>
 38    *     The value of the element's <code>title</code> attribute.
 39    *   </dd>
 40    *   <dt>className</dt>
 41    *   <dd>
 42    *     The class to add to the marker.
 43    *   </dd>
 44    *   <dt>position</dt>
 45    *   <dd>
 46    *     The position of the marker. Type of <code>google.maps.LatLng</code>. Use <code>anchor</code>
 47    *     to determine the placement of the marker in relation to the position.
 48    *   </dd>
 49    *   <dt>anchor</dt>
 50    *   <dd>
 51    *     The anchor of the marker. Type of <code>google.maps.Point</code>. Defaults to the bottom center
 52    *     of the marker.
 53    *   </dd>
 54    *   <dt>row</dt>
 55    *   <dd>
 56    *     The required row information used to configure the marker. Keys: index0, index1
 57    *   </dd>
 58    *   <dt>draggable</dt>
 59    *   <dd>
 60    *     Indicates that the marker is draggable. Defaults to <code>false</code>.
 61    *   </dd>
 62    * </dl>
 63    */
 64   $.simplicityGoogleMarker.OverlayMarker = function (options) {
 65     if (!_initialized) {
 66       throw new Error("Instances of simplicityGoogleMarker must be created using $.simplicityGoogleMarker.create.");
 67     }
 68     google.maps.OverlayView.call(this);
 69     this.options = options;
 70     var template = this.options.template ||
 71       '<div><div class="row"/><div class="ptr"/><div class="shadow"/></div>';
 72     this._element = $.simplicityGoogleMarker._markerTemplateCache[template];
 73     if ('undefined' === typeof this._element) {
 74       this._element = $(template);
 75       $.simplicityGoogleMarker._markerTemplateCache[template] = this._element;
 76     }
 77     this._element = this._element.clone();
 78     this._element.data({
 79       marker: this,
 80       row: this.options.row || {}
 81     });
 82     this._element.addClass('simplicity-google-marker');
 83     if (this.options.className) {
 84       this._element.addClass(this.options.className);
 85     }
 86     if (this.options.html) {
 87       this.html(this.options.html);
 88     }
 89     if (this.options.title) {
 90       this.title(this.options.title);
 91     }
 92     this._latLng = options.position;
 93     this._latLngPt = null;
 94     if (this.options.draggable) {
 95       this._element.draggable({
 96         stop: $.proxy(function (evt, ui) {
 97           this.setLatLngFromOffset(
 98             ui.position.left - ui.originalPosition.left, // dX
 99             ui.position.top - ui.originalPosition.top // dY
100           );
101         }, this)
102       });
103     }
104   };
105   $.simplicityGoogleMarker.setup = function () {
106     if (!_initialized) {
107       $.simplicityGoogleMarker.OverlayMarker.prototype = new google.maps.OverlayView();
108       /**
109        * Implementation of <code>google.maps.OverlayView.onAdd</code>
110        *
111        * @function
112        * @private
113        */
114       $.simplicityGoogleMarker.OverlayMarker.prototype.onAdd = function () {
115         this.getPanes().overlayMouseTarget.appendChild(this._element[0]);
116       };
117       /**
118        * Implementation of <code>google.maps.OverlayView.onRemove</code>
119        *
120        * @function
121        * @private
122        */
123       $.simplicityGoogleMarker.OverlayMarker.prototype.onRemove = function () {
124         this._element.remove();
125       };
126       /**
127        * Implementation of <code>google.maps.OverlayView.draw</code>
128        *
129        * @function
130        * @private
131        */
132       $.simplicityGoogleMarker.OverlayMarker.prototype.draw = function () {
133         if ('undefined' !== typeof this._latLng) { // it is possible to create a marker without a position
134           this._latLngPt = this.getProjection().fromLatLngToDivPixel(this._latLng);
135           var h = this._element.height();
136          // default bottom center
137           var a = this.options.anchor || new google.maps.Point(this._element.width() / 2, h);
138           // $.draggable always uses relative positioning, so offset the y by the height of the element
139           // and the elements created before it.
140           var p = new google.maps.Point(
141             this._latLngPt.x - a.x,
142             this._latLngPt.y - a.y - (this.options.draggable ? this.options.row.index0 * h : 0));
143           this._element.css({
144             top:  p.y + 'px',
145             left: p.x + 'px',
146             'z-Index': -this.options.row.index1
147           });
148         }
149       };
150       /**
151        * Calculates new latLng for a marker based on it's new position relative
152        * to it's previous position and places the marker at that position
153        *
154        * @function
155        * @param {double} dX Offset value for the X axis
156        * @param {double} dY Offset value for the Y axis
157        */
158       $.simplicityGoogleMarker.OverlayMarker.prototype.setLatLngFromOffset = function (dX, dY) {
159         this._latLngPt = new google.maps.Point(dX + this._latLngPt.x, dY + this._latLngPt.y);
160         this._latLng = this.getProjection().fromDivPixelToLatLng(this._latLngPt);
161       };
162       /**
163        * Returns the marker's current position
164        *
165        * @function
166        * @returns {google.maps.LatLng} The position of the marker
167        */
168       $.simplicityGoogleMarker.OverlayMarker.prototype.getPosition = function () {
169         return this._latLng;
170       };
171       /**
172        * Sets the marker's current position
173        *
174        * @function
175        * @param {google.maps.LatLng} latLng The new position of the marker
176        */
177       $.simplicityGoogleMarker.OverlayMarker.prototype.setPosition = function (latLng) {
178         if ('undefined' !== typeof latLng) {
179           this._latLng = latLng;
180           this.draw();
181         }
182       };
183       /**
184        * Returns the marker's element
185        *
186        * @function
187        * @returns {jQuery} The jQuery element used for the marker
188        */
189       $.simplicityGoogleMarker.OverlayMarker.prototype.getElement = function () {
190         return this._element;
191       };
192       /**
193        * Gets or sets the marker's title
194        *
195        * @function
196        * @param {String} title The title to set
197        */
198       $.simplicityGoogleMarker.OverlayMarker.prototype.title = function (title) {
199         if (arguments.length === 0) {
200           return this._element.attr('title');
201         }
202         this._element.attr('title', title);
203       };
204       /**
205        * Gets or sets the marker's HTML
206        *
207        * @function
208        * @param {String} html The html or text to set
209        */
210       $.simplicityGoogleMarker.OverlayMarker.prototype.html = function (html) {
211         if (arguments.length === 0) {
212           return this._element.find(".row").html();
213         }
214         this._element.find(".row").html(html);
215       };
216       _initialized = true;
217     }
218   };
219   /**
220    * Factory Method to create a new $.simplicityGoogleMarker.OverlayMarker. The
221    * parameters to the method are the same as for the constructor of
222    * $.simplicityGoogleMarker.OverlayMarker.
223    *
224    * @name $.simplicityGoogleMarker.createOverlayMarker
225    * @function
226    * @returns {$.simplicityGoogleMarker.OverlayMarker}
227    * @public
228    */
229   $.simplicityGoogleMarker.createOverlayMarker = function () {
230     $.simplicityGoogleMarker.setup();
231     return new $.simplicityGoogleMarker.OverlayMarker(arguments[0]);
232   };
233   /**
234    * Factory Method to create a new google.maps.Marker. The
235    * parameters to the method are the same as for the constructor of
236    * google.maps.Marker.
237    *
238    * @name $.simplicityGoogleMarker.createMarker
239    * @function
240    * @returns {google.maps.Marker}
241    * @public
242    */
243   $.simplicityGoogleMarker.createMarker = function () {
244     return new google.maps.Marker(arguments[0]);
245   };
246 }(jQuery));
