1 /**
  2  * @name $.ui.simplicityNokiaMapShapeCreator
  3  * @namespace A Nokia map.
  4  * <p>
  5  * Adds the ability to show an point and radius or polygon/line to the map which
  6  * is persisted in a target input as geoJSON. This is the building block that
  7  * <code>simplicityMapShapeCreatorUi</code> exposes.
  8  * </p>
  9  *
 10  * @example
 11  *   <div id="map" style="width: 300px; height: 300px;"></div>
 12  *   <script type="text/javascript">
 13  *     $('#map').simplicityNokiaMap();
 14  *     $('#map').simplicityNokiaMapShapeCreator({
 15  *       input: $('<input name="placemark"/>').simplicityInputs()
 16  *     });
 17  *   </script>
 18  *
 19  * @see Nokia Maps JavaScript API V2 <a href="http://api.maps.nokia.com/en/index.html">documentation</a>.
 20  */
 21 (function ($) {
 22   $.widget("ui.simplicityNokiaMapShapeCreator", $.ui.simplicityMapShapeCreator, {
 23     /**
 24      * Widget options.
 25      * <dl>
 26      *   <dt><code>polygonOptions</code></dt>
 27      *   <dd>
 28      *     A dictionary of options that are used to define the look of polygons. Defaults to
 29      *     a map implementation specific value. Defaults to <code>{brush: {color: '#0066cc43'}, width: 0}</code>.
 30      *   </dd>
 31      *   <dt><code>lineStringOptions</code></dt>
 32      *   <dd>
 33      *     A dictionary of options that are used to define the look of line strings. Defaults to
 34      *     <code>{width: 1}</code>.
 35      *   </dd>
 36      * </dl>
 37      * @name $.ui.simplicityNokiaMapShapeCreator.options
 38      */
 39     options: {
 40       lineStringOptions: '',
 41       polygonOptions: ''
 42     },
 43     _create: function () {
 44       $.ui.simplicityMapShapeCreator.prototype._create.apply(this, arguments);
 45       if (this.options.lineStringOptions === '') {
 46         this.options.lineStringOptions = {width: 1};
 47       }
 48       if (this.options.polygonOptions === '') {
 49         this.options.polygonOptions = {brush: {color: '#0066cc43'}, width: 0};
 50       }
 51       this._shapeType = 'Point';
 52       this._currentMapShape = null;
 53       this._mapClickListener = null;
 54       this._map = $(this.element).simplicityNokiaMap('map');
 55       this._mvcCoords = new nokia.maps.geo.Strip([]);
 56       this.creatingMapCursor = 'crosshair';
 57       this.clickMethod = nokia.maps.dom.Page.browser.touch ? "tap" : "click";
 58       this.zoomMapComponent = this._map.getComponentById("zoom.DoubleClick");
 59 
 60       this._vertexMarkerImage = this._getMarkerImage(this.options.vertexMarkerOptions);
 61       this._firstVertexMarkerImage = this._getMarkerImage(this.options.firstVertexMarkerOptions);
 62       this._mapClick = null;
 63       this._addClass('ui-simplicity-nokia-shape-creator');
 64     },
 65     _getMarkerImage: function (markerOptions) {
 66       a = markerOptions.anchor;
 67       return {
 68         "icon": markerOptions.icon,
 69         "anchor": new nokia.maps.util.Point(a[0], a[1])
 70       };
 71     },
 72     _setOption: function (key, value) {
 73       $.ui.simplicityMapShapeCreator.prototype._setOption.apply(this, arguments);
 74       switch (key) {
 75       case "markerOptions":
 76         $.each(this._markers, $.proxy(function (idx, m) {
 77             m.marker.set(this.options.markerOptions);
 78           }, this));
 79         break;
 80       case "firstVertexMarkerOptions":
 81         if (this.options.editMode && this._markers.length > 2) {
 82           this._firstVertexMarkerImage = this._getMarkerImage(this.options.firstVertexMarkerOptions);
 83           this._markers[0].marker.set(this._firstVertexMarkerImage);
 84         }
 85         break;
 86       case "vertexMarkerOptions":
 87         this._vertexMarkerImage = this._getMarkerImage(this.options.vertexMarkerOptions);
 88         var minVertexIdx = this.options.editMode && this._markers.length > 2 ? 0 : -1;
 89         $.each(this._markers, $.proxy(function (idx, m) {
 90           if (idx > minVertexIdx) { // Skip the first vertex if it uses the firstVertex options
 91             m.marker.set(this._vertexMarkerImage);
 92           }
 93         }, this));
 94         break;
 95       }
 96     },
 97     _draw: function () {
 98       if (this.zoomMapComponent !== null) {
 99         if (this.options.editMode) {
100           this._map.removeComponent(this.zoomMapComponent);
101         } else if (this._map.getComponentById("zoom.DoubleClick") === null) {
102           this._map.addComponent(this.zoomMapComponent);
103         }
104       }
105       $(this.element).css({cursor: this._mapCursor()});
106     },
107     _mapCursor: function () {
108       return this.options.editMode ? this.creatingMapCursor : 'inherit';
109     },
110     _addLineString: function () {
111       var lineOptions = $.extend({}, this.options.lineStringOptions);
112       this._currentMapShape = new nokia.maps.map.Polyline(this._mvcCoords, lineOptions);
113       this._map.objects.add(this._currentMapShape);
114       this._shapeType = 'LineString';
115       return this._currentMapShape;
116     },
117     _addPolygon: function () {
118       this._currentMapShape = new nokia.maps.map.Polygon(this._mvcCoords, $.extend({}, this.options.polygonOptions));
119       this._map.objects.add(this._currentMapShape);
120       this._shapeType = 'Polygon';
121       return this._currentMapShape;
122     },
123     _setPolylinePath: function (v) {
124       if (this._polylineRadius) {
125         this._polylineRadius.set('path', v);
126       }
127     },
128     _bufferShape: function () {
129       if (this._geoJson.placemarks.type === 'LineString') {
130         var polyOptions = $.extend({}, this.options.polygonOptions);
131         this._polylineRadius = new nokia.maps.map.Polygon(this._getLineStringBufferPoints(), polyOptions);
132         this._map.objects.add(this._polylineRadius);
133       } else {
134         $.each(this._markers, $.proxy(function (idx, m) {
135           if (m.radiusCircle) {
136             m.radiusCircle.set($.extend({}, {"radius": this._radiusMeters}));
137           } else if (this._radiusMeters) {
138             m.radiusCircle = new nokia.maps.map.Circle(
139               m.marker.coordinate,
140               this._radiusMeters,
141               $.extend({}, this.options.polygonOptions, this.options.circleOptions));
142             this._map.objects.add(m.radiusCircle);
143           }
144           if (this._map.objects.indexOf(m.radiusCircle) === -1) {
145             this._map.objects.add(m.radiusCircle);
146           }
147         }, this));
148       }
149     },
150     _unbufferShape: function () {
151       $.each(this._markers, $.proxy(function (idx, m) {
152         if (m.radiusCircle) {
153           this._map.objects.remove(m.radiusCircle);
154           m.radiusCircle = null;
155         }
156       }, this));
157       if (this._polylineRadius) {
158         this._removeFromMap(this._polylineRadius);
159         this._polylineRadius = null;
160       }
161     },
162     _removeFromMap: function (object) {
163       if (object !== null) {
164         this._map.objects.remove(object);
165       }
166     },
167     _removeShapeFromMap: function () {
168       if (this._currentMapShape !== null) {
169         this._removeFromMap(this._currentMapShape);
170         this._currentMapShape = null;
171       }
172       this._shapeType = 'Point';
173       if (this._polylineRadius) {
174         this._removeFromMap(this._polylineRadius);
175         this._polylineRadius = null;
176       }
177     },
178     _getLatLngFromEvent: function (evt) {
179       var latLng = this._map.pixelToGeo(evt.displayX, evt.displayY);
180       return new nokia.maps.geo.Coordinate(latLng.latitude, latLng.longitude, 0, true);
181     },
182     _makeLatLng: function (lat, lng) {
183       // parameters must be numeric
184       return new nokia.maps.geo.Coordinate(lat, lng);
185     },
186     _latLngAsArray: function (latLng) {
187       return [latLng.latitude, latLng.longitude];
188     },
189     updateBounds: function () {
190       var ui = {
191         coordinates: []
192       };
193       this.boundsHandler()({}, ui);
194       if ($.isArray(ui.coordinates) && ui.coordinates.length > 0) {
195         var bounds = nokia.maps.geo.BoundingBox.coverAll(ui.coordinates);
196         if (typeof bounds !== 'undefined') {
197           this._map.zoomTo(bounds, false);
198         }
199       }
200     },
201     _getGeocodePrecisionData: function (point, ui) {
202       var p = ["houseNumber", "street", "postalCode", "district", "city", "county", "state", "country"];
203       var result = {
204         point: point,
205         addr: ui.item.value
206       };
207       try {
208         var v = ui.item.vendor.address;
209         $.each(p, function (idx, f) {
210           if ('undefined' !== typeof v[f]) {
211             result[f] = v[f];
212           }
213         });
214       } catch (e) {
215       }
216       return result;
217     },
218     setMapCenterFromMarker: function (marker) {
219       if (marker) {
220         this._map.setCenter(marker.coordinate, 'default');
221       }
222     },
223     getMappingProviderId: function () {
224       return "Nokia";
225     },
226     // Map Listeners
227     _addMapClickListener: function () {
228       if (this._mapClick === null) {
229         this._mapClick = $.proxy(this._mapClickHandler, this);
230         this._map.addListener(this.clickMethod, this._mapClick);
231       }
232     },
233     _removeMapClickListener: function () {
234       if (this._mapClick) {
235         this._map.removeListener(this.clickMethod, this._mapClick);
236       }
237       this._mapClick = null;
238     },
239     _addMarker: function (idx, latLng) {
240       var marker = new nokia.maps.map.Marker(
241           latLng,
242           $.extend({}, this._vertexMarkerImage, this.options.markerOptions)
243         );
244       marker.addListener('mouseenter', $.proxy(function (evt) {
245         $(this.element).css({cursor: 'move'});
246       }, this));
247       marker.addListener('mouseout', $.proxy(function (evt) {
248         $(this.element).css({cursor: this._mapCursor()});
249       }, this));
250       this._markers.push({"marker": marker});
251       if (this.options.draggableMarkers) {
252         marker.set({"draggable": true, "zIndex": +150});
253       }
254       this._map.objects.add(marker);
255       return marker;
256     },
257     _setMarkerStyle: function (idx, style) {
258       if (idx < this._markers.length) {
259         var marker = this._markers[idx].marker;
260         if (style === 'vertex') {
261           marker.set($.extend({zIndex: +150}, this._vertexMarkerImage));
262         } else if (style === 'firstVertex') {
263           marker.set($.extend({zIndex: +200}, this._firstVertexMarkerImage));
264         }
265       }
266     },
267     _addMarkerClickListener: function (idx, handler) {
268       if (idx < this._markers.length) {
269         this._markers[idx].marker.addListener(this.clickMethod, $.proxy(handler, this));
270       }
271     },
272     _removeMarkerClickListener: function (idx) {
273       // according to the docs, all async events are removed on destroy()
274     },
275     _addMarkerMoveListener: function (idx, marker, handler) {
276       if (idx < this._markers.length) {
277         marker.addListener('drag', $.proxy(handler, this));
278       }
279     },
280     _removeMarkerMoveListeners: function () {
281       // according to the docs, all async events are removed on destroy()
282     },
283     _addMarkerDragEndListener: function (idx, marker, handler) {
284       if (idx < this._markers.length) {
285         marker.addListener('dragend', $.proxy(handler, this));
286       }
287     },
288     _removeMarkerDragEndListeners: function () {
289       // according to the docs, all async events are removed on destroy()
290     },
291     _getDragPosition: function (marker, evt) {
292       return this._getLatLngFromEvent(evt);
293     },
294     _getMarkerPosition: function (marker) {
295       return marker.coordinate;
296     },
297     _removeMarkers: function () {
298       this._removeMarkerClickListener(0);
299       $.each(this._markers, $.proxy(function (idx, o) {
300         var marker = o.marker;
301         // according to the docs, all async events are removed on destroy()
302         if (o.radiusCircle) {
303           this._map.objects.remove(o.radiusCircle);
304         }
305         this._map.objects.remove(marker);
306       }, this));
307       this._markers = [];
308       this._coordsClear();
309     },
310     _coordsObserver: function (strip, idx, inserted, removed) {
311       if (inserted === 0) {
312       } else {
313         if (this._currentMapShape !== null) {
314           this._currentMapShape.path.set(idx, strip.get(idx));
315         }
316         if (idx < this._markers.length) {
317           if (this._markers[idx].radiusCircle) {
318             this._markers[idx].radiusCircle.set("center", strip.get(idx));
319           }
320         }
321       }
322     },
323     _createCoordsArray: function () {
324       this._mvcCoords = new nokia.maps.geo.Strip([]);
325       this._mvcCoords.addObserver(this._coordsObserver, this);
326       return this._mvcCoords;
327     },
328     _newCoordsArray: function (v) {
329       return new nokia.maps.geo.Strip(v);
330     },
331     _coordsLength: function () {
332       return this._mvcCoords.getLength();
333     },
334     _coordsPush: function (latLng) {
335       var idx = this._mvcCoords.getLength();
336       this._mvcCoords.add(latLng);
337       return idx;
338     },
339     _coordsGetAsDArray: function (idx) {
340       var p = this._mvcCoords.get(idx);
341       return [p.latitude, p.longitude];
342     },
343     _coordsSetAt: function (idx, latLng) {
344       this._mvcCoords.set(idx, latLng);
345     },
346     _coordsClear: function () {
347       this._mvcCoords.removeObserver(this._coordsObserver, this);
348       this._mvcCoords.destroy();
349       this._createCoordsArray();
350     },
351     boundsHandler: function () {
352       return $.proxy(function (evt, ui) {
353         var bounds = ui.coordinates;
354         if ('undefined' !== typeof bounds) {
355           var bb = null;
356           if ('undefined' !== typeof this._polylineRadius && this._polylineRadius !== null) {
357             bb = this._polylineRadius.getBoundingBox(this._map);
358             bounds.push(bb.bottomRight);
359             bounds.push(bb.topLeft);
360           } else {
361             $.each(this._markers, $.proxy(function (idx, m) {
362               bb = m[m.radiusCircle ? "radiusCircle" : "marker"].getBoundingBox(this._map);
363               bounds.push(bb.bottomRight);
364               bounds.push(bb.topLeft);
365             }, this));
366           }
367         }
368       }, this);
369     }
370   });
371 }(jQuery));
