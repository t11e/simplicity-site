1 /**
  2  * @name $.simplicityProxy
  3  * @function
  4  * @description
  5  *
  6  * Variation of $.proxy that doesn't share a guid and thus allows
  7  * for unbinding a single proxied function from an event.
  8  *
  9  * @param fn
 10  *   The function to proxy.
 11  * @param thisObject
 12  *   The object to apply the function to.
 13  *
 14  * @example
 15  *   var clickHandler = $.simplicityProxy(function () {}, this);
 16  *   $('#example').bind('click', clickHandler);
 17  *   // Do something
 18  *   $('#example).unbind('click', clickHandler);
 19  */
 20 (function ($) {
 21   $.simplicityProxy = function (fn, thisObject) {
 22     return function () {
 23       var currentFn = ('string' === typeof fn) ? thisObject[fn] : fn;
 24       return currentFn.apply(thisObject, arguments);
 25     };
 26   };
 27 }(jQuery));
