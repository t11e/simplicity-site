1 /**
  2  * @name $.ui.simplicitySelectSlider
  3  * @namespace Single handled slider bound to a select input.
  4  * <p>
  5  * Creates a progressive enhanced jQuery UI slider that is backed by a <code>select</code> input.
  6  * <p>
  7  * The widget options are passed through to the created slider. For more information see the
  8  * jQuery UI slider <a href="http://jqueryui.com/demos/slider/">documentation</a>.
  9  *
 10  * @example
 11  *   <select id="weight" name="w">
 12  *     <option value="l">Light</option>
 13  *     <option value="m">Medium</option>
 14  *     <option value="h">Heavy</option>
 15  *   <select>
 16  *   <div id="weightSlider"></div>
 17  *   <script type="text/javascript">
 18  *     $('#weightSlider').simplicitySelectSlider({
 19  *       select: '#weight'
 20  *     });
 21  *   </script>
 22  */
 23 (function ($) {
 24   $.widget("ui.simplicitySelectSlider", $.ui.simplicityWidget, {
 25     /**
 26      * Widget options.
 27      *
 28      * <dl>
 29      *   <dt>select</dt>
 30      *   <dd>
 31      *     A single <code>select</code> input to associate the slider handle with.
 32      *   </dd>
 33      *   <dt>secondSelect</dt>
 34      *   <dd>
 35      *     Enables dual-handle mode and specifies a single <code>select</code> input to associate
 36      *     the second slider handle with. Defaults to <code>''</code> which leaves the widget in
 37      *     single handle mode.
 38      *   </dd>
 39      *   <dt>allowHandleOverlap</dt>
 40      *   <dd>
 41      *     Only affects dual-handled sliders and controls whether the two handles can have
 42      *     the same position. Defaults to <code>false</code>.
 43      *   </dd>
 44      *   <dt>populateSecondSelect</dt>
 45      *   <dd>
 46      *     Only affects dual-handled sliders and controls how the widget deals with missing
 47      *     options in the second <code>select</code> input. When <code>true</code> it will
 48      *     automatically populate them with the label from the primary <code>select</code>.
 49      *     Defaults to <code>false</code>.
 50      *   </dd>
 51      *   <dt>changeOnSlide</dt>
 52      *   <dd>
 53      *      Set this to <code>true</code> to cause <code>slide</code> events to change the
 54      *      bound <code>select</code> input. Defaults to <code>false</code>.
 55      *   </dd>
 56      *   <dt>showTicks</dt>
 57      *   <dd>
 58      *      Set this to <code>true</code> to cause the display of tick marks on the slider.
 59      *      Defaults to <code>true</code>.
 60      *   </dd>
 61      *   <dt>showLabels</dt>
 62      *   <dd>
 63      *     Set this to <code>true</code> to cause the display of labels under the tick marks
 64      *     of the slider. All labels are displayed, if you have too many labels then you
 65      *     will want to disable this option. Defaults to <code>true</code>.
 66      *   </dd>
 67      *   <dt>centerLabels</dt>
 68      *   <dd>
 69      *     When labels are displayed this option causes their css to be adjusted to cause them
 70      *     to be centered around their tick mark. You would disable this if you had fixed width
 71      *     labels and wanted to fully control their position from css.
 72      *     Defaults to <code>true</code>.
 73      *   </dd>
 74      *   <dt>justifyEndLabels</dt>
 75      *   <dd>
 76      *     When labels are displayed this option causes the first label to be left justified and
 77      *     the right label to be right justified with the ends of the slider. You would disable
 78      *     this if you had fixed width labels and wanted to fully control their position from css.
 79      *     Defaults to <code>true</code>.
 80      *   </dd>
 81      *   <dt>showTooltip</dt>
 82      *   <dd>
 83      *     Set this to <code>true</code> to cause a tooltip to appear above the slider handle.
 84      *     Defaults to <code>false</code>.
 85      *   </dd>
 86      *   <dt>centerTooltip</dt>
 87      *   <dd>
 88      *     When tooltip display is enabled this option causes the tooltip to be centered on
 89      *     the slider handle. You would disable this if you had a fixed width tooltip and wanted
 90      *     to fully control it's position from css.
 91      *     Defaults to <code>true</code>.
 92      *   </dd>
 93      * </dl>
 94      * @name $.ui.simplicitySelectSlider.options
 95      */
 96     options : {
 97       select: '',
 98       secondSelect: '',
 99       allowHandleOverlap: false,
100       populateSecondSelect: false,
101       changeOnSlide: false,
102       showTicks: true,
103       showLabels: true,
104       centerLabels: true,
105       justifyEndLabels: true,
106       showTooltip: true,
107       centerTooltip: true,
108       scaleTemplate: '' +
109         '<ol class="ui-simplicity-select-slider-scale ui-helper-reset">' +
110           '<li class="ui-simplicity-select-slider-position">' +
111             '<span class="ui-simplicity-select-slider-label"></span>' +
112             '<span class="ui-simplicity-select-slider-tick ui-widget-content"></span>' +
113           '</li>' +
114         '</ol>',
115       tooltipTemplate: '' +
116         '<span class="ui-simplicity-select-slider-tooltip ui-widget-content ui-corner-all">' +
117           '<span class="ui-simplicity-select-slider-tooltip-content"></span>' +
118           '<span class="ui-simplicity-select-slider-tooltip-pointer-down ui-widget-content">' +
119             '<span class="ui-simplicity-select-slider-tooltip-pointer-down-inner"></span>' +
120           '</span>' +
121         '</span>'
122     },
123     _create : function () {
124       this._select = $(this.options.select);
125       if (this._select.length === 0 || !this._select.is('select')) {
126         return;
127       }
128       this._addClass('ui-simplicity-select-slider');
129       this._secondSelect = $(this.options.secondSelect);
130       if (this._secondSelect.length !== 0 && !this._secondSelect.is('select')) {
131         this._secondSelect = $('');
132       }
133       var sliderOptions = {
134         min: 0,
135         max: this._sliderMax()
136       };
137       if (this._secondSelect.length !== 0) {
138         $.extend(sliderOptions, {
139           values: [sliderOptions.min, sliderOptions.max],
140           range: true
141         });
142       }
143       this.element.slider($.extend(sliderOptions, this.options));
144 
145       var scale = $(this.options.scaleTemplate);
146       scale.find('.ui-simplicity-select-slider-position').remove();
147       this.element.append(scale);
148 
149       var tooltip = $(this.options.tooltipTemplate);
150       if (!this.options.showTooltip) {
151         tooltip.hide();
152       }
153       this.element.find('.ui-slider-handle').append(tooltip);
154 
155       this._bind('slidechange', this._sliderChangeHandler);
156       this._bind('slide', this._sliderChangeHandler);
157       this._bind(this._select, 'change', this._selectChangeHandler);
158       this._bind(this._secondSelect, 'change', this._selectChangeHandler);
159       this.refreshScale();
160       this.refreshTooltip();
161     },
162     /**
163     * Override of <code>_setOption</code> that is used to refresh the
164     * slider tick marks, labels and tooltip when the assocaited options
165     * are changed.
166     *
167     * @name $.ui.simplicitySelectSlider._setOption
168     * @function
169     * @private
170     */
171     _setOption: function (option, value) {
172       $.ui.simplicityWidget.prototype._setOption.apply(this, arguments);
173       switch (option) {
174       case 'showTicks':
175       case 'showLabels':
176       case 'centerLabels':
177       case 'justifyEndLabels':
178         this.refreshScale();
179         break;
180       case 'showTooltip':
181       case 'centerTooltip':
182         this.refreshTooltip();
183         break;
184       }
185     },
186     /**
187      * Event handler for the <code>change</code> event on the <code>select</code> input.
188      *
189      * Updates the slider with the new value.
190      *
191      * @name $.ui.simplicitySelectSlider._selectChangeHandler
192      * @function
193      * @private
194      */
195     _selectChangeHandler: function (evt, ui) {
196       if (!this._ignoreChangeEvent) {
197         var changed = false;
198         var max = this._sliderMax();
199         if (max !== this.element.slider('option', 'max')) {
200           // The select input was dynamically changed, alter the accepted range of
201           // values for the slider.
202           this.element.slider('option', 'max', max);
203           this.refreshScale();
204           changed = true;
205         }
206         if (this._secondSelect.length === 0) {
207           // Single handled slider, no mapping necessary
208           var selectIndex = this._select[0].selectedIndex;
209           var sliderIndex = this.element.slider('option', 'value');
210           if (changed || selectIndex !== sliderIndex) {
211             this.element.slider('option', 'value', selectIndex);
212           }
213         } else {
214           // Dual handled slider, use this._select to convert val to/from slider index.
215           var selectVals = [this._select.val(), this._secondSelect.val()];
216           var sliderIndexes = this.element.slider('option', 'values');
217           var sliderVals = [this._sliderIndexToVal(sliderIndexes[0]), this._sliderIndexToVal(sliderIndexes[1])];
218           if (changed || selectVals[0] !== sliderVals[0] || selectVals[1] != sliderVals[1]) {
219             var selectIndexes = [this._sliderIndexFromVal(selectVals[0]), this._sliderIndexFromVal(selectVals[1], 1)];
220             this.element.slider('option', 'values', selectIndexes);
221           }
222         }
223       }
224     },
225     /**
226      * Recreates the scale on the slider. This contains the tick marks and labels.
227      * Tick marks and labels are created only if shown. Toggling the <code>showTicks</code>
228      * or <code>showLabels</code> options will cause this method to be called and the
229      * ticks or labels to be created.
230      *
231      * @name $.ui.simplicitySelectSlider.refreshScale
232      * @function
233      * @private
234      */
235     refreshScale: function () {
236       var scale = this.element.find('.ui-simplicity-select-slider-scale');
237       scale.find('.ui-simplicity-select-slider-position').remove();
238       var positionTemplate = $(this.options.scaleTemplate).find('.ui-simplicity-select-slider-position').remove();
239       if (positionTemplate.length) {
240         var min = this.element.slider('option', 'min');
241         var max = this.element.slider('option', 'max');
242         for (var i = min; i <= max; i += 1) {
243           var left = (i / (max - min) * 100).toFixed(2);
244           var position = positionTemplate.clone()
245             .addClass(i === min ? 'ui-simplicity-select-slider-position-first' : i === max ? 'ui-simplicity-select-slider-position-last' : '')
246             .attr('style', 'left:' + left + '%');
247           if (!this.options.showTicks) {
248             position.find('.ui-simplicity-select-slider-tick').hide();
249           }
250           var label = null;
251           if (this.options.showLabels) {
252             var selectVal = this._sliderIndexToVal(i);
253             var labelText = this._selectValToLabel(selectVal);
254             label = position.find('.ui-simplicity-select-slider-label');
255             label.text(labelText);
256           }
257           scale.append(position);
258           var center = this.options.centerLabels;
259           var justify = this.options.justifyEndLabels;
260           if (label !== null && (center || justify)) {
261             if (justify && i === min) {
262               // Do nothing.
263             } else if (justify && i === max) {
264               label.css('marginLeft', -label.width());
265             } else if (center) {
266               label.css('marginLeft', -label.width() / 2);
267             }
268           }
269         }
270       }
271     },
272     /**
273      * Updates the content of the tooltip(s) and hides or shows it as appropriate.
274      *
275      * @name $.ui.simplicitySelectSlider.refreshTooltip
276      * @function
277      * @private
278      */
279     refreshTooltip: function (handleNum, sliderIndex) {
280       if (arguments.length === 0) {
281         this.refreshTooltip(0);
282         if (this._secondSelect.length !== 0) {
283           this.refreshTooltip(1);
284         }
285         return;
286       }
287       var tooltip = this.element.find('.ui-simplicity-select-slider-tooltip:' + (handleNum === 1 ? 'last' : 'first'));
288       if (tooltip.length) {
289         if (!this.options.showTooltip) {
290           tooltip.hide();
291         }
292         if ('undefined' === typeof sliderIndex) {
293           if (this._secondSelect.length === 0) {
294             sliderIndex = this.element.slider('value');
295           } else {
296             sliderIndex = this.element.slider('values')[handleNum];
297           }
298         }
299         var selectVal = this._sliderIndexToVal(sliderIndex);
300         var label = this._selectValToLabel(selectVal, handleNum);
301         tooltip.find('.ui-simplicity-select-slider-tooltip-content').text(label);
302         if (this.options.showTooltip && this.options.centerTooltip) {
303           var left = -(tooltip.width() / 2) - 2;
304           tooltip.css('marginLeft', left + 'px');
305         }
306         if (this.options.showTooltip && label !== '') {
307           tooltip.show();
308         }
309       }
310     },
311     /**
312      * Event handler for the <code>slidechange</code> and <code>slide</code> events.
313      * Updates the associated <code>select</code> input with the new slider value.
314      * <p>
315      * This handler calls <code>change()</code> on the <code>select</code> input
316      * immediately after changing it's value.
317      *
318      * @name $.ui.simplicitySelectSlider._sliderChangeHandler
319      * @function
320      * @private
321      */
322     _sliderChangeHandler: function (evt, ui) {
323       if (typeof ui.values === 'undefined') {
324         // Single handled slider
325         this._sliderHandleToSelect(evt.type, ui.value);
326       } else {
327         // Dual handled slider
328         if (!this.options.allowHandleOverlap && ui.values[0] === ui.values[1]) {
329           return false;
330         }
331         this._sliderHandleToSelect(evt.type, ui.values[0]);
332         this._sliderHandleToSelect(evt.type, ui.values[1], 1);
333       }
334     },
335     _sliderHandleToSelect: function (eventType, sliderIndex, handleNum) {
336       var select = handleNum === 1 ? this._secondSelect : this._select;
337       if (select.length > 0) {
338         this._ignoreChangeEvent = true;
339         try {
340           // Lookup the canonical value from the primary select
341           var newVal = this._sliderIndexToVal(sliderIndex);
342           // Use to sync with the select bound to this particular handle
343           if (newVal !== select.val()) {
344             select.val(newVal);
345             if (this.options.populateSecondSelect && handleNum === 1 && select.val() !== newVal) {
346               $('<option/>')
347                 .val(newVal)
348                 .text(this._selectValToLabel(newVal))
349                 .appendTo(select);
350               select.val(newVal);
351             }
352           }
353           this.refreshTooltip(handleNum, sliderIndex);
354           if (eventType !== 'slide' || this.options.changeOnSlide) {
355             select.change();
356           }
357         } finally {
358           this._ignoreChangeEvent = false;
359         }
360       }
361     },
362     _sliderMax: function () {
363       var max = this._select.find('option').length - 1;
364       if (this._secondSelect.length !== 0 && this._select.find('option[value=""]').length === 1) {
365         max += 1;
366       }
367       return max;
368     },
369     _sliderIndexToVal: function (index) {
370       var val = $(this._select).find('option:eq(' + index + ')').val();
371       return typeof val !== 'undefined' ? val : '';
372     },
373     _sliderIndexFromVal: function (val, handleNum) {
374       var result = -1;
375       var options = this._select.find('option');
376       if (val !== '') {
377         options.each(function (idx, option) {
378           if ($(option).val() === val) {
379             result = idx;
380             return false;
381           }
382         });
383       }
384       return result !== -1 ? result : handleNum === 1 ? options.length : 0;
385     },
386     _selectValToLabel: function (val, handleNum) {
387       var select = handleNum == 1 ? this._secondSelect : this._select;
388       var options = select.find('option');
389       var result = '';
390       options.each(function (idx, option) {
391         option = $(option);
392         if (option.val() === val) {
393           result = option.text();
394           return false;
395         }
396       });
397       if (result === '' && handleNum === 1) {
398         // Missing label in second select, use the label from the primary select instead.
399         result = this._selectValToLabel(val);
400       }
401       return result;
402     },
403     destroy: function () {
404       this.element.slider('destroy');
405       this.element.children().remove();
406       $.ui.simplicityWidget.prototype.destroy.apply(this, arguments);
407     }
408   });
409 }(jQuery));
