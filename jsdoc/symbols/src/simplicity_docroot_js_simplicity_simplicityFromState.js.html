1 /**
  2  * @name $.fn.simplicityFromState
  3  * @function
  4  * @description
  5  *
  6  * Plugin to apply state to input elements.
  7  * Apply this to input elements to change their state based on a state <code>Object</code>.
  8  *
  9  * @param state
 10  *   The target <code>Object</code> to extract the state to.
 11  * @param triggerChange
 12  *   Optional argument, when <code>true</code> invokes <code>change()</code> on any elements
 13  *   that were changed.
 14  * @param debug
 15  *   Optional argument, when <code>true</code> enables logging via <code>console.log</code>.
 16  *
 17  * @example
 18  *   var state = {
 19  *     "shape": "circle",
 20  *     "size": ["small", "medium"]
 21  *   };
 22  *   $(':input').simplicityFromState(state);
 23  */
 24 (function ($) {
 25   $.fn.simplicityFromState = function (state, triggerChange, debug) {
 26     return this.each(function () {
 27       var element = $(this);
 28       var name = element.attr('name');
 29       if (name !== '') {
 30         var changed = false;
 31         var searchValue = state[name];
 32         if (element.is(':checkbox')) {
 33           if (!$.isArray(searchValue)) {
 34             searchValue = [searchValue];
 35           }
 36           var myValue = element.attr('value');
 37           var checkCheckBox = (-1 !== $.inArray(myValue, searchValue || []));
 38           if (checkCheckBox !== element[0].checked) {
 39             element[0].checked = checkCheckBox;
 40             changed = true;
 41           }
 42         }
 43         else if (element.is(':radio')) {
 44           var checkRadio = (element.attr('value') === searchValue);
 45           if (checkRadio !== element[0].checked) {
 46             element[0].checked = checkRadio;
 47             changed = true;
 48           }
 49         } else {
 50           if (element.val() !== searchValue) {
 51             if (element.is('select') && 'undefined' !== typeof searchValue && searchValue !== null) {
 52               // Automatically add any missing options to a select input
 53               var options = {};
 54               element.find('option').each(function (idx, option) {
 55                 options[$(option).val()] = true;
 56               });
 57               $.each($.isArray(searchValue) ? searchValue : [searchValue], function (idx, value) {
 58                 if (!options[value]) {
 59                   if (debug) {
 60                     console.log('simplicityFromState: Adding dynamic option', value, 'to select', element);
 61                   }
 62                   options[value] = true;
 63                   $('<option/>')
 64                     .val(value)
 65                     .appendTo(element);
 66                 }
 67               });
 68             }
 69             element.val(searchValue);
 70             changed = true;
 71           }
 72         }
 73         if (changed && triggerChange) {
 74           if (debug) {
 75             console.log('simplicityFromState: Triggering change event on', element);
 76           }
 77           element.change();
 78           if (debug) {
 79             console.log('simplicityFromState: Triggered change event on', element);
 80           }
 81         }
 82       }
 83     });
 84   };
 85 }(jQuery));
