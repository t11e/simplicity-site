1 /**
  2  * @name $.ui.simplicityGoogleMapBoundsTracker
  3  * @namespace A Google map.
  4  * <p>
  5  * An invisible jquery ui widget which tracks changes in a Google Map's bounds. Whenever the map bound change, a
  6  * simplicitygooglemapboundstrackerbounds event is triggered with ui.bounds being the normalized map bounds.
  7  *
  8  * @example
  9  *   <div id="map" style="width: 300px; height: 300px;"></div>
 10  *   <script type="text/javascript">
 11  *     $('#map').simplicityGoogleMapBoundsTracker();
 12  *   </script>
 13  *
 14  * @see Google Maps JavaScript API V3 <a href="https://developers.google.com/maps/documentation/javascript/reference">documentation</a>.
 15  */
 16 (function ($) {
 17   $.widget("ui.simplicityGoogleMapBoundsTracker", $.ui.simplicityWidget, {
 18     /**
 19      * Widget options.
 20      *
 21      * <dl>
 22      *   <dt>map</dt>
 23      *   <dd>
 24      *     Optional map instance, if not provided one will be looked up. Defaults to <code>''</code>.
 25      *   </dd>
 26      *   <dt>mapMoveEvents</dt>
 27      *   <dd>
 28      *     Provides an override of which vendor specific map events are used to determine
 29      *     when the position of the map changes. Expects a comma separated list of event names.
 30      *     Defaults to <code>'dragstart,idle,dragend'</code> when <code>idleWhileDragging</code> is
 31      *     enabled, otherwise <code>'idle'</code>.
 32      *   </dd>
 33      *   <dt>idleWhileDragging</dt>
 34      *   <dd>
 35      *     Provided that option <code>mapMoveEvents</code> targets event <code>'idle'</code>, then this option
 36      *     will enable the <code>idle</code> event to trigger a bounds change even while dragging the map.
 37      *     Defaults to <code>false</code>.
 38      *   </dd>
 39      * </dl>
 40      * @name $.ui.simplicityGoogleMapBoundsTracker.options
 41      */
 42     options: {
 43       map: '',
 44       mapMoveEvents: '',
 45       idleWhileDragging: false
 46     },
 47     _create: function () {
 48       this._addClass('ui-simplicity-google-map-bounds-tracker');
 49       this._map = this.options.map !== '' ? this.options.map : this.element.simplicityGoogleMap('map');
 50       this._boundsShapes = [];
 51       this._boundsChangeListeners = {};
 52       this._dragging = false;
 53       if (this.options.mapMoveEvents === '') {
 54         this.options.mapMoveEvents = this.options.idleWhileDragging ? 'idle' : 'dragstart,idle,dragend';
 55       }
 56       $.each(this.options.mapMoveEvents.split(','), $.proxy(function (idx, eventName) {
 57         eventName = $.trim(eventName);
 58         if (eventName !== '') {
 59           var listener = google.maps.event.addListener(this._map, eventName, $.proxy(this._mapBoundsChangeHandler(eventName), this));
 60           this._boundsChangeListeners[eventName] = listener;
 61         }
 62       }, this));
 63     },
 64     /**
 65      * Return the actual map object.
 66      *
 67      * @name $.ui.simplicityGoogleMapBoundsTracker.map
 68      * @function
 69      */
 70     map: function () {
 71       return this._map;
 72     },
 73     _mapBoundsChangeHandler: function (eventName) {
 74       return function () {
 75         if (!this.options.idleWhileDragging) {
 76           if (eventName === 'dragstart') {
 77             this._dragging = true;
 78             return;
 79           } else if (eventName === 'dragend') {
 80             this._dragging = false;
 81           } else if (this._dragging) {
 82             return;
 83           }
 84         }
 85         var bounds = this.bounds();
 86         if (!bounds.bounds.vendor.equals(this._lastBounds)) {
 87           this._lastBounds = bounds.bounds.vendor;
 88           if (this.options.debug) {
 89             console.log('simplicityGoogleMapBoundsTracker: Bounds changed', bounds);
 90           }
 91           this._trigger('bounds', {}, bounds);
 92         }
 93       };
 94     },
 95     /**
 96      * Returns the normalized bounds for this map.
 97      *
 98      * @name $.ui.simplicityGoogleMapBoundsTracker.bounds
 99      * @function
100      */
101     bounds: function () {
102       return this.normalizeBounds(this._map.getBounds(), this._map.getCenter());
103     },
104     /**
105      * Normalizes the bounds for this map.
106      *
107      * @param bounds in vendor supplied format
108      * @param center point in vendor supplied format
109      * @name $.ui.simplicityGoogleMapBoundsTracker.normalizeBounds
110      * @function
111      * @private
112      */
113     normalizeBounds: function (bounds, center) {
114       var result = {
115         map: this._map,
116         bounds: {
117           vendor: bounds,
118           north: bounds.getNorthEast().lat(),
119           east: bounds.getNorthEast().lng(),
120           south: bounds.getSouthWest().lat(),
121           west: bounds.getSouthWest().lng()
122         },
123         center: {
124           vendor: center,
125           latitude: center.lat(),
126           longitude: center.lng()
127         }
128       };
129       var radiusMeters1;
130       var radiusMeters2;
131       if ('undefined' !== typeof google.maps.geometry) {
132         radiusMeters1 = google.maps.geometry.spherical.computeDistanceBetween(
133             center, new google.maps.LatLng(center.lat(), bounds.getNorthEast().lng()));
134         radiusMeters2 = google.maps.geometry.spherical.computeDistanceBetween(
135             center, new google.maps.LatLng(bounds.getNorthEast().lat(), center.lng()));
136       } else {
137         radiusMeters1 = $.simplicityGeoFn.distanceKm(
138             center.lat(), center.lng(), center.lat(), bounds.getNorthEast().lng()) * 1000.0;
139         radiusMeters2 = $.simplicityGeoFn.distanceKm(
140             center.lat(), center.lng(), bounds.getNorthEast().lat(), center.lng()) * 1000.0;
141       }
142       var minMeters = Math.min(radiusMeters1, radiusMeters2);
143       var maxMeters = Math.max(radiusMeters1, radiusMeters2);
144       $.extend(result, {
145         minRadius: {
146           meters: minMeters,
147           feet: minMeters * 3.2808399,
148           miles:  minMeters / 1609.344,
149           km: minMeters / 1000
150         },
151         maxRadius: {
152           meters: maxMeters,
153           feet: maxMeters * 3.2808399,
154           miles:  maxMeters / 1609.344,
155           km: maxMeters / 1000
156         }
157       });
158       return result;
159     },
160     /**
161      * Removes the bounds from the map.
162      *
163      * @name $.ui.simplicityGoogleMapBoundsTracker.hideBounds
164      * @function
165      * @private
166      */
167     hideBounds: function () {
168       $.each(this._boundsShapes, function (idx, shape) {
169         shape.setMap(null);
170       });
171       this._boundsShapes.length = 0;
172     },
173     /**
174      * Adds an overlay for the bounds on the map.
175      *
176      * @param bounds Optional bounds to display, if missing the current bounds are used.
177      * @name $.ui.simplicityGoogleMapBoundsTracker.showBounds
178      * @function
179      * @private
180      */
181     showBounds: function (bounds) {
182       if ('undefined' === typeof bounds) {
183         bounds = this.bounds();
184       }
185       var shapes = {
186         boundsRect: new google.maps.Rectangle({
187           bounds: bounds.bounds.vendor,
188           fillOpacity: 0
189         }),
190         minCircle: new google.maps.Circle({
191           center: bounds.center.vendor,
192           radius: bounds.minRadius.meters,
193           fillOpacity: 0
194         }),
195         maxCircle: new google.maps.Circle({
196           center: bounds.center.vendor,
197           radius: bounds.maxRadius.meters,
198           fillOpacity: 0
199         }),
200         centerLatitude: centerLat = new google.maps.Polyline({
201           path: [
202             new google.maps.LatLng(bounds.center.latitude, bounds.bounds.east),
203             new google.maps.LatLng(bounds.center.latitude, bounds.bounds.west)
204           ],
205           strokeWeight: 1
206         }),
207         centerLongitude: new google.maps.Polyline({
208           path: [
209             new google.maps.LatLng(bounds.bounds.north, bounds.center.longitude),
210             new google.maps.LatLng(bounds.bounds.south, bounds.center.longitude)
211           ],
212           strokeWeight: 1
213         })
214       };
215       this._trigger('boundsshapes', {}, shapes);
216       $.each(shapes, $.proxy(function (name, shape) {
217         shape.setMap(this._map);
218         this._boundsShapes.push(shape);
219       }, this));
220     },
221     destroy: function () {
222       $.each(this._boundsChangeListeners, $.proxy(function (eventName, listener) {
223         google.maps.event.removeListener(listener);
224       }, this));
225       this._boundsChangeListeners = {};
226       $.ui.simplicityWidget.prototype.destroy.apply(this, arguments);
227     }
228   });
229 }(jQuery));
