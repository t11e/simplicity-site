1 /**
  2  * @name $.simplicity.ajaxHelper
  3  * @function
  4  * @description
  5  *
  6  * Plugin to manage asynchronous ajax requests such that only the most
  7  * recent request is currently inflight. Any inflight request will be
  8  * aborted, leaving only the most recent request to complete.
  9  *
 10  * The helper class also keeps track of total requests, success and error
 11  * counts. These statistics are available using the .getStats() method.
 12  *
 13  * @param exeContext
 14  *   The target <code>Object</code> to use for the context of the supplied
 15  *   $.ajax() success and error callback options.
 16  *
 17  * @example
 18  *   var ajaxHelper = new $.simplicity.ajaxHelper(this);
 19  *   var xhr = ajaxHelper.ajax({
 20  *     url: '/my_search_controller.php',
 21  *     type: 'GET',
 22  *     contentType: 'application/json',
 23  *     error: function (data, textStatus, xhr) {...},
 24  *     success: function (data, textStatus, xhr) {...},
 25  *     debug: false
 26  *   });
 27  *   console.log(ajaxHelper.getStats());
 28  */
 29 (function ($) {
 30   $.simplicity = $.simplicity || {};
 31   $.simplicity.ajaxHelper = function (exeContext) {
 32     var numRequest = 0,
 33         numSuccess = 0,
 34         numError = 0,
 35         numSuperceed = 0,
 36         numAbort = 0,
 37         inFlightXhr = null;
 38     var ajax = function (options) {
 39       function requestSuccess(requestId) {
 40         return function (data, textStatus, xhr) {
 41           inFlightXhr = null;
 42           if (requestId !== numRequest) {
 43             if (options.debug) {
 44               console.log('Successful Ajax response superceded with requestId ', requestId, ', current request', numRequest);
 45             }
 46           } else {
 47             if (data === null || ('undefined' === typeof data)) {
 48               numError += 1;
 49               if (options.error !== null && ('undefined' !== typeof options.error)) {
 50                 options.error.apply(this, arguments);
 51               }
 52             } else {
 53               numSuccess += 1;
 54               if (options.success !== null && ('undefined' !== typeof options.success)) {
 55                 options.success.apply(this, arguments);
 56               }
 57             }
 58           }
 59         };
 60       }
 61       function requestError(requestId) {
 62         return function (data, textStatus, xhr) {
 63           inFlightXhr = null;
 64           if (xhr.status === 0 && textStatus === 'abort') {
 65             numAbort += 1;
 66             if (options.debug) {
 67               console.log('Ajax call aborted with requestId ', requestId, ', current request', numRequest);
 68             }
 69           } else if (requestId !== numRequest) {
 70             numSuperceed += 1;
 71             if (options.debug) {
 72               console.log('Ajax error response superceeded with requestId ', requestId, ', current request', numRequest);
 73             }
 74           } else {
 75             numError += 1;
 76             if (options.error !== null && ('undefined' !== typeof options.error)) {
 77               options.error.apply(this, arguments);
 78             }
 79           }
 80         };
 81       }
 82       numRequest += 1;
 83       if (inFlightXhr) {
 84         inFlightXhr.abort();
 85       }
 86       var callbacks = {
 87         success: $.proxy(requestSuccess(numRequest), exeContext),
 88         error: $.proxy(requestError(numRequest), exeContext)
 89       };
 90       var ajaxOptions = $.extend({}, options, callbacks);
 91       inFlightXhr = $.ajax(ajaxOptions);
 92       return inFlightXhr;
 93     };
 94     return {
 95       ajax: ajax,
 96       getStats: function () {
 97         return {
 98           count: numRequest,
 99           success: numSuccess,
100           error: numError,
101           superceed: numSuperceed,
102           abort: numAbort
103         };
104       }
105     };
106   };
107 }(jQuery));
