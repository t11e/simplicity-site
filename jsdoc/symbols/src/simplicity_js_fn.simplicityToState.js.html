1 /**
  2  * @name $.fn.simplicityToState
  3  * @function
  4  * @description
  5  *
  6  * Plugin to extract state from input elements.
  7  * Apply this to input elements to extract their state into a dictionary.
  8  *
  9  * @param state
 10  *   The target <code>Object</code> to extract the state to.
 11  * @param trimValue
 12  *   The state value will be trimmed if this is <code>true</code>.
 13  * @param ignoreEmptyValues
 14  *   The state value will be ignored if it is empty and this is <code>true</code>. Useful
 15  *   for applying state on initial page load with multiple inputs mapped to the same fields.
 16  *
 17  * @example
 18  *   var state = {};
 19  *   $(':input').simplicityToState(state);
 20  *   console.log(state);
 21  */
 22 (function ($) {
 23   $.fn.simplicityToState = function (state, trimValue, ignoreEmptyValues) {
 24     return this.each(function () {
 25       var element = $(this);
 26       var name = $.trim(element.attr('name'));
 27       if (name !== '') {
 28         var value = element.val();
 29         if (element.is(':checkbox')) {
 30           var checked = element[0].checked;
 31           if (checked || !ignoreEmptyValues) {
 32             var values = state[name];
 33             if ('undefined' === typeof values || values === null || values === '') {
 34               values = [];
 35             } else if (!$.isArray(values)) {
 36               values = [values];
 37             }
 38             if (checked) {
 39               if (-1 === $.inArray(value, values)) {
 40                 values.push(value);
 41               }
 42             } else {
 43               values = $.grep(values, function (elem, idx) {
 44                 return elem !== value;
 45               });
 46             }
 47             if (values.length === 0) {
 48               delete state[name];
 49             } else if (values.length === 1) {
 50               state[name] = values[0];
 51             } else {
 52               state[name] = values;
 53             }
 54           }
 55         } else if (element[0].nodeName === 'SELECT') {
 56           if ('undefined' === typeof value || value === '' || value === null) {
 57             value = [];
 58           }
 59           if ($.isArray(value)) {
 60             value = $.grep(value, function (elem) {
 61               return elem !== '';
 62             });
 63             if (value.length === 1) {
 64               value = value[0];
 65             }
 66           }
 67           if (value.length > 0) {
 68             state[name] = value;
 69           } else if (!ignoreEmptyValues) {
 70             delete state[name];
 71           }
 72         } else if (element.is(':radio') && !element[0].checked) {
 73           // Ignore change events from unchecked radio buttons
 74         } else {
 75           // Input other than checkbox or select
 76           if ('undefined' === typeof value || value === null) {
 77             value = '';
 78           }
 79           if (trimValue) {
 80             value = $.trim(value);
 81           }
 82           if (value !== '') {
 83             state[name] = value;
 84           } else if (!ignoreEmptyValues) {
 85             delete state[name];
 86           }
 87         }
 88       }
 89     });
 90   };
 91 }(jQuery));
