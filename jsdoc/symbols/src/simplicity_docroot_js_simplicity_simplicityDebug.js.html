1 /**
  2  * @name $.ui.simplicityDebug
  3  * @namespace Displays text areas for the various event payloads in editable format.
  4  * @private
  5  */
  6 (function ($) {
  7   $.widget("ui.simplicityDebug", $.ui.simplicityWidget, {
  8     options : {
  9       stateElement: 'body',
 10       searchElement: 'body',
 11       template: '<div class="yui3-u-1-2"><fieldset class="ui-widget ui-widget-content"><legend/><textarea cols="68" rows="30"/></fieldset></div>'
 12     },
 13     _create : function () {
 14       this._addClass('ui-simplicity-debug');
 15       this.element.children().remove();
 16 
 17       this
 18         ._bind(this.options.stateElement, 'simplicityStateChange', this._stateChangeHandler)
 19         ._bind(this.options.searchElement, 'simplicitySearchResponse', this._searchResponseHandler);
 20       this._addChild('State', '_state', this._stateChanger);
 21       this._addChild('SearchResponse', '_searchResponse', this._searchResponseChanger);
 22     },
 23     _addChild: function (label, name, handler) {
 24       var state = $(this.options.template);
 25       state.find('legend').text(label);
 26       this[name] = state.find('textarea');
 27       this._bind(this[name], 'change', handler);
 28       this.element.append(state);
 29     },
 30     _stateChangeHandler: function (evt, state) {
 31       this._state.val(JSON.stringify(state, null, '  '));
 32     },
 33     _stateChanger: function (evt, ui) {
 34       $(this.options.stateElement).simplicityState('state', JSON.parse($(evt.target).val()));
 35     },
 36     _searchResponseHandler: function (evt, response) {
 37       this._searchResponse.text(JSON.stringify(response, null, '  '));
 38     },
 39     _searchResponseChanger: function (evt, ui) {
 40       $(this.options.searchElement).simplicityDiscoverySearch('searchResponse', JSON.parse($(evt.target).val()));
 41     },
 42     destroy: function () {
 43       this.element.children().remove();
 44       $.ui.simplicityWidget.prototype.destroy.apply(this, arguments);
 45     }
 46   });
 47 }(jQuery));
