1 /**
  2  * @name $.ui.simplicityMapQuestMap
  3  * @namespace A MapQuest map
  4  * <p>
  5  * MapQuest Map widget. Wraps MapQuest a map as a jquery ui widget. Will optionally create a MapQuest map, or a map can be
  6  * passed in using the widget options.
  7  *
  8  * @example
  9  *   <div id="map" style="width: 300px; height: 300px;"></div>
 10  *   <script type="text/javascript">
 11  *     $('#map').simplicityMapQuestMap();
 12  *   </script>
 13  *
 14  * @see MapQuest JavaScript SDK v6 <a href="http://platform.beta.mapquest.com/sdk/js/v6.0.0/">documentation</a>.
 15  */
 16 (function ($) {
 17   $.widget("ui.simplicityMapQuestMap", $.ui.simplicityWidget, {
 18     /**
 19      * Widget options.
 20      *
 21      * <dl>
 22      *   <dt>map</dt>
 23      *   <dd>
 24      *     Optional map instance, if not provided one will be created. Defaults to <code>''</code>.
 25      *   </dd>
 26      *   <dt>mapOptions</dt>
 27      *   <dd>
 28      *     Options used when creating the map. Defaults to <code>''</code> which is expanded at
 29      *     runtime to
 30      *     <pre>
 31      *     {
 32      *       center: {lat: 0, lng: 0},
 33      *       zoom: 1,
 34      *       mapType: 'map'
 35      *     }
 36      *     </pre>
 37      *     Can be either an <code>Object</code> or a <code>function</code>.
 38      *   </dd>
 39      * </dl>
 40      * @name $.ui.simplicityMapQuestMap.options
 41      */
 42     options : {
 43       map: '',
 44       mapOptions: '',
 45       // The following options are for internal use only
 46       apiKey: '',
 47       mapVersion: '6.0.0'
 48     },
 49     _create: function () {
 50       this._addClass('ui-simplicity-mapquest-map');
 51       if (this.options.map !== '') {
 52         this._map = this.options.map;
 53       } else {
 54         this._initWhenAvailable();
 55       }
 56     },
 57     /**
 58      * Lazy initialization method used to create the map only when the necessary JavaScript
 59      * is available. Intended to be called from any of this widgets public methods that need to
 60      * access the this._map.
 61      *
 62      * @name $.ui.simplicityMapQuestMap._initWhenAvailable
 63      * @function
 64      * @private
 65      */
 66     _initWhenAvailable: function () {
 67       var wasAvailable = 'undefined' !== typeof this._map;
 68       if (wasAvailable) {
 69         // Already available, do nothing
 70       } else if (this.options.map !== '') {
 71         this._map = this.options.map;
 72       } else if ('undefined' !== typeof MQA && 'undefined' !== typeof MQA.TileMap) {
 73         var defaultMapOptions = {
 74           center: {lat: 0, lng: 0},
 75           zoom: 1,
 76           mapType: 'map'
 77         };
 78         var mapOptions;
 79         if (this.options.mapOptions === '') {
 80           mapOptions = defaultMapOptions;
 81         } else if ($.isFunction(this.options.mapOptions)) {
 82           mapOptions = $.extend(defaultMapOptions, this.options.mapOptions.call(this));
 83         } else {
 84           mapOptions = $.extend(defaultMapOptions, this.options.mapOptions);
 85         }
 86         this._map = new MQA.TileMap(this.element[0],
 87             mapOptions.zoom, mapOptions.center, mapOptions.mapType);
 88         this._trigger('create', {}, {
 89           map: this._map
 90         });
 91       }
 92       var isAvailable = 'undefined' !== typeof this._map;
 93       return isAvailable;
 94     },
 95     /**
 96      * Return the actual map object.
 97      *
 98      * @name $.ui.simplicityMapQuestMap.map
 99      * @function
100      */
101     map: function () {
102       this._initWhenAvailable();
103       return this._map;
104     },
105     /**
106      * Try to dynamically load the necessary JavaScript from the upstream vendor that will
107      * allow the map to function.
108      *
109      * @name $.ui.simplicityMapQuestMap.loadMap
110      * @function
111      * @private
112      */
113     loadMap: function () {
114       if ('undefined' === typeof this._map) {
115         if ('undefined' === typeof MQA || 'undefined' === typeof MQA.TileMap) {
116           var src = 'http://mapquestapi.com/sdk/js/v' + this.options.mapVersion + '/mqa.toolkit.js';
117           if ('undefined' === typeof $.simplicityLoadJs) {
118             src = src + '?key=YOUR_API_KEY';
119             alert('Dynamic loading of MapQuest maps is not supported. Enable this widget by adding the following script tag to your page:\n\n' +
120               '<script src="http://mapquestapi.com/sdk/js/v6.0.0/mqa.toolkit.js?key=YOUR_KEY"></script>');
121           } else {
122             if (this.options.apiKey !== '') {
123               src = src + '?key=' + this.options.apiKey;
124             }
125             $.simplicityLoadJs(src, $.proxy(function () {
126               MQA.withModule('mapinit', $.proxy(function () {
127                 this.refreshMap();
128               }, this));
129             }, this));
130           }
131         }
132       }
133     },
134     destroy: function () {
135       delete this._map;
136       $.ui.simplicityWidget.prototype.destroy.apply(this, arguments);
137     }
138   });
139 }(jQuery));
