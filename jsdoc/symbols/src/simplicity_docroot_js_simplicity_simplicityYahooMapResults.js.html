1 /**
  2  * @name $.ui.simplicityYahooMapResults
  3  * @namespace A Yahoo! map.
  4  * <p>
  5  * Widget that listens for <code>simplicitySearchResponse</code> events which it uses to add markers to the map for the search results.
  6  *
  7  * @example
  8  *   <div id="map" style="width: 300px; height: 300px;"></div>
  9  *   <script type="text/javascript">
 10  *     $('#map').simplicityYahooMap();
 11  *   </script>
 12  *
 13  * @see Yahoo! Maps Web Services - AJAX API <a href="http://developer.yahoo.com/maps/ajax/">documentation</a>.
 14  */
 15 (function ($) {
 16   $.widget("ui.simplicityYahooMapResults", $.ui.simplicityWidget, {
 17     /**
 18      * Widget options.
 19      *
 20      * <dl>
 21      *   <dt>searchElement</dt>
 22      *   <dd>
 23      *     The simplicityDiscoverySearch widget that this widget binds it's events to. Defaults to <code>'body'</code>.
 24      *   </dd>
 25      *   <dt>latitudeField</dt>
 26      *   <dd>
 27      *     Field to find the latitude of the result item in the <code>simplicityResultSet</code>
 28      *     item properties. Defaults to <code>'latitude'</code>.
 29      *   </dd>
 30      *   <dt>longitudeField</dt>
 31      *   <dd>
 32      *     Field to find the longitude of the result item in the <code>simplicityResultSet</code>
 33      *     item properties. Defaults to <code>'longitude'</code>.
 34      *   </dd>
 35      *   <dt>map</dt>
 36      *   <dd>
 37      *     Optional map instance, if not provided one will be created. Defaults to <code>''</code>.
 38      *   </dd>
 39      *   <dt>updateBounds<dt>
 40      *   <dd>
 41      *     When true the map is panned and zoomed to best fit the search
 42      *     results that are added as part of the <code>simplicitySearchResponse</code>
 43      *     event handler. Defaults to <code>true</code>.
 44      *   </dd>
 45      * </dl>
 46      * @name $.ui.simplicityYahooMapResults.options
 47      */
 48     options : {
 49       searchElement: 'body',
 50       latitudeField: 'latitude',
 51       longitudeField: 'longitude',
 52       map: '',
 53       updateBounds: true
 54     },
 55     _create: function () {
 56       this._addClass('ui-simplicity-yahoo-map-results');
 57       this._map = this.options.map !== '' ? this.options.map : this.element.simplicityYahooMap('map');
 58       this._markers = [];
 59       this._locations = [];
 60       this
 61         ._bind(this.options.searchElement, 'simplicitySearchResponse', this._resultSetHandler)
 62         ._bind('simplicityyahoomapboundscoordinatorcalculatebounds', this._calcBoundsHandler);
 63     },
 64     /**
 65      * Return the actual map object.
 66      *
 67      * @name $.ui.simplicityYahooMapResults.map
 68      * @function
 69      */
 70     map: function () {
 71       return this._map;
 72     },
 73     /**
 74      * Makes the widget re-handle the last <code>simplicityResultSet</code> event to reapply
 75      * any map markers.
 76      *
 77      * @name $.ui.simplicityYahooMapResults.refreshMap
 78      * @function
 79      */
 80     refreshMap: function () {
 81       this.removeMarkers();
 82       this.addMarkers();
 83     },
 84     /**
 85      * Event handler for the <code>simplicitySearchResponse</code> event. Extracts the coordinates
 86      * of each result item by using the property fields defined by the
 87      * <code>latitudeField</code> and <code>longitudeField</code> options of this widget and
 88      * places a marker on the map for each valid coordinate. The map is then reset to best
 89      * display the current set of markers.
 90      *
 91      * @name $.ui.simplicityYahooMapResults._resultSetHandler
 92      * @function
 93      * @private
 94      */
 95     _resultSetHandler: function (evt, searchResponse) {
 96       this.removeMarkers();
 97       this.addMarkers(searchResponse);
 98     },
 99     _calcBoundsHandler: function (evt, ui) {
100       if ($.isArray(ui.locations) && this.options.updateBounds) {
101         $.merge(ui.locations, this._locations);
102       }
103     },
104     /**
105      * Removes any markers that were added to the map by <code>addMarkers</code>.
106      *
107      * @name $.ui.simplicityYahooMapResults.removeMarkers
108      * @function
109      * @private
110      */
111     removeMarkers: function () {
112       $.each(this._markers, $.proxy(function (idx, marker) {
113         var eventData = {
114           map: this._map,
115           marker: marker
116         };
117         this._trigger('removemarker', {}, eventData);
118         marker = eventData.marker;
119         if ('undefined' !== typeof marker) {
120           this._map.removeOverlay(marker);
121         }
122       }, this));
123       this._markers.length = 0;
124       this._locations.length = 0;
125     },
126     /**
127      * Adds any markers that can be extracted from the given <code>searchResponse</code>.
128      *
129      * @name $.ui.simplicityYahooMapResults.addMarkers
130      * @function
131      * @private
132      */
133     addMarkers: function (searchResponse) {
134       if ('undefined' === typeof searchResponse) {
135         searchResponse = $(this.options.searchElement).simplicityDiscoverySearch('searchResponse');
136       }
137       $.fn.simplicityDiscoverySearchItemEnumerator(searchResponse, $.proxy(function (idx, row) {
138         var properties = row.properties;
139         if ('undefined' !== typeof properties) {
140           var latitude = properties[this.options.latitudeField];
141           var longitude = properties[this.options.longitudeField];
142           if ('undefined' !== typeof latitude && 'undefined' !== typeof longitude) {
143             latitude = Number(latitude);
144             longitude = Number(longitude);
145             var point = new YGeoPoint(latitude, longitude);
146             var marker = new YMarker(point);
147             var eventData = {
148               row: row,
149               map: this._map,
150               marker: marker
151             };
152             this._trigger('marker', {}, eventData);
153             marker = eventData.marker;
154             if ('undefined' !== typeof marker) {
155               this._locations.push(point);
156               this._markers.push(marker);
157               this._map.addOverlay(marker);
158             }
159           }
160         }
161       }, this));
162     }
163   });
164 }(jQuery));
