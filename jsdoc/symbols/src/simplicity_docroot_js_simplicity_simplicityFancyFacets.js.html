1 /**
  2  * @name $.ui.simplicityFancyFacets
  3  * @namespace Composed variant of simplicityFancySelect with support for a flyout and selected area.
  4  *
  5  * A more complex extension of simplicityFancySelect that supports a flyout
  6  * for extra options as well as a selected area for the current selection.
  7  *
  8  * @example
  9  *   <select id="mySelect" name="example">
 10  *     <option value="1">one</option>
 11  *     <option value="2">two</option>
 12  *     <option value="3">three</option>
 13  *   <select>
 14  *   <div id="myFancy"></div>
 15  *   <script type="text/javascript">
 16  *     $('#myFancy')
 17  *       .simplicityFancyFacets({
 18  *         select: '#mySelect',
 19  *       })
 20  *      .hide();
 21  *   </script>
 22  */
 23 (function ($) {
 24   $.widget("ui.simplicityFancyFacets", $.ui.simplicityWidget, {
 25      /**
 26      * Widget options.
 27      *
 28      * <dl>
 29      *   <dt>select</dt>
 30      *   <dd>
 31      *     Mandatory option that points to the <code>select</code> input to be enhanced. Can be a DOM node, jQuery node or selector.
 32      *   </dd>
 33      *   <dt>selectedArea</dt>
 34      *   <dd>
 35      *     When <code>true</code> causes selected options to appear in the selected area.
 36      *     Defaults to <code>false</code>.
 37      *   </dd>
 38      *   <dt>availableContainsSelected</dt>
 39      *   <dd>
 40      *     When <code>true</code> causes selected options to appear in the available area.
 41      *     Defaults to <code>true</code>.
 42      *   </dd>
 43      *   <dt>overflowContainsAvailable</dt>
 44      *   <dd>
 45      *     When <code>true</code> causes availble options to appear in the overflow area.
 46      *     Defaults to <code>false</code>.
 47      *   </dd>
 48      *   <dt>overflowContainsSelected</dt>
 49      *   <dd>
 50      *     When <code>true</code> causes selected options to appear in the overflow area.
 51      *     Defaults to <code>true</code>.
 52      *   </dd>
 53      *   <dt>availableLimit</dt>
 54      *   <dd>
 55      *     Determines how many options should appear in the available area. A negative number
 56      *     indicates no limit.
 57      *     Defaults to <code>5</code>.
 58      *   </dd>
 59      *   <dt>hideWhenEmpty</dt>
 60      *   <dd>
 61      *     When <code>true</code> the target element is hidden if the total number of
 62      *     options drops to zero and shown when it is larger.
 63      *     Defaults to <code>false</code>.
 64      *   </dd>
 65      *   <dt>template</dt>
 66      *   <dd>
 67      *     When <code>''</code> falls back on the original DOM contents of the attached element.
 68      *     Replaces the contents otherwise.
 69      *     See source code for default value.
 70      *   </dd>
 71      *   <dt>availableTemplate</dt>
 72      *   <dd>
 73      *     The default template to use for the available facets block of <code>simplicityFancySelect</code>,
 74      *     (selector '.available-container'). Also acts as the default for <code>selectedTemplate</code>
 75      *     and <code>overflowTemplate</code>.
 76      *     <p>Default:</p>
 77      *     <pre>
 78      *<ul class="options ui-helper-clearfix">
 79      *  <li class="option ui-helper-clearfix">
 80      *    <a href="#" class="label"/>
 81      *    <span class="count"/>
 82      *  </li>
 83      *</ul></pre>
 84      *   </dd>
 85      *   <dt>selectedTemplate</dt>
 86      *   <dd>
 87      *     When <code>''</code> falls back to the value for option <code>availableTemplate</code>.
 88      *     Otherwise it determines the template to use for <code>simplicityFancySelect</code>
 89      *     when rendering the selected facets (selector '.selected-container'). Default is <code>''</code>.
 90      *   </dd>
 91      *   <dt>overflowTemplate</dt>
 92      *   <dd>
 93      *     When <code>''</code> falls back to the value for option <code>availableTemplate</code>.
 94      *     Otherwise it determines the template to use for <code>simplicityFancySelect</code>
 95      *     when rendering the overflow facets (selector '.overflow-container'). Default is <code>''</code>.
 96      *   </dd>
 97      *   <dt>pathTemplate</dt>
 98      *   <dd>
 99      *     When <code>''</code> falls back to the value for option <code>availableTemplate</code>.
100      *     Otherwise it determines the template to use for <code>simplicityFancySelect</code>
101      *     when rendering the facets navigation path (selector '.path-container'). Default is <code>''</code>.
102      *   </dd>
103      * </dl>
104      * @name $.ui.simplicityFancyFacets.options
105      */
106     options : {
107       select: '',
108       selectedArea: false,
109       availableContainsSelected: true,
110       overflowContainsAvailable: false,
111       overflowContainsSelected: true,
112       availableLimit: 5,
113       hideWhenEmpty: false,
114       template:
115         '<div class="path-container"/>' +
116         '<div class="selected-container"/>' +
117         '<div class="available-container"/>' +
118         '<div class="overflow-opener"><a href="#more">More…</a></div>' +
119         '<div class="overflow-closer"><a href="#less">Less…</a></div>' +
120         '<div class="overflow-flyout">' +
121           '<div class="overflow-container"/>' +
122           '<div class="overflow-closer">Less…</div>' +
123         '</div>',
124       availableTemplate: '' +
125         '<ul class="options ui-helper-clearfix">' +
126           '<li class="option ui-helper-clearfix">' +
127             '<a href="#" class="label"/>' +
128             '<span class="count"/>' +
129           '</li>' +
130         '</ul>',
131       selectedTemplate: '',
132       overflowTemplate: '',
133       pathTemplate: ''
134     },
135     _create : function () {
136       var select = $(this.options.select);
137       if (select.length === 0 || !select.is('select')) {
138         return;
139       }
140       this._addClass('ui-simplicity-fancy-facets');
141       if (this.options.template !== '') {
142         this.element.children().remove();
143         $(this.options.template).appendTo(this.element);
144       }
145       this.element.find('.path-container')
146         .simplicityFancySelect({
147           select: select,
148           firstPathOnly: true,
149           firstPathSelections: true,
150           hideWhenEmpty: true,
151           template: this.options.pathTemplate || this.options.availableTemplate
152         });
153       this.element.find('.selected-container')
154         .simplicityFancySelect({
155           select: select,
156           refreshOnCreate: false,
157           ignorePath: true,
158           firstPathSelections: true,
159           template: this.options.selectedTemplate || this.options.availableTemplate
160         });
161       this.element.find('.available-container')
162         .simplicityFancySelect({
163           select: select,
164           refreshOnCreate: false,
165           ignorePath: true,
166           firstPathSelections: true,
167           template: this.options.overflowTemplate || this.options.availableTemplate
168         });
169       this.element.find('.overflow-container')
170         .simplicityFancySelect({
171           select: select,
172           refreshOnCreate: false,
173           ignorePath: true,
174           firstPathSelections: true,
175           template: this.options.availableTemplate
176         });
177       // This change handler must be bound *after* the contained widgets are
178       // created so we can be sure it is called after their change handlers.
179       this._bind(select, 'change', this._changeHandler);
180       this.element.find('.overflow-flyout').simplicityFlyout();
181       this._overflowOpen = false;
182       var overflowOpener = this.element.find('.overflow-opener');
183       var overflowCloser = this.element.find('.overflow-closer');
184       overflowOpener.hide();
185       overflowCloser.hide();
186       this
187         ._bind(overflowOpener, 'click', this._overflowOpenerHandler)
188         ._bind(overflowCloser, 'click', this._overflowCloserHandler);
189       this._applyOptions();
190     },
191     _changeHandler: function (evt) {
192       if (this.options.hideWhenEmpty) {
193         if (this.size() > 0) {
194           this.element.show();
195         } else {
196           this.element.hide();
197         }
198       }
199       var overflowOpener = this.element.find('.overflow-opener');
200       var overflowCloser = this.element.find('.overflow-closer');
201       if (this.element.find('.overflow-container').simplicityFancySelect('size') > 0) {
202         overflowOpener.addClass('overflow-contains-selected');
203         overflowCloser.addClass('overflow-contains-selected');
204         if (!this._overflowOpen) {
205           this.closeOverflow(true);
206         }
207       } else {
208         overflowOpener.removeClass('overflow-contains-selected');
209         overflowCloser.removeClass('overflow-contains-selected');
210         this.closeOverflow(true);
211       }
212     },
213     /**
214      * Returns the target <code>select</code> input.
215      *
216      * @name $.ui.simplicityFancyFacets.select
217      * @function
218      */
219     select: function () {
220       return $(this.options.select);
221     },
222     /**
223      * Returns the total number of options currently displayed.
224      *
225      * @name $.ui.simplicityFancyFacets.size
226      * @function
227      */
228     size: function () {
229       var size = 0;
230       this.element.find('.path-container, .selected-container, .available-container, .overflow-container').each(function (idx, elem) {
231         size += $(elem).simplicityFancySelect('size');
232       });
233       return size;
234     },
235     _overflowOpenerHandler: function (evt) {
236       this.openOverflow();
237       evt.preventDefault();
238     },
239     _overflowCloserHandler: function (evt) {
240       this.closeOverflow();
241       evt.preventDefault();
242     },
243     /**
244      * Opens the overflow if it is closed.
245      *
246      * @name $.ui.simplicityFancyFacets.openOverflow
247      * @function
248      */
249     openOverflow: function () {
250       if (!this._overflowOpen) {
251         this._overflowOpen = true;
252         this.element.find('.overflow-opener').hide();
253         this.element.find('.overflow-closer').show();
254         this.element.find('.overflow-flyout').simplicityFlyout('open');
255       }
256     },
257     /**
258      * Closes the overflow if it is open.
259      *
260      * @name $.ui.simplicityFancyFacets.closeOverflow
261      * @function
262      */
263     closeOverflow: function (force) {
264       if (this._overflowOpen || force) {
265         this._overflowOpen = false;
266         var container = this.element.find('.overflow-container');
267         var opener = this.element.find('.overflow-opener');
268         var closer = this.element.find('.overflow-closer');
269         if (container.simplicityFancySelect('size') > 0) {
270           opener.show();
271         } else {
272           opener.hide();
273         }
274         closer.hide();
275         this.element.find('.overflow-flyout').simplicityFlyout('close');
276       }
277     },
278     _setOptions: function () {
279       this._batchOptions = true;
280       try {
281         $.ui.simplicityWidget.prototype._setOptions.apply(this, arguments);
282       } finally {
283         this._batchOptions = false;
284         this._applyOptions();
285       }
286     },
287     _setOption: function (option, value) {
288       $.ui.simplicityWidget.prototype._setOption.apply(this, arguments);
289       if (!this._batchOptions) {
290         this._applyOptions();
291       }
292     },
293     _applyOptions: function () {
294       this.element.find('.selected-container')
295         .simplicityFancySelect('option', {
296           displaySelected: true,
297           displayUnselected: false,
298           displayCount: this.options.selectedArea ? -1 : 0,
299           skipCount: 0,
300           hideWhenEmpty: true,
301           template: this.options.selectedTemplate || this.options.availableTemplate
302         });
303       this.element.find('.available-container')
304         .simplicityFancySelect({
305           displaySelected: this.options.availableContainsSelected,
306           displayUnselected: true,
307           displayCount: this.options.availableLimit,
308           skipCount: 0,
309           template: this.options.availableTemplate
310         });
311       this.element.find('.overflow-container')
312         .simplicityFancySelect({
313           displaySelected: this.options.overflowContainsSelected,
314           displayUnselected: true,
315           displayCount: Number(this.options.availableLimit) === -1 ? 0 : -1,
316           skipSelected: true,
317           skipUnselected: true,
318           skipCount: this.options.overflowContainsAvailable ? 0 : this.options.availableLimit,
319           template: this.options.overflowTemplate || this.options.availableTemplate
320         });
321     },
322     destroy: function () {
323       this.element.children().remove();
324       $.ui.simplicityWidget.prototype.destroy.apply(this, arguments);
325     }
326   });
327 }(jQuery));
