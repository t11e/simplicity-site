1 /**
  2  * @name $.ui.simplicityGoogleMapLoader
  3  * @namespace A Google map.
  4  * <p>
  5  * Google Map widget.
  6  *
  7  * @example
  8  *   <div id="map" style="width: 300px; height: 300px;"></div>
  9  *   <script type="text/javascript">
 10  *     $('#map').simplicityGoogleMapLoader();
 11  *   </script>
 12  *
 13  * @see Google Maps JavaScript API V3 <a href="https://developers.google.com/maps/documentation/javascript/reference">documentation</a>.
 14  */
 15 (function ($, window) {
 16   $.widget("ui.simplicityGoogleMapLoader", $.ui.simplicityWidget, {
 17     /**
 18      * Widget options.
 19      *
 20      * <dl>
 21      * </dl>
 22      * @name $.ui.simplicityGoogleMapLoader.options
 23      */
 24     options: {
 25       apiKey: '',
 26       mapVersion: '3',
 27       mapParams: 'sensor=false'
 28     },
 29     _create: function () {
 30       this._addClass('ui-simplicity-google-map-loader');
 31       this.loadLoader();
 32       this.loadMapsApi();
 33     },
 34     loadLoader: function () {
 35       if ('undefined' === typeof google || 'undefined' === typeof google.load) {
 36         // The Google JS loader is missing, get it
 37         var callbackName = 'simplicityGoogleMapLoaderGoogleLoaderCallback';
 38         window[callbackName] = $.proxy(this.loadMapsApi, this);
 39         var src = 'http://www.google.com/jsapi?callback=' + callbackName;
 40         if (this.options.apiKey !== '') {
 41           src = src + '&key=' + this.options.apiKey;
 42         }
 43         $('head').append($('<script type="text/javascript" src="' + src + '"></script>'));
 44       }
 45     },
 46     loadMapsApi: function () {
 47       if ('undefined' !== typeof google && 'undefined' !== typeof google.load && 'undefined' === typeof google.maps) {
 48         // The Google JS loader is available, use it to load the maps API
 49         google.load('maps', this.options.mapVersion, {
 50           other_params: this.options.mapParams,
 51           callback: $.proxy(function () {
 52             this._trigger('ready', {});
 53           }, this)
 54         });
 55       }
 56     }
 57   });
 58 }(jQuery, window));
