1 /**
  2  * @name $.ui.simplicityGoogleMap
  3  * @namespace A Google map.
  4  * <p>
  5  * Google Map widget. Wraps a Google map as a jquery ui widget. Will optionally create a google map, or a map can be
  6  * passed in using the widget options.
  7  *
  8  * @example
  9  *   <div id="map" style="width: 300px; height: 300px;"></div>
 10  *   <script type="text/javascript">
 11  *     $('#map').simplicityGoogleMap();
 12  *   </script>
 13  *
 14  * @see Google Maps JavaScript API V3 <a href="https://developers.google.com/maps/documentation/javascript/reference">documentation</a>.
 15  */
 16 (function ($) {
 17   $.widget("ui.simplicityGoogleMap", $.ui.simplicityWidget, {
 18     /**
 19      * Widget options.
 20      *
 21      * <dl>
 22      *   <dt>map</dt>
 23      *   <dd>
 24      *     Optional map instance, if not provided one will be created. Defaults to <code>''</code>.
 25      *   </dd>
 26      *   <dt>mapOptions</dt>
 27      *   <dd>
 28      *     Options used when creating the map. Defaults to <code>''</code> which is expanded at
 29      *     runtime to
 30      *     <pre>
 31      *     {
 32      *       center: new google.maps.LatLng(0,0),
 33      *       zoom: 1,
 34      *       mapTypeId: google.maps.MapTypeId.ROADMAP
 35      *     }
 36      *     </pre>
 37      *     Can be either an <code>Object</code> or a <code>function</code>.
 38      *   </dd>
 39      * </dl>
 40      * @name $.ui.simplicityGoogleMap.options
 41      */
 42     options: {
 43       map: '',
 44       mapOptions: ''
 45     },
 46     _create: function () {
 47       this._addClass('ui-simplicity-google-map');
 48       if (this.options.map !== '') {
 49         this._map = this.options.map;
 50       } else {
 51         var defaultMapOptions = {
 52           center: new google.maps.LatLng(0, 0),
 53           zoom: 1,
 54           mapTypeId: google.maps.MapTypeId.ROADMAP
 55         };
 56         var mapOptions;
 57         if (this.options.mapOptions === '') {
 58           mapOptions = defaultMapOptions;
 59         } else if ($.isFunction(this.options.mapOptions)) {
 60           mapOptions = $.extend(defaultMapOptions, this.options.mapOptions.call(this));
 61         } else {
 62           mapOptions = $.extend(defaultMapOptions, this.options.mapOptions);
 63         }
 64         this._map = new google.maps.Map(this.element[0], mapOptions);
 65         this._trigger('create', {}, {
 66           map: this._map
 67         });
 68       }
 69     },
 70     /**
 71      * Return the actual map object.
 72      *
 73      * @name $.ui.simplicityGoogleMap.map
 74      * @function
 75      */
 76     map: function () {
 77       return this._map;
 78     },
 79     destroy: function () {
 80       delete this._map;
 81       $.ui.simplicityWidget.prototype.destroy.apply(this, arguments);
 82     }
 83   });
 84 }(jQuery));
