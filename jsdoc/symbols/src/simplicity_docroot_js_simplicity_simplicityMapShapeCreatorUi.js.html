166     _create: function () {
167       if (this.options.shapeCreator !== '') {
168         this._shapeCreator = this._locateShapeCreator($(this.options.shapeCreator), new RegExp("^simplicity(.+)MapShapeCreator$"));
169       }
170       if (this.options.mapElement !== '' && 'undefined' === typeof this._shapeCreator) {
171         this._shapeCreator = this._locateShapeCreator($(this.options.mapElement), new RegExp("^simplicity(.+)Map$"));
172       }
173       if ('undefined' === typeof this._shapeCreator) {
174         if (this.options.debug) {
175           throw new Error("ShapeCreator could not be determined from either options.shapeCreator or options.mapElement.");
176         }
177         return;
178       }
179       if (!this._shapeCreator("option", "debug") && this.options.debug) {
180         this._shapeCreator("option", "debug", true);
181       }
182       if (this.options.helpPosition === '') {
183         this.options.helpPosition = {left: 300, top: -148};
184       }
185       this._settings_COOKIE = "__t11e_pd_st";
186       this._settings = {};
187       this._settings.helpHidden = false;
188       this._getSettings();
189       this._geoJson = {};
190       var widgetNames = this._shapeCreator("getMappingProviderWidgetNames");
191       this
192         ._addClass('ui-simplicity-map-shape-creator-ui')
193         ._bind(this._shapeCreator, widgetNames.shapeCreator.toLowerCase() + 'shapechange', this._shapeChangedHandler);
194       if (this.options.updateBounds) {
195         this._bind(this._shapeCreator, widgetNames.boundsCoordinator.toLowerCase() + 'calculatebounds', this._shapeCreator("boundsHandler"));
196       }
197       this._createUi();
198       this._shapeCreator("option", "radius", this._radius);
199     },
200     _setOption: function (key, value) {
201       $.ui.simplicityWidget.prototype._setOption.apply(this, arguments);
202       if (key === "geocoderInput") {
203         this._geocoderInput = $(value);
204         this._configureGeocoder();
205       } else if (key === "radiusElement") {
206         this._radiusInput.remove();
207         this._radiusInput = $(value).appendTo(this.element.find('.radius .control'));
208         this._geoJson.properties.radius = this._radius;
209         this._configureRadiusInput();
210       }

212     _locateShapeCreator: function (target, regex) {
213       var candidateWidgetNames = $.map(target.data(), function (value, key) {
214         return regex.test(key) ? key : undefined;
215       });
216       if (candidateWidgetNames.length !== 1) {
217         if (this.options.debug) {
218           if (candidateWidgetNames.length === 0) {
219             console.log('Missing mapping widget ' + regex.source);
220           } else {
221             console.log('Ambigious mapping widget: ' + candidateWidgetNames);
222           }
223         }
224         return;
225       }
226       var match = candidateWidgetNames[0].match(regex);
227       // $1 is in match[1]
228       var widgetName = 'simplicity' + match[1] + 'MapShapeCreator';
229       var widget = target[widgetName];
230       if ('undefined' === typeof target.data(widgetName)) {
231         // instantiate the widget since the element has been set up
232         widget.call(target);
233       }
234       return function () {
235         return widget.apply(target, arguments);
236       };

238     /**
239      * Handler for dealing with changes to the radius.
240      * @name $.ui.simplicityMapShapeCreatorUi._radiusChangedHandler
241      * @function

244     _radiusChangedHandler: function (evt) {

247     },
248     /**
249      * Handler for dealing with changes to the underlying geoJson shape.
250      * @name $.ui.simplicityMapShapeCreatorUi._shapeChangedHandler
251      * @function
252      * @private
253      */
254     _shapeChangedHandler: function (evt, ui) {
255       var geoJson = ui.geoJson;
256       if (geoJson.placemarks.coordinates.length > 0) {
257         if (geoJson.placemarks.type === 'Polygon' && this._mapclicklabel.prop('checked')) {
258           this._disable();
259           if (this._shapeCreator("option", "editMode")) {
260             this._shapeCreator("option", "editMode", false);
261           }
262         }
263         this.element.find('.clear').fadeIn();
264         this._radiusInput.prop("disabled", geoJson.placemarks.type === 'Polygon');
265         if (!this._radiusInput.prop("disabled")) {
266           if (this._radius !== geoJson.properties.radius) {
267             this._radiusInput.val(this._radius);
268             this._radius = this._radiusInput.val();
269             this._geoJson.properties.radius = this._radius;
270           }
271         }
272         this._geocoderInput.val(
273           'undefined' !== typeof geoJson.properties.geocoder ? geoJson.properties.geocoder.addr || '' : '');
274         if (geoJson.properties.editMode) {
275           this._enable();
276         }
277       } else {
278         this.element.find('.clear').fadeOut();
279         this._geocoderInput.val('');
280         this._radiusInput.prop("disabled", false);
281       }
282       this._geoJson = geoJson;
283     },
284     /**
285      * Builds the Ui.
286      * @name $.ui.simplicityMapShapeCreatorUi._createUi
287      * @function
288      * @private
289      */
290     _createUi: function () {
291       this._helpMsg = $(this.options.helpTemplate).draggable().hide().css(this.options.helpPosition);
292       this._helpMsg.find(".background").css({ opacity: 0.5});
293       this._bind(this._helpMsg.find(".closer"), 'click',
294         function (evt) {
295           evt.preventDefault();
296           this._helpButton.fadeIn();
297           this._helpMsg.fadeOut('slow', $.proxy(function () {
298             this._helpMsg.detach();
299             this._settings.helpHidden = true;
300             this._saveSettings();
301           }, this));
302         });
303 
304       this.element
305         .data('previousHTML', this.element.html())
306         .html(this.options.template);
307 
308       this._mapclicklabel = this.element.find('.drawBtn')
309         .prop('checked', false);
310 
311       this._bind(this._mapclicklabel, 'click',
312         function (evt) {
313           if (this._mapclicklabel.prop('checked')) {
314             this._mapclicklabel.prop('checked', false);
315             this._disable();
316             this._shapeCreator("option", "editMode", false);
317             this._trigger('edit', {}, {edit: false});
318           } else {
319             this._mapclicklabel.prop('checked', true);
320             if ('Polygon' === this._shapeCreator("geometry")) {
321               this._shapeCreator("convertPolygonToLineString", this._radius);
322             }
323             this._shapeCreator("option", "editMode", true);
324             this._enable();
325             this._radius = this._radiusInput.val() ? +this._radiusInput.val() : 0;
326             if (this._radius !== this._shapeCreator("option", "radius")) {
327               this._shapeCreator("option", "radius", this._radius);
328             }
329             this._shapeCreator("updateBounds"); // show whole shape
330             this._trigger('edit', {}, {edit: true});
331           }
332         });
333 
334       this._helpButton = this.element.find('.showhelp').hide();
335       this._bind(this._helpButton, 'click',
336         function () {
337           if (this._settings.helpHidden) {
338             this._helpMsg.appendTo(this.element);
339             this._helpMsg.fadeIn('slow');
340             this._helpButton.fadeOut();
341             this._settings.helpHidden = false;
342             this._saveSettings();
343           }
344         });
345 
346       this._bind(this.element.find('.clear').hide(), 'click',
347         function (evt) {
348           this.element.find('.clear').fadeOut();
349           this._shapeCreator("reset");
350           this._geocoderInput.val('');
351         });
352 
353       if (this.options.allowMultiPoint) {
354         this._bind(this.element.find('.linkerBtn'), 'click',
355           function (evt) {
356             var connected = !this._shapeCreator("option", "connected");
357             this.element.find('.linkerBtn').prop('checked', connected);
358             this._setConnected(connected);
359             this._shapeCreator("option", "radius", this._radius);
360             this._shapeCreator("option", "connected", connected);
361           });
362         this.element.find('.linkerBtn').hide();
363       } else {
364         this.element.find('.linkerBtn').remove();
365       }
366 
367       // radius
368       this._radiusInput = $(this.options.radiusElement).appendTo(this.element.find('.radius .control'));
369       this._configureRadiusInput();
370 
371       // geocoder
372       if ('string' === typeof this.options.geocoderInput) {
373         this._geocoderInput = $(this.options.geocoderInput)
374           .autocomplete({
375             autoFocus: true
376           });
377       } else {
378         this._geocoderInput = $(this.options.geocoderInput);
379       }
380       this._configureGeocoder();
381 
382       if (this._shapeCreator("option", "editMode")) {
383         this._enable();
384       }
385       this._setConnected(this._shapeCreator("geometry") !== 'MultiPoint');
386     },
387     /**
388      * Helper method that configures a simplicity radius control.
389      * @name $.ui.simplicityMapShapeCreatorUi._configureRadiusInput
390      * @function

394       if (this._radiusInput.length > 0) {

398         this._radiusInput.appendTo(this.element.find('.radius').children().remove());
399       }

401     /**
402      * Helper method that configures a simplicity geocoder control.
403      *
404      * <p>Triggers a simplicitymapshapecreatoruigeocodedpoint event. Handlers for that event receive a ui
405      * object with geoJson and geocoderCallbackData members.</p>
406      *
407      * @name $.ui.simplicityMapShapeCreatorUi._configureGeocoder
408      * @function
409      * @private
410      */
411     _configureGeocoder: function () {
412       if (this._geocoderInput.length > 0) {
413         this.element.find('.geocoder')
414           .append(this._geocoderInput);
415         var selectCallback = $.proxy(function (evt, ui) {
416           if (ui.item) {
417             var changed = true;
418             try {
419               if ('undefined' !== typeof this._geoJson.properties.geocoder) {
420                 changed = ui.item.value !== this._geoJson.properties.geocoder.addr;
421               }
422             } finally {
423               if (changed) {
424                 this._shapeCreator("reset", false);
425                 this._shapeCreator("option", "radius", this._radius);
426                 var marker = this._shapeCreator("newGeocodedPoint", ui);
427                 this._shapeCreator("setMapCenterFromMarker", marker);
428                 var geo_ui = {geoJson: this._shapeCreator("geoJson"), geocoderCallbackData: ui};
429                 this._trigger('geocodedpoint', {}, geo_ui);
430               }
431             }
432           }
433         }, this);
434         var autocompleteMenu = $('<span/>')
435           .addClass("ui-simplicity-map-shape-creator-ui-geocoder-menu")
436           .appendTo($('body'));
437         var simplicityGeocoder = this._shapeCreator("getMappingProviderWidgetNames").geocoder;
438         var geocoder = (this.options.geocoderElement === '' ?
439           this._geocoderInput : $(this.options.geocoderElement))[simplicityGeocoder]();
440         this._geocoderInput
441           .autocomplete({
442             source: geocoder[simplicityGeocoder]("autocompleteSource"),
443             select: selectCallback,
444             change: selectCallback,
445             appendTo: autocompleteMenu
446           });
447         if (this._geocoderInput.watermark) {
448           this._geocoderInput.watermark(this._geocoderInput.attr("placeholder"));
449         }
450       }
451     },
452     /**
453      * Helper method that enables the Ui for editing shapes
454      * @name $.ui.simplicityMapShapeCreatorUi._enable

458     _enable: function () {
459       if ($(this.options.searchElement).data("simplicityDiscoverySearch")) {
460         $(this.options.searchElement).simplicityDiscoverySearch("option", "searchOnStateChange", false);
461       }
462       this._mapclicklabel
463         .prop('checked', true)
464         .attr("title", "Stop drawing.")
465         .find('span')
466         .text('Stop');
467       if (!this._settings.helpHidden) {
468         this._helpMsg.appendTo(this.element);
469         this._helpMsg.fadeIn('slow');
470       }
471       this._radiusInput.prop("disabled", false);
472       this.element.find(".linkerBtn").fadeIn();
473       this._geocoderInput.attr("disabled", "true").addClass("disabled");
474       this._helpButton[this._settings.helpHidden ? "fadeIn" : "fadeOut"]();
475     },
476     /**
477      * Helper method that disables (stops) editing
478      * @name $.ui.simplicityMapShapeCreatorUi._disable
479      * @function
480      * @private
481      * @param {boolean} [forceChange] Determines if editing should be force stopped on the shape. Defaults to <code>true</code>

484       if ($(this.options.searchElement).data("simplicityDiscoverySearch")) {
485         $(this.options.searchElement)
486           .simplicityDiscoverySearch("option", "searchOnStateChange", true)
487           .simplicityDiscoverySearch("search");
488       }
489       this._mapclicklabel
490         .prop('checked', false)
491         .attr("title", "Start drawing a shape to search on the map.")
492         .find('span')
493         .text('Draw');
494       this._helpMsg.fadeOut('slow', $.proxy(function () {
495         this._helpMsg.detach();
496       }, this));
497       this.element.find(".linkerBtn").fadeOut();
498       this._geocoderInput.removeAttr("disabled").removeClass("disabled");
499       this._helpButton.fadeOut();
500     },

516     /**
517      * Saves UI state to a cookie
518      * @name $.ui.simplicityMapShapeCreatorUi._saveSettings
519      * @function
520      * @private
521      * @param {String} cookiename The name of the cookie in which to store the state.
522      */
523     _saveSettings: function () {
524       var cookieName = this._settings_COOKIE;
525       var expiresDays = 6;
526       var exdate = new Date();
527       exdate.setDate(exdate.getDate() + expiresDays);
528       var persistValue = {
529         h: this._settings.helpHidden ? 1 : 0
530       };
531       document.cookie = cookieName + "=" + escape(JSON.stringify(persistValue)) + "; expires=" + exdate.toUTCString();
532     },
533     /**
534      * Sets the UI state from a cookie
535      * @name $.ui.simplicityMapShapeCreatorUi._getSettings
536      * @function
537      * @private
538      * @param {String} cookiename The name of the cookie from which to retrieve the state.
539      */
540     _getSettings: function () {
541       var key, val, p, settings, cookies = document.cookie.split(";");
542       var cookieName = this._settings_COOKIE;
543       $.each(cookies, $.proxy(function (i, cookie) {
544         p = cookies[i].indexOf("=");
545         key = cookies[i].substr(0, p);
546         val = cookies[i].substr(p + 1);
547         key = key.replace(/^\s+|\s+$/g, "");
548         if (key === cookieName) {
549           settings = JSON.parse(unescape(val));
550           if ('undefined' !== typeof settings.h) {
551             this._settings.helpHidden = settings.h === 1;
552           }
553         }
554       }, this));
555     },
556     destroy: function () {
557       this._helpMsg.remove();
558       this.element.html(this.element.data('previousHTML'));
559       this.element.removeData('previousHTML');
560       $.ui.simplicityWidget.prototype.destroy.apply(this, arguments);
561     }
